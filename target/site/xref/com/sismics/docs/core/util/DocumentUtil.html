1   package com.sismics.docs.core.util;
2   
3   import com.sismics.docs.core.constant.AclType;
4   import com.sismics.docs.core.constant.PermType;
5   import com.sismics.docs.core.dao.AclDao;
6   import com.sismics.docs.core.dao.DocumentDao;
7   import com.sismics.docs.core.model.jpa.Acl;
8   import com.sismics.docs.core.model.jpa.Document;
9   
10  /**
11   * Document utilities.
12   *
13   * @author bgamard
14   */
15  public class DocumentUtil {
16      /**
17       * Create a document and add the base ACLs.
18       *
19       * @param document Document
20       * @param userId User creating the document
21       * @return Created document
22       */
23      public static Document createDocument(Document document, String userId) {
24          DocumentDao documentDao = new DocumentDao();
25          String documentId = documentDao.create(document, userId);
26  
27          // Create read ACL
28          AclDao aclDao = new AclDao();
29          Acl acl = new Acl();
30          acl.setPerm(PermType.READ);
31          acl.setType(AclType.USER);
32          acl.setSourceId(documentId);
33          acl.setTargetId(userId);
34          aclDao.create(acl, userId);
35  
36          // Create write ACL
37          acl = new Acl();
38          acl.setPerm(PermType.WRITE);
39          acl.setType(AclType.USER);
40          acl.setSourceId(documentId);
41          acl.setTargetId(userId);
42          aclDao.create(acl, userId);
43  
44          return document;
45      }
46  }
