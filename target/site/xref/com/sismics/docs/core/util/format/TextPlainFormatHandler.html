22  public class TextPlainFormatHandler implements FormatHandler {
23      @Override
24      public boolean accept(String mimeType) {
25          return mimeType.equals(MimeType.TEXT_CSV) || mimeType.equals(MimeType.TEXT_PLAIN);
26      }
27  
28      @Override
29      public BufferedImage generateThumbnail(Path file) throws Exception {
30          Document output = new Document(PageSize.A4, 40, 40, 40, 40);
31          Path tempFile = AppContext.getInstance().getFileService().createTemporaryFile();
32          OutputStream pdfOutputStream = Files.newOutputStream(tempFile);
33          PdfWriter.getInstance(output, pdfOutputStream);
34  
35          output.open();
36          String content = Files.readString(file, StandardCharsets.UTF_8);
37          Font font = FontFactory.getFont("LiberationMono-Regular");
38          Paragraph paragraph = new Paragraph(content, font);
39          paragraph.setAlignment(Element.ALIGN_LEFT);
40          output.add(paragraph);
41          output.close();
42  
43          // Use the PDF format handler
44          return new PdfFormatHandler().generateThumbnail(tempFile);
45      }
46  
47      @Override
48      public String extractContent(String language, Path file) throws Exception {
49          return Files.readString(file, StandardCharsets.UTF_8);
50      }
