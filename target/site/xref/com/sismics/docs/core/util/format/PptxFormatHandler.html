24  /**
25   * PPTX format handler.
26   *
27   * @author bgamard
28   */
29  public class PptxFormatHandler implements FormatHandler {
30      /**
31       * Cached PPTX loaded file.
32       */
33      private XMLSlideShow slideShow;
34  
35      @Override
36      public boolean accept(String mimeType) {
37          return MimeType.OFFICE_PRESENTATION.equals(mimeType);
38      }
39  
40      @Override
41      public BufferedImage generateThumbnail(Path file) throws Exception {
42          XMLSlideShow pptx = loadPPtxFile(file);
43          if (pptx.getSlides().size() > 0) {
44              return generateImageFromSlide(pptx, 0);
45          }
46  
47          return null;
48      }
49  
50      @Override
51      public String extractContent(String language, Path file) throws Exception {
52          XMLSlideShow pptx = loadPPtxFile(file);
53          return new SlideShowExtractor<>(pptx).getText();
54      }
55  
56      @Override
57      public void appendToPdf(Path file, PDDocument doc, boolean fitImageToPage, int margin, MemoryUsageSetting memUsageSettings, Closer closer) throws Exception {
58          XMLSlideShow pptx = loadPPtxFile(file);
59          List<XSLFSlide> slides = pptx.getSlides();
60          Dimension pgsize = pptx.getPageSize();
61          for (int slideIndex = 0; slideIndex < slides.size(); slideIndex++) {
62              // One PDF page per slide
63              PDPage page = new PDPage(new PDRectangle(pgsize.width, pgsize.height));
64              try (PDPageContentStream contentStream = new PDPageContentStream(doc, page)) {
65                  BufferedImage bim = generateImageFromSlide(pptx, slideIndex);
66                  PDImageXObject pdImage = LosslessFactory.createFromImage(doc, bim);
67                  contentStream.drawImage(pdImage, 0, page.getMediaBox().getHeight() - pdImage.getHeight());
68              }
69              doc.addPage(page);
70          }
71      }
72  
73      private XMLSlideShow loadPPtxFile(Path file) throws Exception {
74          if (slideShow == null) {
75              try (InputStream inputStream = Files.newInputStream(file)) {
76                  slideShow = new XMLSlideShow(inputStream);
77              }
78          }
79          return slideShow;
80      }
81  
82      /**
83       * Generate an image from a PPTX slide.
84       *
85       * @param pptx PPTX
86       * @param slideIndex Slide index
87       * @return Image
88       */
89      private BufferedImage generateImageFromSlide(XMLSlideShow pptx, int slideIndex) {
90          Dimension pgsize = pptx.getPageSize();
91          BufferedImage img = new BufferedImage(pgsize.width, pgsize.height,BufferedImage.TYPE_INT_RGB);
92          Graphics2D graphics = img.createGraphics();
93          graphics.setPaint(Color.white);
94          graphics.fill(new Rectangle2D.Float(0, 0, pgsize.width, pgsize.height));
95          pptx.getSlides().get(slideIndex).draw(graphics);
96          return img;
97      }
98  }
