54      /**
55       * Optical character recognition on an image.
56       *
57       * @param language Language to OCR
58       * @param image Buffered image
59       * @return Content extracted
60       * @throws Exception e
61       */
62      public static String ocrFile(String language, BufferedImage image) throws Exception {
63          // Upscale, grayscale and deskew the image
64          BufferedImage resizedImage = Scalr.resize(image, Scalr.Method.AUTOMATIC, Scalr.Mode.AUTOMATIC, 3500, Scalr.OP_ANTIALIAS, Scalr.OP_GRAYSCALE);
65          image.flush();
66          ImageDeskew imageDeskew = new ImageDeskew(resizedImage);
67          BufferedImage deskewedImage = Scalr.rotate(resizedImage, - imageDeskew.getSkewAngle(), Scalr.OP_ANTIALIAS, Scalr.OP_GRAYSCALE);
68          resizedImage.flush();
69          Path tmpFile = AppContext.getInstance().getFileService().createTemporaryFile();
70          ImageIO.write(deskewedImage, "tiff", tmpFile.toFile());
71  
72          List<String> result = Lists.newLinkedList(Arrays.asList("tesseract", tmpFile.toAbsolutePath().toString(), "stdout", "-l", language));
73          ProcessBuilder pb = new ProcessBuilder(result);
74          Process process = pb.start();
75  
76          // Consume the process error stream
77          final String commandName = pb.command().get(0);
78          new InputStreamReaderThread(process.getErrorStream(), commandName).start();
79  
80          // Consume the data as text
81          try (InputStream is = process.getInputStream()) {
82              return CharStreams.toString(new InputStreamReader(is, StandardCharsets.UTF_8));
83          }
84      }

86      /**
87       * Remove a file from the storage filesystem.
88       * 
89       * @param fileId ID of file to delete
90       */
91      public static void delete(String fileId) throws IOException {
92          Path storedFile = DirectoryUtil.getStorageDirectory().resolve(fileId);
93          Path webFile = DirectoryUtil.getStorageDirectory().resolve(fileId + "_web");
94          Path thumbnailFile = DirectoryUtil.getStorageDirectory().resolve(fileId + "_thumb");
95          
96          if (Files.exists(storedFile)) {
97              Files.delete(storedFile);
98          }
99          if (Files.exists(webFile)) {
100             Files.delete(webFile);
101         }
102         if (Files.exists(thumbnailFile)) {
103             Files.delete(thumbnailFile);
104         }
105     }

107     /**
108      * Create a new file.
109      *
110      * @param name File name, can be null
111      * @param previousFileId ID of the previous version of the file, if the new file is a new version
112      * @param unencryptedFile Path to the unencrypted file
113      * @param fileSize File size
114      * @param language File language, can be null if associated to no document
115      * @param userId User ID creating the file
116      * @param documentId Associated document ID or null if no document
117      * @return File ID
118      * @throws Exception e
119      */
120     public static String createFile(String name, String previousFileId, Path unencryptedFile, long fileSize, String language, String userId, String documentId) throws Exception {
121         // Validate mime type
122         String mimeType;
123         try {
124             mimeType = MimeTypeUtil.guessMimeType(unencryptedFile, name);
125         } catch (IOException e) {
126             throw new IOException("ErrorGuessMime", e);
127         }
128 
129         // Validate user quota
130         UserDao userDao = new UserDao();
131         User user = userDao.getById(userId);
132         if (user.getStorageCurrent() + fileSize > user.getStorageQuota()) {
133             throw new IOException("QuotaReached");
134         }
135 
136         // Validate global quota
137         String globalStorageQuotaStr = System.getenv(Constants.GLOBAL_QUOTA_ENV);
138         if (!Strings.isNullOrEmpty(globalStorageQuotaStr)) {
139             long globalStorageQuota = Long.parseLong(globalStorageQuotaStr);
140             long globalStorageCurrent = userDao.getGlobalStorageCurrent();
141             if (globalStorageCurrent + fileSize > globalStorageQuota) {
142                 throw new IOException("QuotaReached");
143             }
144         }
145 
146         // Prepare the file
147         File file = new File();
148         file.setOrder(0);
149         file.setVersion(0);
150         file.setLatestVersion(true);
151         file.setDocumentId(documentId);
152         file.setName(StringUtils.abbreviate(name, 200));
153         file.setMimeType(mimeType);
154         file.setUserId(userId);
155         file.setSize(fileSize);
156 
157         // Get files of this document
158         FileDao fileDao = new FileDao();
159         if (documentId != null) {
160             if (previousFileId == null) {
161                 // It's not a new version, so put it in last order
162                 file.setOrder(fileDao.getByDocumentId(userId, documentId).size());
163             } else {
164                 // It's a new version, update the previous version
165                 File previousFile = fileDao.getActiveById(previousFileId);
166                 if (previousFile == null || !previousFile.getDocumentId().equals(documentId)) {
167                     throw new IOException("Previous version mismatch");
168                 }
169 
170                 if (previousFile.getVersionId() == null) {
171                     previousFile.setVersionId(UUID.randomUUID().toString());
172                 }
173 
174                 // Copy the previous file metadata
175                 file.setOrder(previousFile.getOrder());
176                 file.setVersionId(previousFile.getVersionId());
177                 file.setVersion(previousFile.getVersion() + 1);
178 
179                 // Update the previous file
180                 previousFile.setLatestVersion(false);
181                 fileDao.update(previousFile);
182             }
183         }
184 
185         // Create the file
186         String fileId = fileDao.create(file, userId);
187 
188         // Save the file
189         Cipher cipher = EncryptionUtil.getEncryptionCipher(user.getPrivateKey());
190         Path path = DirectoryUtil.getStorageDirectory().resolve(file.getId());
191         try (InputStream inputStream = Files.newInputStream(unencryptedFile)) {
192             Files.copy(new CipherInputStream(inputStream, cipher), path);
193         }
194 
195         // Update the user quota
196         user.setStorageCurrent(user.getStorageCurrent() + fileSize);
197         userDao.updateQuota(user);
198 
199         // Raise a new file created event and document updated event if we have a document
200         startProcessingFile(fileId);
201         FileCreatedAsyncEvent fileCreatedAsyncEvent = new FileCreatedAsyncEvent();
202         fileCreatedAsyncEvent.setUserId(userId);
203         fileCreatedAsyncEvent.setLanguage(language);
204         fileCreatedAsyncEvent.setFileId(file.getId());
205         fileCreatedAsyncEvent.setUnencryptedFile(unencryptedFile);
206         ThreadLocalContext.get().addAsyncEvent(fileCreatedAsyncEvent);
207 
208         if (documentId != null) {
209             DocumentUpdatedAsyncEvent documentUpdatedAsyncEvent = new DocumentUpdatedAsyncEvent();
210             documentUpdatedAsyncEvent.setUserId(userId);
211             documentUpdatedAsyncEvent.setDocumentId(documentId);
212             ThreadLocalContext.get().addAsyncEvent(documentUpdatedAsyncEvent);
213         }
214 
215         return fileId;
216     }

248     /**
249      * Get the size of a file on disk.
250      *
251      * @param fileId the file id
252      * @param user   the file owner
253      * @return the size or -1 if something went wrong
254      */
255     public static long getFileSize(String fileId, User user) {
256         // To get the size we copy the decrypted content into a null output stream
257         // and count the copied byte size.
258         Path storedFile = DirectoryUtil.getStorageDirectory().resolve(fileId);
259         if (! Files.exists(storedFile)) {
260             log.debug("File does not exist " + fileId);
261             return File.UNKNOWN_SIZE;
262         }
263         try (InputStream fileInputStream = Files.newInputStream(storedFile);
264              InputStream inputStream = EncryptionUtil.decryptInputStream(fileInputStream, user.getPrivateKey());
265              CountingInputStream countingInputStream = new CountingInputStream(inputStream);
266         ) {
267             IOUtils.copy(countingInputStream, NullOutputStream.NULL_OUTPUT_STREAM);
268             return countingInputStream.getByteCount();
269         } catch (Exception e) {
270             log.debug("Can't find size of file " + fileId, e);
271             return File.UNKNOWN_SIZE;
272         }
