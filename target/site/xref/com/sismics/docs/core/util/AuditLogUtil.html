1   package com.sismics.docs.core.util;

3   import com.sismics.docs.core.constant.AuditLogType;
4   import com.sismics.docs.core.dao.AuditLogDao;
5   import com.sismics.docs.core.model.jpa.AuditLog;
6   import com.sismics.docs.core.model.jpa.Loggable;
7   import com.sismics.util.context.ThreadLocalContext;
8   
9   import jakarta.persistence.EntityManager;
10  
11  /**
12   * Audit log utilities.
13   * 
14   * @author bgamard
15   */
16  public class AuditLogUtil {
17      /**
18       * Create an audit log.
19       * 
20       * @param loggable Loggable
21       * @param type Audit log type
22       * @param userId User ID
23       */
24      public static void create(Loggable loggable, AuditLogType type, String userId) {
25          if (userId == null) {
26              userId = "admin";
27          }
28  
29          // Get the entity ID
30          EntityManager em = ThreadLocalContext.get().getEntityManager();
31          String entityId = (String) em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier(loggable);
32          
33          // Create the audit log
34          AuditLogDao auditLogDao = new AuditLogDao();
35          AuditLog auditLog = new AuditLog();
36          auditLog.setUserId(userId);
37          auditLog.setEntityId(entityId);
38          auditLog.setEntityClass(loggable.getClass().getSimpleName());
39          auditLog.setType(type);
40          auditLog.setMessage(loggable.toMessage());
41          auditLogDao.create(auditLog);
42      }
43  }
