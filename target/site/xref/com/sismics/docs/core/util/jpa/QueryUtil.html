3   import java.util.Map.Entry;

5   import jakarta.persistence.EntityManager;
6   import jakarta.persistence.Query;

8   import com.sismics.util.context.ThreadLocalContext;
9   
10  /**
11   * Query utilities.
12   *
13   * @author jtremeaux 
14   */
15  public class QueryUtil {
16  
17      /**
18       * Creates a native query from the query parameters.
19       * 
20       * @param queryParam Query parameters
21       * @return Native query
22       */
23      public static Query getNativeQuery(QueryParam queryParam) {
24          EntityManager em = ThreadLocalContext.get().getEntityManager();
25          Query query = em.createNativeQuery(queryParam.getQueryString());
26          for (Entry<String, Object> entry : queryParam.getParameterMap().entrySet()) {
27              query.setParameter(entry.getKey(), entry.getValue());
28          }
29          return query;
30      }
31      
32      /**
33       * Returns sorted query parameters.
34       * 
35       * @param queryParam Query parameters
36       * @param sortCriteria Sort criteria
37       * @return Sorted query parameters
38       */
39      public static QueryParam getSortedQueryParam(QueryParam queryParam, SortCriteria sortCriteria) {
40          StringBuilder sb = new StringBuilder(queryParam.getQueryString());
41          if (sortCriteria != null) {
42              sb.append(" order by c");
43              sb.append(sortCriteria.getColumn());
44              sb.append(sortCriteria.isAsc() ? " asc" : " desc");
45          }
46          
47          return new QueryParam(sb.toString(), queryParam.getParameterMap());
48      }
49  }
