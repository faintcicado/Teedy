42      /**
43       * Convert a document and its files to a merged PDF file.
44       * 
45       * @param documentDto Document DTO
46       * @param fileList List of files
47       * @param fitImageToPage Fit images to the page
48       * @param metadata Add a page with metadata
49       * @param margin Margins in millimeters
50       * @param outputStream Output stream to write to, will be closed
51       */
52      public static void convertToPdf(DocumentDto documentDto, List<File> fileList,
53              boolean fitImageToPage, boolean metadata, int margin, OutputStream outputStream) throws Exception {
54          // Setup PDFBox
55          Closer closer = Closer.create();
56          MemoryUsageSetting memUsageSettings = MemoryUsageSetting.setupMixed(1000000); // 1MB max memory usage
57          memUsageSettings.setTempDir(new java.io.File(System.getProperty("java.io.tmpdir"))); // To OS temp
58  
59          // Create a blank PDF
60          try (PDDocument doc = new PDDocument(memUsageSettings)) {
61              // Add metadata
62              if (metadata) {
63                  PDPage page = new PDPage();
64                  doc.addPage(page);
65                  try (PdfPage pdfPage = new PdfPage(doc, page, margin * Constants.MM_PER_INCH, DocsPDType1Font.HELVETICA, 12)) {
66                      SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
67                      pdfPage.addText(documentDto.getTitle(), true, DocsPDType1Font.HELVETICA_BOLD, 16)
68                          .newLine()
69                          .addText("Created by " + documentDto.getCreator()
70                              + " on " + dateFormat.format(new Date(documentDto.getCreateTimestamp())), true)
71                          .newLine()
72                          .addText(documentDto.getDescription())
73                          .newLine();
74                      if (!Strings.isNullOrEmpty(documentDto.getSubject())) {
75                          pdfPage.addText("Subject: " + documentDto.getSubject());
76                      }
77                      if (!Strings.isNullOrEmpty(documentDto.getIdentifier())) {
78                          pdfPage.addText("Identifier: " + documentDto.getIdentifier());
79                      }
80                      if (!Strings.isNullOrEmpty(documentDto.getPublisher())) {
81                          pdfPage.addText("Publisher: " + documentDto.getPublisher());
82                      }
83                      if (!Strings.isNullOrEmpty(documentDto.getFormat())) {
84                          pdfPage.addText("Format: " + documentDto.getFormat());
85                      }
86                      if (!Strings.isNullOrEmpty(documentDto.getSource())) {
87                          pdfPage.addText("Source: " + documentDto.getSource());
88                      }
89                      if (!Strings.isNullOrEmpty(documentDto.getType())) {
90                          pdfPage.addText("Type: " + documentDto.getType());
91                      }
92                      if (!Strings.isNullOrEmpty(documentDto.getCoverage())) {
93                          pdfPage.addText("Coverage: " + documentDto.getCoverage());
94                      }
95                      if (!Strings.isNullOrEmpty(documentDto.getRights())) {
96                          pdfPage.addText("Rights: " + documentDto.getRights());
97                      }
98                      pdfPage.addText("Language: " + documentDto.getLanguage())
99                          .newLine()
100                         .addText("Files in this document : " + fileList.size(), false, DocsPDType1Font.HELVETICA_BOLD, 12);
101                 }
102             }
103             
104             // Add files
105             for (File file : fileList) {
106                 Path storedFile = DirectoryUtil.getStorageDirectory().resolve(file.getId());
107 
108                 // Decrypt the file to a temporary file
109                 Path unencryptedFile = EncryptionUtil.decryptFile(storedFile, file.getPrivateKey());
110                 FormatHandler formatHandler = FormatHandlerUtil.find(file.getMimeType());
111                 if (formatHandler != null) {
112                     formatHandler.appendToPdf(unencryptedFile, doc, fitImageToPage, margin, memUsageSettings, closer);
113                 }
114             }
115             
116             doc.save(outputStream); // Write to the output stream
117             closer.close(); // Close all remaining opened PDF
118         }
119     }

121     /**
122      * Register fonts.
123      */
124     public static void registerFonts() {
125         URL url = Resources.getResource("fonts/LiberationMono-Regular.ttf");
126         try (InputStream is = url.openStream()) {
127             Path file = Files.createTempFile("sismics_docs_font_mono", ".ttf");
128             try (OutputStream os = Files.newOutputStream(file)) {
129                 ByteStreams.copy(is, os);
130             }
131             FontFactory.register(file.toAbsolutePath().toString(), "LiberationMono-Regular");
132             FontFactory.registerDirectories();
133         } catch (IOException e) {
134             log.error("Error loading font", e);
135         }
136     }
