35      /**
36       * Update routing ACLs according to the current route step.
37       *
38       * @param documentId Document ID
39       * @param currentStep Current route step
40       * @param previousStep Previous route step
41       * @param userId User ID
42       */
43      public static void updateAcl(String documentId, RouteStepDto currentStep, RouteStepDto previousStep, String userId) {
44          AclDao aclDao = new AclDao();
45  
46          if (previousStep != null) {
47              // Remove the previous ACL
48              aclDao.delete(documentId, PermType.READ, previousStep.getTargetId(), userId, AclType.ROUTING);
49          }
50  
51          if (currentStep != null) {
52              // Create a temporary READ ACL
53              Acl acl = new Acl();
54              acl.setPerm(PermType.READ);
55              acl.setType(AclType.ROUTING);
56              acl.setSourceId(documentId);
57              acl.setTargetId(currentStep.getTargetId());
58              aclDao.create(acl, userId);
59          }
60  
61          // Raise a document updated event
62          DocumentUpdatedAsyncEvent event = new DocumentUpdatedAsyncEvent();
63          event.setUserId(userId);
64          event.setDocumentId(documentId);
65          ThreadLocalContext.get().addAsyncEvent(event);
66      }

68      /**
69       * Send an email when a route step is validated.
70       *
71       * @param documentId Document ID
72       * @param routeStepDto Route step DTO
73       */
74      public static void sendRouteStepEmail(String documentId, RouteStepDto routeStepDto) {
75          DocumentDao documentDao = new DocumentDao();
76          Document document = documentDao.getById(documentId);
77  
78          List<UserDto> userDtoList = Lists.newArrayList();
79                  UserDao userDao = new UserDao();
80          switch (AclTargetType.valueOf(routeStepDto.getTargetType())) {
81              case USER:
82                  userDtoList.addAll(userDao.findByCriteria(new UserCriteria().setUserId(routeStepDto.getTargetId()), null));
83                  break;
84              case GROUP:
85                  userDtoList.addAll(userDao.findByCriteria(new UserCriteria().setGroupId(routeStepDto.getTargetId()), null));
86                  break;
87          }
88  
89          // Fire route step validate events
90          for (UserDto userDto : userDtoList) {
91              RouteStepValidateEvent routeStepValidateEvent = new RouteStepValidateEvent();
92              routeStepValidateEvent.setUser(userDto);
93              routeStepValidateEvent.setDocument(document);
94              AppContext.getInstance().getMailEventBus().post(routeStepValidateEvent);
95          }
96      }

98      /**
99       * Find the first route model name matching a target type and name.
100      *
101      * @param targetType Target type
102      * @param targetName Target name
103      * @return Route model name or null if none is matching
104      */
105     public static String findRouteModelNameByTargetName(AclTargetType targetType, String targetName) {
106         RouteModelDao routeModelDao = new RouteModelDao();
107         List<RouteModel> routeModelList = routeModelDao.findAll();
108         for (RouteModel routeModel : routeModelList) {
109             try (JsonReader reader = Json.createReader(new StringReader(routeModel.getSteps()))) {
110                 JsonArray stepsJson = reader.readArray();
111                 for (int order = 0; order < stepsJson.size(); order++) {
112                     JsonObject step = stepsJson.getJsonObject(order);
113                     JsonObject target = step.getJsonObject("target");
114                     AclTargetType routeTargetType = AclTargetType.valueOf(target.getString("type"));
115                     String routeTargetName = target.getString("name");
116                     if (targetType == routeTargetType && targetName.equals(routeTargetName)) {
117                         return routeModel.getName();
118                     }
119                 }
120             }
121         }
122         return null;
123     }
