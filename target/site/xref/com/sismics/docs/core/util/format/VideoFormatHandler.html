20  /**
21   * Video format handler.
22   *
23   * @author bgamard
24   */
25  public class VideoFormatHandler implements FormatHandler {
26      @Override
27      public boolean accept(String mimeType) {
28          return mimeType.equals(MimeType.VIDEO_MP4) || mimeType.equals(MimeType.VIDEO_WEBM);
29      }
30  
31      @Override
32      public BufferedImage generateThumbnail(Path file) throws IOException {
33          List<String> result = Lists.newLinkedList(Arrays.asList("ffmpeg", "-i"));
34          result.add(file.toAbsolutePath().toString());
35          result.addAll(Arrays.asList("-vf", "thumbnail", "-frames:v", "1", "-f", "mjpeg", "-"));
36          ProcessBuilder pb = new ProcessBuilder(result);
37          Process process = pb.start();
38  
39          // Consume the process error stream
40          final String commandName = pb.command().get(0);
41          new InputStreamReaderThread(process.getErrorStream(), commandName).start();
42  
43          // Consume the data as an image
44          try (InputStream is = process.getInputStream()) {
45              return ImageIO.read(is);
46          }
47      }
48  
49      @Override
50      public String extractContent(String language, Path file) {
51          List<String> result = Lists.newLinkedList();
52          result.add("mediainfo");
53          result.add(file.toAbsolutePath().toString());
54          ProcessBuilder pb = new ProcessBuilder(result);
55          Process process;
56          try {
57              process = pb.start();
58          } catch (IOException e) {
59              return null;
60          }
61  
62          // Consume the process error stream
63          final String commandName = pb.command().get(0);
64          new InputStreamReaderThread(process.getErrorStream(), commandName).start();
65  
66          // Consume the data as a string
67          try (InputStream is = process.getInputStream()) {
68              return new String(ByteStreams.toByteArray(is), StandardCharsets.UTF_8);
69          } catch (Exception e) {
70              return null;
71          }
72      }
73  
74      @Override
75      public void appendToPdf(Path file, PDDocument doc, boolean fitImageToPage, int margin, MemoryUsageSetting memUsageSettings, Closer closer) {
76          // Video cannot be appended to PDF files
77      }
78  }
