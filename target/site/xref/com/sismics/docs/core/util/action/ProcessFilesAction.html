20  /**
21   * Action to process all files.
22   *
23   * @author bgamard
24   */
25  public class ProcessFilesAction implements Action {
26      /**
27       * Logger.
28       */
29      private static final Logger log = LoggerFactory.getLogger(ProcessFilesAction.class);
30  
31      @Override
32      public void execute(DocumentDto documentDto, JsonObject action) {
33          FileDao fileDao = new FileDao();
34          List<File> fileList = fileDao.getByDocumentId(null, documentDto.getId());
35  
36          try {
37              for (File file : fileList) {
38                  // Get the creating user
39                  UserDao userDao = new UserDao();
40                  User user = userDao.getById(file.getUserId());
41  
42                  // Decrypt the file
43                  Path storedFile = DirectoryUtil.getStorageDirectory().resolve(file.getId());
44                  Path unencryptedFile = EncryptionUtil.decryptFile(storedFile, user.getPrivateKey());
45  
46                  // Start the asynchronous processing
47                  FileUtil.startProcessingFile(file.getId());
48                  FileUpdatedAsyncEvent event = new FileUpdatedAsyncEvent();
49                  event.setUserId("admin");
50                  event.setLanguage(documentDto.getLanguage());
51                  event.setFileId(file.getId());
52                  event.setUnencryptedFile(unencryptedFile);
53                  ThreadLocalContext.get().addAsyncEvent(event);
54              }
55          } catch (Exception e) {
56              log.error("Error processing a file", e);
57          }
58      }
59  
60      @Override
61      public void validate(JsonObject action) {
62          // No parameter, so always OK
63      }
64  }
