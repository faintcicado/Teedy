9   import jakarta.json.JsonObject;
10  import java.util.List;
11  import java.util.Set;
12  
13  /**
14   * Action to add a tag.
15   *
16   * @author bgamard
17   */
18  public class AddTagAction extends TagAction {
19      @Override
20      public void execute(DocumentDto documentDto, JsonObject action) {
21          if (action.getString("tag") == null) {
22              return;
23          }
24  
25          TagDao tagDao = new TagDao();
26          List<TagDto> tagAddDtoList = tagDao.findByCriteria(new TagCriteria().setId(action.getString("tag")), null);
27          if (tagAddDtoList.isEmpty()) {
28              // The tag has been deleted since the route model creation
29              return;
30          }
31  
32          List<TagDto> tagDtoList = tagDao.findByCriteria(new TagCriteria().setDocumentId(documentDto.getId()), null);
33          Set<String> tagIdSet = Sets.newHashSet(tagAddDtoList.get(0).getId());
34          for (TagDto tagDto : tagDtoList) {
35              tagIdSet.add(tagDto.getId());
36          }
37  
38          tagDao.updateTagList(documentDto.getId(), tagIdSet);
39      }
40  }
