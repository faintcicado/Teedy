24      /**
25       * Creates a new route.
26       *
27       * @param route Route
28       * @param userId User ID
29       * @return New ID
30       */
31      public String create(Route route, String userId) {
32          // Create the UUID
33          route.setId(UUID.randomUUID().toString());
34  
35          // Create the route
36          EntityManager em = ThreadLocalContext.get().getEntityManager();
37          route.setCreateDate(new Date());
38          em.persist(route);
39  
40          // Create audit log
41          AuditLogUtil.create(route, AuditLogType.CREATE, userId);
42  
43          return route.getId();
44      }
45  
46      /**
47       * Returns the list of all routes.
48       *
49       * @param criteria Search criteria
50       * @param sortCriteria Sort criteria
51       * @return List of routes
52       */
53      public List<RouteDto> findByCriteria(RouteCriteria criteria, SortCriteria sortCriteria) {
54          Map<String, Object> parameterMap = new HashMap<String, Object>();
55          List<String> criteriaList = new ArrayList<>();
56  
57          StringBuilder sb = new StringBuilder("select r.RTE_ID_C c0, r.RTE_NAME_C c1, r.RTE_CREATEDATE_D c2");
58          sb.append(" from T_ROUTE r ");
59  
60          // Add search criterias
61          if (criteria.getDocumentId() != null) {
62              criteriaList.add("r.RTE_IDDOCUMENT_C = :documentId");
63              parameterMap.put("documentId", criteria.getDocumentId());
64          }
65          criteriaList.add("r.RTE_DELETEDATE_D is null");
66  
67          sb.append(" where ");
68          sb.append(Joiner.on(" and ").join(criteriaList));
69  
70          // Perform the search
71          QueryParam queryParam = QueryUtil.getSortedQueryParam(new QueryParam(sb.toString(), parameterMap), sortCriteria);
72          @SuppressWarnings("unchecked")
73          List<Object[]> l = QueryUtil.getNativeQuery(queryParam).getResultList();
74  
75          // Assemble results
76          List<RouteDto> dtoList = new ArrayList<>();
77          for (Object[] o : l) {
78              int i = 0;
79              RouteDto dto = new RouteDto();
80              dto.setId((String) o[i++]);
81              dto.setName((String) o[i++]);
82              dto.setCreateTimestamp(((Timestamp) o[i++]).getTime());
83              dtoList.add(dto);
84          }
85          return dtoList;
86      }

88      /**
89       * Deletes a route and the associated steps.
90       *
91       * @param routeId Route ID
92       * @param userId User ID
93       */
94      public void deleteRoute(String routeId, String userId) {
95          EntityManager em = ThreadLocalContext.get().getEntityManager();
96  
97          // Create audit log
98          Route route = em.find(Route.class, routeId);
99          AuditLogUtil.create(route, AuditLogType.DELETE, userId);
100 
101         em.createNativeQuery("update T_ROUTE_STEP rs set RTP_DELETEDATE_D = :dateNow where rs.RTP_IDROUTE_C = :routeId and rs.RTP_DELETEDATE_D is null")
102                 .setParameter("routeId", routeId)
103                 .setParameter("dateNow", new Date())
104                 .executeUpdate();
105 
106         em.createNativeQuery("update T_ROUTE r set RTE_DELETEDATE_D = :dateNow where r.RTE_ID_C = :routeId and r.RTE_DELETEDATE_D is null")
107                 .setParameter("routeId", routeId)
108                 .setParameter("dateNow", new Date())
109                 .executeUpdate();
110     }
