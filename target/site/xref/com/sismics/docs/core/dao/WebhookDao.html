24      /**
25       * Returns the list of all webhooks.
26       * 
27       * @param criteria Search criteria
28       * @param sortCriteria Sort criteria
29       * @return List of webhooks
30       */
31      public List<WebhookDto> findByCriteria(WebhookCriteria criteria, SortCriteria sortCriteria) {
32          Map<String, Object> parameterMap = new HashMap<>();
33          List<String> criteriaList = new ArrayList<>();
34          
35          StringBuilder sb = new StringBuilder("select w.WHK_ID_C as c0, w.WHK_EVENT_C as c1, w.WHK_URL_C as c2, w.WHK_CREATEDATE_D as c3 ");
36          sb.append(" from T_WEBHOOK w ");
37  
38          // Add search criterias
39          if (criteria.getEvent() != null) {
40              criteriaList.add("w.WHK_EVENT_C = :event");
41              parameterMap.put("event", criteria.getEvent().name());
42          }
43          criteriaList.add("w.WHK_DELETEDATE_D is null");
44  
45          sb.append(" where ");
46          sb.append(Joiner.on(" and ").join(criteriaList));
47  
48          // Perform the search
49          QueryParam queryParam = QueryUtil.getSortedQueryParam(new QueryParam(sb.toString(), parameterMap), sortCriteria);
50          @SuppressWarnings("unchecked")
51          List<Object[]> l = QueryUtil.getNativeQuery(queryParam).getResultList();
52          
53          // Assemble results
54          List<WebhookDto> webhookDtoList = new ArrayList<>();
55          for (Object[] o : l) {
56              int i = 0;
57              WebhookDto webhookDto = new WebhookDto()
58                  .setId((String) o[i++])
59                  .setEvent((String) o[i++])
60                  .setUrl((String) o[i++])
61                  .setCreateTimestamp(((Timestamp) o[i]).getTime());
62              webhookDtoList.add(webhookDto);
63          }
64          
65          return webhookDtoList;
66      }

68      /**
69       * Returns a webhook by ID.
70       *
71       * @param id Webhook ID
72       * @return Webhook
73       */
74      public Webhook getActiveById(String id) {
75          EntityManager em = ThreadLocalContext.get().getEntityManager();
76          Query q = em.createQuery("select w from Webhook w where w.id = :id and w.deleteDate is null");
77          q.setParameter("id", id);
78          try {
79              return (Webhook) q.getSingleResult();
80          } catch (NoResultException e) {
81              return null;
82          }
83      }

85      /**
86       * Creates a new webhook.
87       *
88       * @param webhook Webhook
89       * @return New ID
90       */
91      public String create(Webhook webhook) {
92          // Create the UUID
93          webhook.setId(UUID.randomUUID().toString());
94  
95          // Create the webhook
96          EntityManager em = ThreadLocalContext.get().getEntityManager();
97          webhook.setCreateDate(new Date());
98          em.persist(webhook);
99  
100         return webhook.getId();
101     }

103     /**
104      * Deletes a webhook.
105      *
106      * @param webhookId Webhook ID
107      */
108     public void delete(String webhookId) {
109         EntityManager em = ThreadLocalContext.get().getEntityManager();
110 
111         // Get the group
112         Query q = em.createQuery("select w from Webhook w where w.id = :id and w.deleteDate is null");
113         q.setParameter("id", webhookId);
114         Webhook webhookDb = (Webhook) q.getSingleResult();
115 
116         // Delete the group
117         Date dateNow = new Date();
118         webhookDb.setDeleteDate(dateNow);
119     }
