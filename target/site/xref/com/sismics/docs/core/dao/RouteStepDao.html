26      /**
27       * Creates a new route step.
28       *
29       * @param routeStep Route step
30       * @return New ID
31       */
32      public String create(RouteStep routeStep) {
33          // Create the UUID
34          routeStep.setId(UUID.randomUUID().toString());
35  
36          // Create the route step
37          EntityManager em = ThreadLocalContext.get().getEntityManager();
38          routeStep.setCreateDate(new Date());
39          em.persist(routeStep);
40  
41          return routeStep.getId();
42      }

44      /**
45       * Get the current route step from a document.
46       *
47       * @param documentId Document ID
48       * @return Current route step
49       */
50      public RouteStepDto getCurrentStep(String documentId) {
51          List<RouteStepDto> routeStepDtoList = findByCriteria(new RouteStepCriteria()
52                  .setDocumentId(documentId)
53                  .setEndDateIsNull(true), new SortCriteria(6, true));
54          if (routeStepDtoList.isEmpty()) {
55              return null;
56          }
57          return routeStepDtoList.get(0);
58      }

60      /**
61       * Returns the list of all route steps.
62       *
63       * @param criteria Search criteria
64       * @param sortCriteria Sort criteria
65       * @return List of route steps
66       */
67      public List<RouteStepDto> findByCriteria(RouteStepCriteria criteria, SortCriteria sortCriteria) {
68          Map<String, Object> parameterMap = new HashMap<>();
69          List<String> criteriaList = new ArrayList<>();
70  
71          StringBuilder sb = new StringBuilder("select rs.RTP_ID_C, rs.RTP_NAME_C c0, rs.RTP_TYPE_C c1, rs.RTP_TRANSITION_C c2, rs.RTP_COMMENT_C c3, rs.RTP_IDTARGET_C c4, u.USE_USERNAME_C as targetUsername, g.GRP_NAME_C, rs.RTP_ENDDATE_D c5, uv.USE_USERNAME_C as validatorUsername, rs.RTP_IDROUTE_C, rs.RTP_TRANSITIONS_C, rs.RTP_ORDER_N c6")
72              .append(" from T_ROUTE_STEP rs ")
73              .append(" join T_ROUTE r on r.RTE_ID_C = rs.RTP_IDROUTE_C ")
74              .append(" left join T_USER uv on uv.USE_ID_C = rs.RTP_IDVALIDATORUSER_C ")
75              .append(" left join T_USER u on u.USE_ID_C = rs.RTP_IDTARGET_C ")
76              .append(" left join T_SHARE s on s.SHA_ID_C = rs.RTP_IDTARGET_C ")
77              .append(" left join T_GROUP g on g.GRP_ID_C = rs.RTP_IDTARGET_C ");
78  
79          // Add search criterias
80          if (criteria.getDocumentId() != null) {
81              criteriaList.add("r.RTE_IDDOCUMENT_C = :documentId");
82              parameterMap.put("documentId", criteria.getDocumentId());
83          }
84          if (criteria.getRouteId() != null) {
85              criteriaList.add("rs.RTP_IDROUTE_C = :routeId");
86              parameterMap.put("routeId", criteria.getRouteId());
87          }
88          if (criteria.getEndDateIsNull() != null) {
89              criteriaList.add("RTP_ENDDATE_D is " + (criteria.getEndDateIsNull() ? "" : "not") + " null");
90          }
91          criteriaList.add("rs.RTP_DELETEDATE_D is null");
92  
93          sb.append(" where ");
94          sb.append(Joiner.on(" and ").join(criteriaList));
95  
96          // Perform the search
97          QueryParam queryParam = QueryUtil.getSortedQueryParam(new QueryParam(sb.toString(), parameterMap), sortCriteria);
98          @SuppressWarnings("unchecked")
99          List<Object[]> l = QueryUtil.getNativeQuery(queryParam).getResultList();
100 
101         // Assemble results
102         List<RouteStepDto> dtoList = new ArrayList<>();
103         for (Object[] o : l) {
104             int i = 0;
105             RouteStepDto dto = new RouteStepDto();
106             dto.setId((String) o[i++]);
107             dto.setName((String) o[i++]);
108             dto.setType(RouteStepType.valueOf((String) o[i++]));
109             dto.setTransition((String) o[i++]);
110             dto.setComment((String) o[i++]);
111             dto.setTargetId((String) o[i++]);
112             String userName = (String) o[i++];
113             String groupName = (String) o[i++];
114             if (userName != null) {
115                 dto.setTargetName(userName);
116                 dto.setTargetType(AclTargetType.USER.name());
117             }
118             if (groupName != null) {
119                 dto.setTargetName(groupName);
120                 dto.setTargetType(AclTargetType.GROUP.name());
121             }
122             Timestamp endDateTimestamp = (Timestamp) o[i++];
123             dto.setEndDateTimestamp(endDateTimestamp == null ? null : endDateTimestamp.getTime());
124             dto.setValidatorUserName((String) o[i++]);
125             dto.setRouteId((String) o[i++]);
126             dto.setTransitions((String) o[i]);
127             dtoList.add(dto);
128         }
129         return dtoList;
130     }

132     /**
133      * End a route step.
134      *
135      * @param id Route step ID
136      * @param transition Transition
137      * @param comment Comment
138      * @param validatorUserId Validator user ID
139      */
140     public void endRouteStep(String id, RouteStepTransition transition, String comment, String validatorUserId) {
141         StringBuilder sb = new StringBuilder("update T_ROUTE_STEP r ");
142         sb.append(" set RTP_ENDDATE_D = :endDate, RTP_TRANSITION_C = :transition, RTP_COMMENT_C = :comment, RTP_IDVALIDATORUSER_C = :validatorUserId ");
143         sb.append(" where r.RTP_ID_C = :id");
144 
145         EntityManager em = ThreadLocalContext.get().getEntityManager();
146         Query q = em.createNativeQuery(sb.toString());
147         q.setParameter("endDate", new Date());
148         q.setParameter("transition", transition.name());
149         q.setParameter("comment", comment);
150         q.setParameter("validatorUserId", validatorUserId);
151         q.setParameter("id", id);
152         q.executeUpdate();
153     }
