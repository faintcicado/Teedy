11  /**
12   * Relation DAO.
13   * 
14   * @author bgamard
15   */
16  public class RelationDao {
17      /**
18       * Get all relations from/to a document.
19       * 
20       * @param documentId Document ID
21       * @return List of relations
22       */
23      @SuppressWarnings("unchecked")
24      public List<RelationDto> getByDocumentId(String documentId) {
25          EntityManager em = ThreadLocalContext.get().getEntityManager();
26          StringBuilder sb = new StringBuilder("select d.DOC_ID_C, d.DOC_TITLE_C, r.REL_IDDOCFROM_C ");
27          sb.append(" from T_RELATION r ");
28          sb.append(" join T_DOCUMENT d on d.DOC_ID_C = r.REL_IDDOCFROM_C and r.REL_IDDOCFROM_C != :documentId or d.DOC_ID_C = r.REL_IDDOCTO_C and r.REL_IDDOCTO_C != :documentId ");
29          sb.append(" where (r.REL_IDDOCFROM_C = :documentId or r.REL_IDDOCTO_C = :documentId) ");
30          sb.append(" and r.REL_DELETEDATE_D is null ");
31          sb.append(" order by d.DOC_TITLE_C ");
32          
33          // Perform the query
34          Query q = em.createNativeQuery(sb.toString());
35          q.setParameter("documentId", documentId);
36          List<Object[]> l = q.getResultList();
37          
38          // Assemble results
39          List<RelationDto> relationDtoList = new ArrayList<>();
40          for (Object[] o : l) {
41              int i = 0;
42              RelationDto relationDto = new RelationDto();
43              relationDto.setId((String) o[i++]);
44              relationDto.setTitle((String) o[i++]);
45              String fromDocId = (String) o[i];
46              relationDto.setSource(documentId.equals(fromDocId));
47              relationDtoList.add(relationDto);
48          }
49          return relationDtoList;
50      }
51      
52      /**
53       * Update relations on a document.
54       * 
55       * @param documentId Document ID
56       * @param documentIdSet Set of document ID
57       */
58      public void updateRelationList(String documentId, Set<String> documentIdSet) {
59          EntityManager em = ThreadLocalContext.get().getEntityManager();
60          
61          // Get current relations from this document
62          Query q = em.createQuery("select r from Relation r where r.fromDocumentId = :documentId and r.deleteDate is null");
63          q.setParameter("documentId", documentId);
64          @SuppressWarnings("unchecked")
65          List<Relation> relationList = q.getResultList();
66          
67          // Deleting relations no longer there
68          for (Relation relation : relationList) {
69              if (!documentIdSet.contains(relation.getToDocumentId())) {
70                  relation.setDeleteDate(new Date());
71              }
72          }
73          
74          // Adding new relations
75          for (String targetDocId : documentIdSet) {
76              boolean found = false;
77              for (Relation relation : relationList) {
78                  if (relation.getToDocumentId().equals(targetDocId)) {
79                      found = true;
80                      break;
81                  }
82              }
83              
84              if (!found) {
85                  Relation relation = new Relation();
86                  relation.setId(UUID.randomUUID().toString());
87                  relation.setFromDocumentId(documentId);
88                  relation.setToDocumentId(targetDocId);
89                  em.persist(relation);
90              }
91          }
92      }
93  }
