24  public class AuditLogDao {
25      /**
26       * Creates a new audit log.
27       * 
28       * @param auditLog Audit log
29       * @return New ID
30       */
31      public String create(AuditLog auditLog) {
32          // Create the UUID
33          auditLog.setId(UUID.randomUUID().toString());
34          
35          // Create the audit log
36          EntityManager em = ThreadLocalContext.get().getEntityManager();
37          auditLog.setCreateDate(new Date());
38          em.persist(auditLog);
39          
40          return auditLog.getId();
41      }
42      
43      /**
44       * Searches audit logs by criteria.
45       * 
46       * @param paginatedList List of audit logs (updated by side effects)
47       * @param criteria Search criteria
48       * @param sortCriteria Sort criteria
49       */
50      public void findByCriteria(PaginatedList<AuditLogDto> paginatedList, AuditLogCriteria criteria, SortCriteria sortCriteria) {
51          Map<String, Object> parameterMap = new HashMap<>();
52          
53          StringBuilder baseQuery = new StringBuilder("select l.LOG_ID_C c0, l.LOG_CREATEDATE_D c1, u.USE_USERNAME_C c2, l.LOG_IDENTITY_C c3, l.LOG_CLASSENTITY_C c4, l.LOG_TYPE_C c5, l.LOG_MESSAGE_C c6 from T_AUDIT_LOG l ");
54          baseQuery.append(" join T_USER u on l.LOG_IDUSER_C = u.USE_ID_C ");
55          List<String> queries = Lists.newArrayList();
56          
57          // Adds search criteria
58          if (criteria.getDocumentId() != null) {
59              // ACL on document is not checked here, rights have been checked before
60              queries.add(baseQuery + " where l.LOG_IDENTITY_C = :documentId ");
61              queries.add(baseQuery + " where l.LOG_IDENTITY_C in (select f.FIL_ID_C from T_FILE f where f.FIL_IDDOC_C = :documentId) ");
62              queries.add(baseQuery + " where l.LOG_IDENTITY_C in (select c.COM_ID_C from T_COMMENT c where c.COM_IDDOC_C = :documentId) ");
63              queries.add(baseQuery + " where l.LOG_IDENTITY_C in (select a.ACL_ID_C from T_ACL a where a.ACL_SOURCEID_C = :documentId) ");
64              queries.add(baseQuery + " where l.LOG_IDENTITY_C in (select r.RTE_ID_C from T_ROUTE r where r.RTE_IDDOCUMENT_C = :documentId) ");
65              parameterMap.put("documentId", criteria.getDocumentId());
66          }
67          
68          if (criteria.getUserId() != null) {
69              if (criteria.isAdmin()) {
70                  // For admin users, display all logs except ACL logs
71                  queries.add(baseQuery + " where l.LOG_CLASSENTITY_C != 'Acl' ");
72              } else {
73                  // Get all logs originating from the user, not necessarly on owned items
74                  // Filter out ACL logs
75                  queries.add(baseQuery + " where l.LOG_IDUSER_C = :userId and l.LOG_CLASSENTITY_C != 'Acl' ");
76                  parameterMap.put("userId", criteria.getUserId());
77              }
78          }
79          
80          // Perform the search
81          QueryParam queryParam = new QueryParam(Joiner.on(" union ").join(queries), parameterMap);
82          List<Object[]> l = PaginatedLists.executePaginatedQuery(paginatedList, queryParam, sortCriteria);
83          
84          // Assemble results
85          List<AuditLogDto> auditLogDtoList = new ArrayList<>();
86          for (Object[] o : l) {
87              int i = 0;
88              AuditLogDto auditLogDto = new AuditLogDto();
89              auditLogDto.setId((String) o[i++]);
90              auditLogDto.setCreateTimestamp(((Timestamp) o[i++]).getTime());
91              auditLogDto.setUsername((String) o[i++]);
92              auditLogDto.setEntityId((String) o[i++]);
93              auditLogDto.setEntityClass((String) o[i++]);
94              auditLogDto.setType(AuditLogType.valueOf((String) o[i++]));
95              auditLogDto.setMessage((String) o[i++]);
96              auditLogDtoList.add(auditLogDto);
97          }
98  
99          paginatedList.setResultList(auditLogDtoList);
100     }
