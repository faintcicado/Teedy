18  public class ContributorDao {
19      /**
20       * Creates a new contributor.
21       * 
22       * @param contributor Contributor
23       * @return New ID
24       */
25      public String create(Contributor contributor) {
26          // Create the UUID
27          contributor.setId(UUID.randomUUID().toString());
28          
29          // Create the contributor
30          EntityManager em = ThreadLocalContext.get().getEntityManager();
31          em.persist(contributor);
32          
33          return contributor.getId();
34      }
35      
36      /**
37       * Returns the list of all contributors by document.
38       * 
39       * @param documentId Document ID
40       * @return List of contributors
41       */
42      @SuppressWarnings("unchecked")
43      public List<Contributor> findByDocumentId(String documentId) {
44          EntityManager em = ThreadLocalContext.get().getEntityManager();
45          Query q = em.createQuery("select c from Contributor c where c.documentId = :documentId");
46          q.setParameter("documentId", documentId);
47          return q.getResultList();
48      }
49      
50      /**
51       * Returns the list of all contributors by document.
52       * 
53       * @param documentId Document ID
54       * @return List of contributors
55       */
56      @SuppressWarnings("unchecked")
57      public List<ContributorDto> getByDocumentId(String documentId) {
58          EntityManager em = ThreadLocalContext.get().getEntityManager();
59          StringBuilder sb = new StringBuilder("select distinct u.USE_USERNAME_C, u.USE_EMAIL_C from T_CONTRIBUTOR c ");
60          sb.append(" join T_USER u on u.USE_ID_C = c.CTR_IDUSER_C ");
61          sb.append(" where c.CTR_IDDOC_C = :documentId ");
62          Query q = em.createNativeQuery(sb.toString());
63          q.setParameter("documentId", documentId);
64          List<Object[]> l = q.getResultList();
65          
66          // Assemble results
67          List<ContributorDto> contributorDtoList = new ArrayList<>();
68          for (Object[] o : l) {
69              int i = 0;
70              ContributorDto contributorDto = new ContributorDto();
71              contributorDto.setUsername((String) o[i++]);
72              contributorDto.setEmail((String) o[i]);
73              contributorDtoList.add(contributorDto);
74          }
75          return contributorDtoList;
76      }
77  }
