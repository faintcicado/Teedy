5   import com.sismics.docs.core.constant.Constants;
6   import com.sismics.docs.core.dao.dto.UserDto;
7   import com.sismics.docs.core.event.RouteStepValidateEvent;
8   import com.sismics.docs.core.util.TransactionUtil;
9   import com.sismics.util.EmailUtil;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import java.util.HashMap;
14  import java.util.Map;
15  
16  /**
17   * Listener for route step validate.
18   *
19   * @author bgamard
20   */
21  public class RouteStepValidateAsyncListener {
22      /**
23       * Logger.
24       */
25      private static final Logger log = LoggerFactory.getLogger(RouteStepValidateAsyncListener.class);
26  
27      /**
28       * Handle events.
29       * 
30       * @param event Event
31       */
32      @Subscribe
33      @AllowConcurrentEvents
34      public void on(final RouteStepValidateEvent event) {
35          if (log.isInfoEnabled()) {
36              log.info("Route step validate event: " + event.toString());
37          }
38          
39          TransactionUtil.handle(() -> {
40              final UserDto user = event.getUser();
41  
42              // Send route step validated email
43              Map<String, Object> paramRootMap = new HashMap<>();
44              paramRootMap.put("user_name", user.getUsername());
45              paramRootMap.put("document_id", event.getDocument().getId());
46              paramRootMap.put("document_title", event.getDocument().getTitle());
47  
48              EmailUtil.sendEmail(Constants.EMAIL_TEMPLATE_ROUTE_STEP_VALIDATE, user, paramRootMap);
49          });
50      }
51  }
