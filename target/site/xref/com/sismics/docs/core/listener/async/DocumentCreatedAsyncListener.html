1   package com.sismics.docs.core.listener.async;
2   
3   import com.google.common.eventbus.AllowConcurrentEvents;
4   import com.google.common.eventbus.Subscribe;
5   import com.sismics.docs.core.dao.ContributorDao;
6   import com.sismics.docs.core.dao.DocumentDao;
7   import com.sismics.docs.core.event.DocumentCreatedAsyncEvent;
8   import com.sismics.docs.core.model.context.AppContext;
9   import com.sismics.docs.core.model.jpa.Contributor;
10  import com.sismics.docs.core.model.jpa.Document;
11  import com.sismics.docs.core.util.TransactionUtil;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  /**
16   * Listener on document created.
17   * 
18   * @author bgamard
19   */
20  public class DocumentCreatedAsyncListener {
21      /**
22       * Logger.
23       */
24      private static final Logger log = LoggerFactory.getLogger(DocumentCreatedAsyncListener.class);
25  
26      /**
27       * Document created.
28       * 
29       * @param event Document created event
30       */
31      @Subscribe
32      @AllowConcurrentEvents
33      public void on(final DocumentCreatedAsyncEvent event) {
34          if (log.isInfoEnabled()) {
35              log.info("Document created event: " + event.toString());
36          }
37  
38          TransactionUtil.handle(() -> {
39              // Fetch a fresh document
40              Document document = new DocumentDao().getById(event.getDocumentId());
41              if (document == null) {
42                  // The document has been deleted since
43                  return;
44              }
45  
46              // Add the first contributor (the creator of the document)
47              ContributorDao contributorDao = new ContributorDao();
48              Contributor contributor = new Contributor();
49              contributor.setDocumentId(event.getDocumentId());
50              contributor.setUserId(event.getUserId());
51              contributorDao.create(contributor);
52  
53              // Update index
54              AppContext.getInstance().getIndexingHandler().createDocument(document);
55          });
56      }
57  }
