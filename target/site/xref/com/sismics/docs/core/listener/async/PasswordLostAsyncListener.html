3   import com.google.common.eventbus.AllowConcurrentEvents;
4   import com.google.common.eventbus.Subscribe;
5   import com.sismics.docs.core.constant.Constants;
6   import com.sismics.docs.core.dao.dto.UserDto;
7   import com.sismics.docs.core.event.PasswordLostEvent;
8   import com.sismics.docs.core.model.jpa.PasswordRecovery;
9   import com.sismics.docs.core.util.TransactionUtil;
10  import com.sismics.util.EmailUtil;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import java.util.HashMap;
15  import java.util.Map;
16  
17  /**
18   * Listener for password recovery requests.
19   *
20   * @author jtremeaux 
21   */
22  public class PasswordLostAsyncListener {
23      /**
24       * Logger.
25       */
26      private static final Logger log = LoggerFactory.getLogger(PasswordLostAsyncListener.class);
27  
28      /**
29       * Handle events.
30       * 
31       * @param event Event
32       */
33      @Subscribe
34      @AllowConcurrentEvents
35      public void on(final PasswordLostEvent event) {
36          if (log.isInfoEnabled()) {
37              log.info("Password lost event: " + event.toString());
38          }
39          
40          TransactionUtil.handle(() -> {
41              final UserDto user = event.getUser();
42              final PasswordRecovery passwordRecovery = event.getPasswordRecovery();
43  
44              // Send the password recovery email
45              Map<String, Object> paramRootMap = new HashMap<>();
46              paramRootMap.put("user_name", user.getUsername());
47              paramRootMap.put("password_recovery_key", passwordRecovery.getId());
48  
49              EmailUtil.sendEmail(Constants.EMAIL_TEMPLATE_PASSWORD_RECOVERY, user, paramRootMap);
50          });
51      }
52  }
