40      /**
41       * Parse a query according to the specified syntax, eg.:
42       * tag:assurance tag:other before:2012 after:2011-09 shared:yes lang:fra thing
43       *
44       * @param search        Search query
45       * @param allTagDtoList List of tags
46       * @return DocumentCriteria
47       */
48      public static DocumentCriteria parseSearchQuery(String search, List<TagDto> allTagDtoList) {
49          DocumentCriteria documentCriteria = new DocumentCriteria();
50          if (Strings.isNullOrEmpty(search)) {
51              return documentCriteria;
52          }
53  
54          String[] criteriaList = search.split(" +");
55          List<String> simpleQuery = new ArrayList<>();
56          List<String> fullQuery = new ArrayList<>();
57          for (String criteria : criteriaList) {
58              String[] params = criteria.split(":");
59              if (params.length != 2 || Strings.isNullOrEmpty(params[0]) || Strings.isNullOrEmpty(params[1])) {
60                  // This is not a special criteria, do a fulltext search on it
61                  fullQuery.add(criteria);
62                  continue;
63              }
64              String paramName = params[0];
65              String paramValue = params[1];
66  
67              switch (paramName) {
68                  case "tag":
69                  case "!tag":
70                      parseTagCriteria(documentCriteria, paramValue, allTagDtoList, paramName.startsWith("!"));
71                      break;
72                  case "after":
73                  case "before":
74                  case "uafter":
75                  case "ubefore":
76                      parseDateCriteria(documentCriteria, paramValue, DATES_FORMATTER, paramName.startsWith("u"), paramName.endsWith("before"));
77                      break;
78                  case "uat":
79                  case "at":
80                      parseDateAtCriteria(documentCriteria, paramValue, params[0].startsWith("u"));
81                      break;
82                  case "shared":
83                      documentCriteria.setShared(paramValue.equals("yes"));
84                      break;
85                  case "lang":
86                      parseLangCriteria(documentCriteria, paramValue);
87                      break;
88                  case "mime":
89                      documentCriteria.setMimeType(paramValue);
90                      break;
91                  case "by":
92                      parseByCriteria(documentCriteria, paramValue);
93                      break;
94                  case "workflow":
95                      documentCriteria.setActiveRoute(paramValue.equals(WORKFLOW_ME));
96                      break;
97                  case "simple":
98                      simpleQuery.add(paramValue);
99                      break;
100                 case "full":
101                     fullQuery.add(paramValue);
102                     break;
103                 case "title":
104                     documentCriteria.getTitleList().add(paramValue);
105                     break;
106                 default:
107                     fullQuery.add(criteria);
108                     break;
109             }
110         }
111 
112         documentCriteria.setSimpleSearch(Joiner.on(" ").join(simpleQuery));
113         documentCriteria.setFullSearch(Joiner.on(" ").join(fullQuery));
114         return documentCriteria;
115     }

118     /**
119      * Fill the document criteria with various possible parameters
120      *
121      * @param documentCriteria    structure to be filled
122      * @param searchBy            author
123      * @param searchCreatedAfter  creation moment after
124      * @param searchCreatedBefore creation moment before
125      * @param searchFull          full search
126      * @param searchLang          lang
127      * @param searchMime          mime type
128      * @param searchShared        share state
129      * @param searchSimple        search in
130      * @param searchTag           tags or parent tags
131      * @param searchNotTag        tags or parent tags to ignore
132      * @param searchTitle         title
133      * @param searchUpdatedAfter  update moment after
134      * @param searchUpdatedBefore update moment before
135      * @param searchWorkflow      exiting workflow
136      * @param allTagDtoList       list of existing tags
137      */
138     public static void addHttpSearchParams(
139             DocumentCriteria documentCriteria,
140             String searchBy,
141             String searchCreatedAfter,
142             String searchCreatedBefore,
143             String searchFull,
144             String searchLang,
145             String searchMime,
146             Boolean searchShared,
147             String searchSimple,
148             String searchTag,
149             String searchNotTag,
150             String searchTitle,
151             String searchUpdatedAfter,
152             String searchUpdatedBefore,
153             String searchWorkflow,
154             List<TagDto> allTagDtoList
155     ) {
156         if (searchBy != null) {
157             parseByCriteria(documentCriteria, searchBy);
158         }
159         if (searchCreatedAfter != null) {
160             parseDateCriteria(documentCriteria, searchCreatedAfter, DAY_FORMATTER, false, false);
161         }
162         if (searchCreatedBefore != null) {
163             parseDateCriteria(documentCriteria, searchCreatedBefore, DAY_FORMATTER, false, true);
164         }
165         if (searchFull != null) {
166             documentCriteria.setFullSearch(Joiner.on(" ").join(searchFull.split(PARAMETER_WITH_MULTIPLE_VALUES_SEPARATOR)));
167         }
168         if (searchLang != null) {
169             parseLangCriteria(documentCriteria, searchLang);
170         }
171         if (searchMime != null) {
172             documentCriteria.setMimeType(searchMime);
173         }
174         if ((searchShared != null) && searchShared) {
175             documentCriteria.setShared(true);
176         }
177         if (searchSimple != null) {
178             documentCriteria.setSimpleSearch(Joiner.on(" ").join(searchSimple.split(PARAMETER_WITH_MULTIPLE_VALUES_SEPARATOR)));
179         }
180         if (searchTitle != null) {
181             documentCriteria.getTitleList().addAll(Arrays.asList(searchTitle.split(PARAMETER_WITH_MULTIPLE_VALUES_SEPARATOR)));
182         }
183         if (searchTag != null) {
184             for (String tag : searchTag.split(PARAMETER_WITH_MULTIPLE_VALUES_SEPARATOR)) {
185                 parseTagCriteria(documentCriteria, tag, allTagDtoList, false);
186             }
187         }
188         if (searchNotTag != null) {
189             for (String tag : searchNotTag.split(PARAMETER_WITH_MULTIPLE_VALUES_SEPARATOR)) {
190                 parseTagCriteria(documentCriteria, tag, allTagDtoList, true);
191             }
192         }
193         if (searchUpdatedAfter != null) {
194             parseDateCriteria(documentCriteria, searchUpdatedAfter, DAY_FORMATTER, true, false);
195         }
196         if (searchUpdatedBefore != null) {
197             parseDateCriteria(documentCriteria, searchUpdatedBefore, DAY_FORMATTER, true, true);
198         }
199         if ((WORKFLOW_ME.equals(searchWorkflow))) {
200             documentCriteria.setActiveRoute(true);
201         }
202     }

204     private static void parseDateCriteria(DocumentCriteria documentCriteria, String value, DateTimeFormatter formatter, boolean isUpdated, boolean isBefore) {
205         try {
206             DateTime date = formatter.parseDateTime(value);
207             if (isBefore) {
208                 if (isUpdated) {
209                     documentCriteria.setUpdateDateMax(date.toDate());
210                 } else {
211                     documentCriteria.setCreateDateMax(date.toDate());
212                 }
213             } else {
214                 if (isUpdated) {
215                     documentCriteria.setUpdateDateMin(date.toDate());
216                 } else {
217                     documentCriteria.setCreateDateMin(date.toDate());
218                 }
219             }
220         } catch (IllegalArgumentException e) {
221             // Invalid date, returns no documents
222             documentCriteria.setCreateDateMin(new Date(0));
223             documentCriteria.setCreateDateMax(new Date(0));
224         }
225     }

227     private static void parseDateAtCriteria(DocumentCriteria documentCriteria, String value, boolean isUpdated) {
228         try {
229             switch (value.length()) {
230                 case 10: {
231                     DateTime date = DATES_FORMATTER.parseDateTime(value);
232                     if (isUpdated) {
233                         documentCriteria.setUpdateDateMin(date.toDate());
234                         documentCriteria.setUpdateDateMax(date.plusDays(1).minusSeconds(1).toDate());
235                     } else {
236                         documentCriteria.setCreateDateMin(date.toDate());
237                         documentCriteria.setCreateDateMax(date.plusDays(1).minusSeconds(1).toDate());
238                     }
239                     break;
240                 }
241                 case 7: {
242                     DateTime date = MONTH_FORMATTER.parseDateTime(value);
243                     if (isUpdated) {
244                         documentCriteria.setUpdateDateMin(date.toDate());
245                         documentCriteria.setUpdateDateMax(date.plusMonths(1).minusSeconds(1).toDate());
246                     } else {
247                         documentCriteria.setCreateDateMin(date.toDate());
248                         documentCriteria.setCreateDateMax(date.plusMonths(1).minusSeconds(1).toDate());
249                     }
250                     break;
251                 }
252                 case 4: {
253                     DateTime date = YEAR_FORMATTER.parseDateTime(value);
254                     if (isUpdated) {
255                         documentCriteria.setUpdateDateMin(date.toDate());
256                         documentCriteria.setUpdateDateMax(date.plusYears(1).minusSeconds(1).toDate());
257                     } else {
258                         documentCriteria.setCreateDateMin(date.toDate());
259                         documentCriteria.setCreateDateMax(date.plusYears(1).minusSeconds(1).toDate());
260                     }
261                     break;
262                 }
263                 default: {
264                     // Invalid format, returns no documents
265                     documentCriteria.setCreateDateMin(new Date(0));
266                     documentCriteria.setCreateDateMax(new Date(0));
267                 }
268             }
269         } catch (IllegalArgumentException e) {
270             // Invalid date, returns no documents
271             documentCriteria.setCreateDateMin(new Date(0));
272             documentCriteria.setCreateDateMax(new Date(0));
273         }
274     }

276     private static void parseTagCriteria(DocumentCriteria documentCriteria, String value, List<TagDto> allTagDtoList, boolean exclusion) {
277         List<TagDto> tagDtoList = TagUtil.findByName(value, allTagDtoList);
278         if (tagDtoList.isEmpty()) {
279             // No tag found, the request must return nothing
280             documentCriteria.getTagIdList().add(Lists.newArrayList(UUID.randomUUID().toString()));
281         } else {
282             List<String> tagIdList = Lists.newArrayList();
283             for (TagDto tagDto : tagDtoList) {
284                 tagIdList.add(tagDto.getId());
285                 List<TagDto> childrenTagDtoList = TagUtil.findChildren(tagDto, allTagDtoList);
286                 for (TagDto childrenTagDto : childrenTagDtoList) {
287                     tagIdList.add(childrenTagDto.getId());
288                 }
289             }
290             if (exclusion) {
291                 documentCriteria.getExcludedTagIdList().add(tagIdList);
292             } else {
293                 documentCriteria.getTagIdList().add(tagIdList);
294             }
295         }
296     }

298     private static void parseLangCriteria(DocumentCriteria documentCriteria, String value) {
299         // New language criteria
300         if (Constants.SUPPORTED_LANGUAGES.contains(value)) {
301             documentCriteria.setLanguage(value);
302         } else {
303             // Unsupported language, returns no documents
304             documentCriteria.setLanguage(UUID.randomUUID().toString());
305         }
306     }

308     private static void parseByCriteria(DocumentCriteria documentCriteria, String value) {
309         User user = new UserDao().getActiveByUsername(value);
310         if (user == null) {
311             // This user doesn't exist, return nothing
312             documentCriteria.setCreatorId(UUID.randomUUID().toString());
313         } else {
314             // This user exists, search its documents
315             documentCriteria.setCreatorId(user.getId());
316         }
