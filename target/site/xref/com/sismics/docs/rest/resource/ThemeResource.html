91      /**
92       * Change the theme configuration.
93       *
94       * @api {post} /theme Change the theme configuration
95       * @apiName PostTheme
96       * @apiGroup Theme
97       * @apiParam {String} name Application name
98       * @apiParam {String} color Main color
99       * @apiParam {String} css Custom CSS
100      * @apiSuccess {String} status Status OK
101      * @apiError (client) ForbiddenError Access denied
102      * @apiError (client) ValidationError Validation error
103      * @apiPermission admin
104      * @apiVersion 1.5.0
105      *
106      * @param color Theme color
107      * @param name Application name
108      * @param css Custom CSS
109      * @return Response
110      */
111     @POST
112     public Response theme(@FormParam("color") String color,
113               @FormParam("name") String name,
114               @FormParam("css") String css) {
115         if (!authenticate()) {
116             throw new ForbiddenClientException();
117         }
118         checkBaseFunction(BaseFunction.ADMIN);
119 
120         // Validate input data
121         ValidationUtil.validateHexColor(color, "color", true);
122         name = ValidationUtil.validateLength(name, "name", 3, 30, true);
123 
124         // Update the JSON
125         JsonObjectBuilder json = getMutableThemeConfig();
126         if (Strings.isNullOrEmpty(color)) {
127             json.add("color", JsonValue.NULL);
128         } else {
129             json.add("color", color);
130         }
131         if (Strings.isNullOrEmpty(name)) {
132             json.add("name", JsonValue.NULL);
133         } else {
134             json.add("name", name);
135         }
136         json.add("css", JsonUtil.nullable(css));
137 
138         // Persist the new configuration
139         ConfigDao configDao = new ConfigDao();
140         configDao.update(ConfigType.THEME, json.build().toString());
141 
142         // Always return OK
143         JsonObjectBuilder response = Json.createObjectBuilder()
144                 .add("status", "ok");
145         return Response.ok().entity(response.build()).build();
146     }

148     /**
149      * Change a theme image.
150      *
151      * @api {put} /theme/image/:type Change a theme image
152      * @apiDescription This resource accepts only multipart/form-data.
153      * @apiName PutThemeImage
154      * @apiGroup Theme
155      * @apiParam {String="logo","background"} type Image type
156      * @apiParam {String} image Image data
157      * @apiSuccess {String} status Status OK
158      * @apiError (client) ForbiddenError Access denied
159      * @apiError (client) NoImageProvided An image is required
160      * @apiError (server) CopyError Error copying the image to the theme directory
161      * @apiPermission admin
162      * @apiVersion 1.5.0
163      *
164      * @param type Image type
165      * @param imageBodyPart Image data
166      * @return Response
167      */
168     @PUT
169     @Path("image/{type: logo|background}")
170     @Consumes("multipart/form-data")
171     public Response images(@PathParam("type") String type,
172             @FormDataParam("image") FormDataBodyPart imageBodyPart) {
173         if (!authenticate()) {
174             throw new ForbiddenClientException();
175         }
176         checkBaseFunction(BaseFunction.ADMIN);
177         if (imageBodyPart == null) {
178             throw new ClientException("NoImageProvided", "An image is required");
179         }
180 
181         // Only a background or a logo is handled
182         java.nio.file.Path filePath = DirectoryUtil.getThemeDirectory().resolve(type);
183 
184         // Copy the image to the theme directory
185         try (InputStream inputStream = imageBodyPart.getValueAs(InputStream.class)) {
186             Files.copy(inputStream, filePath, StandardCopyOption.REPLACE_EXISTING);
187         } catch (Exception e) {
188             throw new ServerException("CopyError", "Error copying the image to the theme directory", e);
189         }
190 
191         return Response.ok().build();
192     }

194     /**
195      * Get theme images.
196      *
197      * @api {get} /theme/image/:type Get a theme image
198      * @apiName GetThemeImage
199      * @apiGroup Theme
200      * @apiParam {String="logo","background"} type Image type
201      * @apiSuccess {String} image The whole response is the image
202      * @apiPermission none
203      * @apiVersion 1.5.0
204      *
205      * @param type Image type
206      * @return Response
207      */
208     @GET
209     @Produces("image/*")
210     @Path("image/{type: logo|background}")
211     public Response getImage(@PathParam("type") final String type) {
212         final java.nio.file.Path filePath = DirectoryUtil.getThemeDirectory().resolve(type);
213 
214         // Copy the image to the response output
215         return Response.ok(new StreamingOutput() {
216             @Override
217             public void write(OutputStream outputStream) throws IOException, WebApplicationException {
218                 InputStream inputStream = null;
219                 try {
220                     if (Files.exists(filePath)) {
221                         inputStream = Files.newInputStream(filePath);
222                     } else {
223                         inputStream = getClass().getResource("/image/" + (type.equals("logo") ? "logo.png" : "background.jpg")).openStream();
224                     }
225                     ByteStreams.copy(inputStream, outputStream);
226                 } finally {
227                     try {
228                         if (inputStream != null) {
229                             inputStream.close();
230                         }
231                         outputStream.close();
232                     } catch (IOException e) {
233                         // Ignore
234                     }
235                 }
236             }
237         })
238         .header(HttpHeaders.CONTENT_TYPE, "image/*")
239         .header(HttpHeaders.CACHE_CONTROL, "public")
240         .header(HttpHeaders.EXPIRES, HttpUtil.buildExpiresHeader(3_600_000L * 24L * 15L))
241         .build();
242     }

244     /**
245      * Returns the theme configuration object.
246      *
247      * @return Theme configuration
248      */
249     private JsonObject getThemeConfig() {
250         ConfigDao configDao = new ConfigDao();
251         Config themeConfig = configDao.getById(ConfigType.THEME);
252         if (themeConfig == null) {
253             return Json.createObjectBuilder().build();
254         }
255 
256         try (JsonReader reader = Json.createReader(new StringReader(themeConfig.getValue()))) {
257             return reader.readObject();
258         }
259     }

261     /**
262      * Returns a mutable theme configuration.
263      *
264      * @return Json builder
265      */
266     private JsonObjectBuilder getMutableThemeConfig() {
267         JsonObject themeConfig = getThemeConfig();
268         JsonObjectBuilder json = Json.createObjectBuilder();
269 
270         for (Map.Entry<String, JsonValue> entry : themeConfig.entrySet()) {
271             json.add(entry.getKey(), entry.getValue());
272         }
273 
274         return json;
275     }
