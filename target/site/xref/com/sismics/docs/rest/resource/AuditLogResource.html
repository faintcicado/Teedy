25  /**
26   * Audit log REST resources.
27   * 
28   * @author bgamard
29   */
30  @Path("/auditlog")
31  public class AuditLogResource extends BaseResource {
32      /**
33       * Returns the list of all logs for a document or user.
34       *
35       * @api {get} /auditlog Get audit logs
36       * @apiDescription If no document ID is provided, logs for the current user will be returned.
37       * @apiName GetAuditlog
38       * @apiGroup Auditlog
39       * @apiParam {String} [document] Document ID
40       * @apiSuccess {String} total Total number of logs
41       * @apiSuccess {Object[]} logs List of logs
42       * @apiSuccess {String} logs.id ID
43       * @apiSuccess {String} logs.username Username
44       * @apiSuccess {String} logs.target Entity ID
45       * @apiSuccess {String="Acl","Comment","Document","File","Group","Tag","User","RouteModel","Route"} logs.class Entity type
46       * @apiSuccess {String="CREATE","UPDATE","DELETE"} logs.type Type
47       * @apiSuccess {String} logs.message Message
48       * @apiSuccess {Number} logs.create_date Create date (timestamp)
49       * @apiError (client) ForbiddenError Access denied
50       * @apiError (client) NotFound Document not found
51       * @apiPermission user
52       * @apiVersion 1.5.0
53       *
54       * @return Response
55       */
56      @GET
57      public Response list(@QueryParam("document") String documentId) {
58          if (!authenticate()) {
59              throw new ForbiddenClientException();
60          }
61          
62          // On a document or a user?
63          PaginatedList<AuditLogDto> paginatedList = PaginatedLists.create(20, 0);
64          SortCriteria sortCriteria = new SortCriteria(1, false);
65          AuditLogCriteria criteria = new AuditLogCriteria();
66          if (Strings.isNullOrEmpty(documentId)) {
67              // Search logs for a user
68              criteria.setUserId(principal.getId());
69              criteria.setAdmin(SecurityUtil.skipAclCheck(getTargetIdList(null)));
70          } else {
71              // Check ACL on the document
72              AclDao aclDao = new AclDao();
73              if (!aclDao.checkPermission(documentId, PermType.READ, getTargetIdList(null))) {
74                  throw new NotFoundException();
75              }
76              criteria.setDocumentId(documentId);
77          }
78          
79          // Search the logs
80          AuditLogDao auditLogDao = new AuditLogDao();
81          auditLogDao.findByCriteria(paginatedList, criteria, sortCriteria);
82          
83          // Assemble the results
84          JsonArrayBuilder logs = Json.createArrayBuilder();
85          for (AuditLogDto auditLogDto : paginatedList.getResultList()) {
86              logs.add(Json.createObjectBuilder()
87                      .add("id", auditLogDto.getId())
88                      .add("username", auditLogDto.getUsername())
89                      .add("target", auditLogDto.getEntityId())
90                      .add("class", auditLogDto.getEntityClass())
91                      .add("type", auditLogDto.getType().name())
92                      .add("message", JsonUtil.nullable(auditLogDto.getMessage()))
93                      .add("create_date", auditLogDto.getCreateTimestamp()));
94          }
95  
96          // Send the response
97          JsonObjectBuilder response = Json.createObjectBuilder()
98                  .add("logs", logs)
99                  .add("total", paginatedList.getResultCount());
100         return Response.ok().entity(response.build()).build();
101     }
102 }
