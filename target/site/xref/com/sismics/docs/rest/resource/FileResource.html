63       * Add a file (with or without a document).
64       *
65       * @api {put} /file Add a file
66       * @apiDescription A file can be added without associated document, and will go in a temporary storage waiting for one.
67       * This resource accepts only multipart/form-data.
68       * @apiName PutFile
69       * @apiGroup File
70       * @apiParam {String} [id] Document ID
71       * @apiParam {String} [previousFileId] ID of the file to replace by this new version
72       * @apiParam {String} file File data
73       * @apiSuccess {String} status Status OK
74       * @apiSuccess {String} id File ID
75       * @apiSuccess {Number} size File size (in bytes)
76       * @apiError (client) ForbiddenError Access denied
77       * @apiError (client) ValidationError Validation error
78       * @apiError (client) NotFound Document not found
79       * @apiError (server) StreamError Error reading the input file
80       * @apiError (server) ErrorGuessMime Error guessing mime type
81       * @apiError (client) QuotaReached Quota limit reached
82       * @apiError (server) FileError Error adding a file
83       * @apiPermission user
84       * @apiVersion 1.5.0
85       *
86       * @param documentId Document ID
87       * @param fileBodyPart File to add
88       * @return Response
89       */
90      @PUT
91      @Consumes("multipart/form-data")
92      public Response add(
93              @FormDataParam("id") String documentId,
94              @FormDataParam("previousFileId") String previousFileId,
95              @FormDataParam("file") FormDataBodyPart fileBodyPart) {
96          if (!authenticate()) {
97              throw new ForbiddenClientException();
98          }
99          
100         // Validate input data
101         ValidationUtil.validateRequired(fileBodyPart, "file");
102 
103         // Get the document
104         DocumentDto documentDto = null;
105         if (Strings.isNullOrEmpty(documentId)) {
106             documentId = null;
107         } else {
108             DocumentDao documentDao = new DocumentDao();
109             documentDto = documentDao.getDocument(documentId, PermType.WRITE, getTargetIdList(null));
110             if (documentDto == null) {
111                 throw new NotFoundException();
112             }
113         }
114         
115         // Keep unencrypted data temporary on disk
116         String name = fileBodyPart.getContentDisposition() != null ?
117                 URLDecoder.decode(fileBodyPart.getContentDisposition().getFileName(), StandardCharsets.UTF_8) : null;
118         java.nio.file.Path unencryptedFile;
119         long fileSize;
120         try {
121             unencryptedFile = AppContext.getInstance().getFileService().createTemporaryFile(name);
122             Files.copy(fileBodyPart.getValueAs(InputStream.class), unencryptedFile, StandardCopyOption.REPLACE_EXISTING);
123             fileSize = Files.size(unencryptedFile);
124         } catch (IOException e) {
125             throw new ServerException("StreamError", "Error reading the input file", e);
126         }
127 
128         try {
129             String fileId = FileUtil.createFile(name, previousFileId, unencryptedFile, fileSize, documentDto == null ?
130                     null : documentDto.getLanguage(), principal.getId(), documentId);
131 
132             // Always return OK
133             JsonObjectBuilder response = Json.createObjectBuilder()
134                     .add("status", "ok")
135                     .add("id", fileId)
136                     .add("size", fileSize);
137             return Response.ok().entity(response.build()).build();
138         } catch (IOException e) {
139             throw new ClientException(e.getMessage(), e.getMessage(), e);
140         } catch (Exception e) {
141             throw new ServerException("FileError", "Error adding a file", e);
142         }
143     }

145     /**
146      * Attach a file to a document.
147      *
148      * @api {post} /file/:fileId/attach Attach a file to a document
149      * @apiName PostFileAttach
150      * @apiGroup File
151      * @apiParam {String} fileId File ID
152      * @apiParam {String} id Document ID
153      * @apiSuccess {String} status Status OK
154      * @apiError (client) ForbiddenError Access denied
155      * @apiError (client) ValidationError Validation error
156      * @apiError (client) IllegalFile File not orphan
157      * @apiError (server) AttachError Error attaching file to document
158      * @apiPermission user
159      * @apiVersion 1.5.0
160      *
161      * @param id File ID
162      * @return Response
163      */
164     @POST
165     @Path("{id: [a-z0-9\\-]+}/attach")
166     public Response attach(
167             @PathParam("id") String id,
168             @FormParam("id") String documentId) {
169         if (!authenticate()) {
170             throw new ForbiddenClientException();
171         }
172 
173         // Validate input data
174         ValidationUtil.validateRequired(documentId, "documentId");
175         
176         // Get the current user
177         UserDao userDao = new UserDao();
178         User user = userDao.getById(principal.getId());
179         
180         // Get the document and the file
181         DocumentDao documentDao = new DocumentDao();
182         FileDao fileDao = new FileDao();
183         File file = fileDao.getFile(id, principal.getId());
184         DocumentDto documentDto = documentDao.getDocument(documentId, PermType.WRITE, getTargetIdList(null));
185         if (file == null || documentDto == null) {
186             throw new NotFoundException();
187         }
188         
189         // Check that the file is orphan
190         if (file.getDocumentId() != null) {
191             throw new ClientException("IllegalFile", MessageFormat.format("File not orphan: {0}", id));
192         }
193         
194         // Update the file
195         file.setDocumentId(documentId);
196         file.setOrder(fileDao.getByDocumentId(principal.getId(), documentId).size());
197         fileDao.update(file);
198         
199         // Raise a new file updated event and document updated event (it wasn't sent during file creation)
200         try {
201             java.nio.file.Path storedFile = DirectoryUtil.getStorageDirectory().resolve(id);
202             java.nio.file.Path unencryptedFile = EncryptionUtil.decryptFile(storedFile, user.getPrivateKey());
203             FileUtil.startProcessingFile(id);
204             FileUpdatedAsyncEvent fileUpdatedAsyncEvent = new FileUpdatedAsyncEvent();
205             fileUpdatedAsyncEvent.setUserId(principal.getId());
206             fileUpdatedAsyncEvent.setLanguage(documentDto.getLanguage());
207             fileUpdatedAsyncEvent.setFileId(file.getId());
208             fileUpdatedAsyncEvent.setUnencryptedFile(unencryptedFile);
209             ThreadLocalContext.get().addAsyncEvent(fileUpdatedAsyncEvent);
210             
211             DocumentUpdatedAsyncEvent documentUpdatedAsyncEvent = new DocumentUpdatedAsyncEvent();
212             documentUpdatedAsyncEvent.setUserId(principal.getId());
213             documentUpdatedAsyncEvent.setDocumentId(documentId);
214             ThreadLocalContext.get().addAsyncEvent(documentUpdatedAsyncEvent);
215         } catch (Exception e) {
216             throw new ServerException("AttachError", "Error attaching file to document", e);
217         }
218 
219         // Always return OK
220         JsonObjectBuilder response = Json.createObjectBuilder()
221                 .add("status", "ok");
222         return Response.ok().entity(response.build()).build();
223     }

341      * @apiVersion 1.5.0
342      *
343      * @param documentId Document ID
344      * @param idList List of files ID in the new order
345      * @return Response
346      */
347     @POST
348     @Path("reorder")
349     public Response reorder(
350             @FormParam("id") String documentId,
351             @FormParam("order") List<String> idList) {
352         if (!authenticate()) {
353             throw new ForbiddenClientException();
354         }
355         
356         // Validate input data
357         ValidationUtil.validateRequired(documentId, "id");
358         ValidationUtil.validateRequired(idList, "order");
359         
360         // Get the document
361         AclDao aclDao = new AclDao();
362         if (!aclDao.checkPermission(documentId, PermType.WRITE, getTargetIdList(null))) {
363             throw new NotFoundException();
364         }
365         
366         // Reorder files
367         FileDao fileDao = new FileDao();
368         for (File file : fileDao.getByDocumentId(principal.getId(), documentId)) {
369             int order = idList.lastIndexOf(file.getId());
370             if (order != -1) {
371                 file.setOrder(order);
372             }
373         }
374 
375         // Raise a document updated event
376         DocumentUpdatedAsyncEvent event = new DocumentUpdatedAsyncEvent();
377         event.setUserId(principal.getId());
378         event.setDocumentId(documentId);
379         ThreadLocalContext.get().addAsyncEvent(event);
380         
381         // Always return OK
382         JsonObjectBuilder response = Json.createObjectBuilder()
383                 .add("status", "ok");
384         return Response.ok().entity(response.build()).build();
385     }

545     /**
546      * Returns a file.
547      *
548      * @api {get} /file/:id/data Get a file data
549      * @apiName GetFile
550      * @apiGroup File
551      * @apiParam {String} id File ID
552      * @apiParam {String} share Share ID
553      * @apiParam {String="web","thumb","content"} [size] Size variation
554      * @apiSuccess {Object} file The file data is the whole response
555      * @apiError (client) SizeError Size must be web or thumb
556      * @apiError (client) ForbiddenError Access denied or document not visible
557      * @apiError (client) NotFound File not found
558      * @apiError (server) ServiceUnavailable Error reading the file
559      * @apiPermission none
560      * @apiVersion 1.5.0
561      *
562      * @param fileId File ID
563      * @return Response
564      */
565     @GET
566     @Path("{id: [a-z0-9\\-]+}/data")
567     @Produces(MediaType.APPLICATION_OCTET_STREAM)
568     public Response data(
569             @PathParam("id") final String fileId,
570             @QueryParam("share") String shareId,
571             @QueryParam("size") String size) {
572         authenticate();
573         
574         if (size != null && !Lists.newArrayList("web", "thumb", "content").contains(size)) {
575             throw new ClientException("SizeError", "Size must be web, thumb or content");
576         }
577 
578         // Get the file
579         File file = findFile(fileId, shareId);
580 
581         // Get the stored file
582         UserDao userDao = new UserDao();
583         java.nio.file.Path storedFile;
584         String mimeType;
585         boolean decrypt;
586         if (size != null) {
587             if (size.equals("content")) {
588                 return Response.ok(Strings.nullToEmpty(file.getContent()))
589                         .header(HttpHeaders.CONTENT_TYPE, "text/plain; charset=utf-8")
590                         .build();
591             }
592 
593             storedFile = DirectoryUtil.getStorageDirectory().resolve(fileId + "_" + size);
594             mimeType = MimeType.IMAGE_JPEG; // Thumbnails are JPEG
595             decrypt = true; // Thumbnails are encrypted
596             if (!Files.exists(storedFile)) {
597                 try {
598                     storedFile = Paths.get(getClass().getResource("/image/file-" + size + ".png").toURI());
599                 } catch (URISyntaxException e) {
600                     // Ignore
601                 }
602                 mimeType = MimeType.IMAGE_PNG;
603                 decrypt = false;
604             }
605         } else {
606             storedFile = DirectoryUtil.getStorageDirectory().resolve(fileId);
607             mimeType = file.getMimeType();
608             decrypt = true; // Original files are encrypted
609         }
610         
611         // Stream the output and decrypt it if necessary
612         StreamingOutput stream;
613         
614         // A file is always encrypted by the creator of it
615         User user = userDao.getById(file.getUserId());
616         
617         // Write the decrypted file to the output
618         try {
619             InputStream fileInputStream = Files.newInputStream(storedFile);
620             final InputStream responseInputStream = decrypt ?
621                     EncryptionUtil.decryptInputStream(fileInputStream, user.getPrivateKey()) : fileInputStream;
622                     
623             stream = outputStream -> {
624                 try {
625                     ByteStreams.copy(responseInputStream, outputStream);
626                 } finally {
627                     try {
628                         responseInputStream.close();
629                         outputStream.close();
630                     } catch (IOException e) {
631                         // Ignore
632                     }
633                 }
634             };
635         } catch (Exception e) {
636             return Response.status(Status.SERVICE_UNAVAILABLE).build();
637         }
638 
639         Response.ResponseBuilder builder = Response.ok(stream)
640                 .header(HttpHeaders.CONTENT_DISPOSITION, "inline; filename=\"" + file.getFullName("data") + "\"")
641                 .header(HttpHeaders.CONTENT_TYPE, mimeType);
642         if (decrypt) {
643             // Cache real files
644             builder.header(HttpHeaders.CACHE_CONTROL, "private")
645                     .header(HttpHeaders.EXPIRES, HttpUtil.buildExpiresHeader(3_600_000L * 24L * 365L));
646         } else {
647             // Do not cache the temporary thumbnail
648             builder.header(HttpHeaders.CACHE_CONTROL, "no-store, must-revalidate")
649                     .header(HttpHeaders.EXPIRES, "0");
650         }
651         return builder.build();

663      * @apiError (client) NotFoundException Document not found
664      * @apiError (server) InternalServerError Error creating the ZIP file

668      * @param documentId Document ID
669      * @param shareId Share ID
670      * @return Response
671      */
672     @GET
673     @Path("zip")
674     @Produces({MediaType.APPLICATION_OCTET_STREAM, MediaType.TEXT_PLAIN})
675     public Response zip(
676             @QueryParam("id") String documentId,
677             @QueryParam("share") String shareId) {
678         authenticate();
679         
680         // Get the document
681         DocumentDao documentDao = new DocumentDao();
682         DocumentDto documentDto = documentDao.getDocument(documentId, PermType.READ, getTargetIdList(shareId));
683         if (documentDto == null) {

686 
687         // Get files associated with this document
688         FileDao fileDao = new FileDao();
689         final List<File> fileList = fileDao.getByDocumentId(principal.getId(), documentId);
690         String zipFileName = documentDto.getTitle().replaceAll("\\W+", "_");
691         return sendZippedFiles(zipFileName, fileList);

720     /**
721      * Sent the content of a list of files.
722      */
723     private Response sendZippedFiles(String zipFileName, List<File> fileList) {
724         final UserDao userDao = new UserDao();
725 
726         // Create the ZIP stream
727         StreamingOutput stream = outputStream -> {
728             try (ZipOutputStream zipOutputStream = new ZipOutputStream(outputStream)) {
729                 // Add each file to the ZIP stream
730                 int index = 0;
731                 for (File file : fileList) {
732                     java.nio.file.Path storedfile = DirectoryUtil.getStorageDirectory().resolve(file.getId());
733                     InputStream fileInputStream = Files.newInputStream(storedfile);
734 
735                     // Add the decrypted file to the ZIP stream
736                     // Files are encrypted by the creator of them
737                     User user = userDao.getById(file.getUserId());
738                     try (InputStream decryptedStream = EncryptionUtil.decryptInputStream(fileInputStream, user.getPrivateKey())) {
739                         ZipEntry zipEntry = new ZipEntry(index + "-" + file.getFullName(Integer.toString(index)));
740                         zipOutputStream.putNextEntry(zipEntry);
741                         ByteStreams.copy(decryptedStream, zipOutputStream);
742                         zipOutputStream.closeEntry();
743                     } catch (Exception e) {
744                         throw new WebApplicationException(e);
745                     }
746                     index++;
747                 }
748             }
749             outputStream.close();
750         };
751         
752         // Write to the output
753         return Response.ok(stream)
754                 .header("Content-Type", "application/zip")
755                 .header("Content-Disposition", "attachment; filename=\"" + zipFileName + ".zip\"")
756                 .build();
757     }

792     /**
793      * Check if a file is accessible to the current user
794      * @param shareId Share ID
795      * @param file
796      */
797     private void checkFileAccessible(String shareId, File file) {
798         if (file.getDocumentId() == null) {
799             // It's an orphan file
800             if (!file.getUserId().equals(principal.getId())) {
801                 // But not ours
802                 throw new ForbiddenClientException();
803             }
804         } else {
805             // Check document accessibility
806             AclDao aclDao = new AclDao();
807             if (!aclDao.checkPermission(file.getDocumentId(), PermType.READ, getTargetIdList(shareId))) {
808                 throw new ForbiddenClientException();
809             }
810         }
811     }
