175     @GET
176     @Path("{id: [a-z0-9\\-]+}")
177     public Response get(
178             @PathParam("id") String documentId,
179             @QueryParam("share") String shareId,
180             @QueryParam("files") Boolean files) {
181         authenticate();
182 
183         DocumentDao documentDao = new DocumentDao();
184         DocumentDto documentDto = documentDao.getDocument(documentId, PermType.READ, getTargetIdList(shareId));
185         if (documentDto == null) {
186             throw new NotFoundException();
187         }
188 
189         JsonObjectBuilder document = createDocumentObjectBuilder(documentDto)
190                 .add("creator", documentDto.getCreator())
191                 .add("coverage", JsonUtil.nullable(documentDto.getCoverage()))
192                 .add("file_count", documentDto.getFileCount())
193                 .add("format", JsonUtil.nullable(documentDto.getFormat()))
194                 .add("identifier", JsonUtil.nullable(documentDto.getIdentifier()))
195                 .add("publisher", JsonUtil.nullable(documentDto.getPublisher()))
196                 .add("rights", JsonUtil.nullable(documentDto.getRights()))
197                 .add("source", JsonUtil.nullable(documentDto.getSource()))
198                 .add("subject", JsonUtil.nullable(documentDto.getSubject()))
199                 .add("type", JsonUtil.nullable(documentDto.getType()));
200 
201         List<TagDto> tagDtoList = null;
202         if (principal.isAnonymous()) {
203             // No tags in anonymous mode (sharing)
204             document.add("tags", Json.createArrayBuilder());
205         } else {
206             // Add tags visible by the current user on this document
207             TagDao tagDao = new TagDao();
208             tagDtoList = tagDao.findByCriteria(
209                     new TagCriteria()
210                             .setTargetIdList(getTargetIdList(null)) // No tags for shares
211                             .setDocumentId(documentId),
212                     new SortCriteria(1, true));
213             document.add("tags", createTagsArrayBuilder(tagDtoList));
214         }
215 
216         // Add ACL
217         AclUtil.addAcls(document, documentId, getTargetIdList(shareId));
218 
219         // Add computed ACL
220         if (tagDtoList != null) {
221             JsonArrayBuilder aclList = Json.createArrayBuilder();
222             for (TagDto tagDto : tagDtoList) {
223                 AclDao aclDao = new AclDao();
224                 List<AclDto> aclDtoList = aclDao.getBySourceId(tagDto.getId(), AclType.USER);
225                 for (AclDto aclDto : aclDtoList) {
226                     aclList.add(Json.createObjectBuilder()
227                             .add("perm", aclDto.getPerm().name())
228                             .add("source_id", tagDto.getId())
229                             .add("source_name", tagDto.getName())
230                             .add("source_color", tagDto.getColor())
231                             .add("id", aclDto.getTargetId())
232                             .add("name", JsonUtil.nullable(aclDto.getTargetName()))
233                             .add("type", aclDto.getTargetType()));
234                 }
235             }
236             document.add("inherited_acls", aclList);
237         }
238 
239         // Add contributors
240         ContributorDao contributorDao = new ContributorDao();
241         List<ContributorDto> contributorDtoList = contributorDao.getByDocumentId(documentId);
242         JsonArrayBuilder contributorList = Json.createArrayBuilder();
243         for (ContributorDto contributorDto : contributorDtoList) {
244             contributorList.add(Json.createObjectBuilder()
245                     .add("username", contributorDto.getUsername())
246                     .add("email", contributorDto.getEmail()));
247         }
248         document.add("contributors", contributorList);
249 
250         // Add relations
251         RelationDao relationDao = new RelationDao();
252         List<RelationDto> relationDtoList = relationDao.getByDocumentId(documentId);
253         JsonArrayBuilder relationList = Json.createArrayBuilder();
254         for (RelationDto relationDto : relationDtoList) {
255             relationList.add(Json.createObjectBuilder()
256                     .add("id", relationDto.getId())
257                     .add("title", relationDto.getTitle())
258                     .add("source", relationDto.isSource()));
259         }
260         document.add("relations", relationList);
261 
262         // Add current route step
263         RouteStepDto routeStepDto = new RouteStepDao().getCurrentStep(documentId);
264         if (routeStepDto != null && !principal.isAnonymous()) {
265             JsonObjectBuilder step = routeStepDto.toJson();
266             step.add("transitionable", getTargetIdList(null).contains(routeStepDto.getTargetId()));
267             document.add("route_step", step);
268         }
269 
270         // Add custom metadata
271         MetadataUtil.addMetadata(document, documentId);
272 
273         // Add files
274         if (Boolean.TRUE == files) {
275             FileDao fileDao = new FileDao();
276             List<File> fileList = fileDao.getByDocumentsIds(Collections.singleton(documentId));
277 
278             JsonArrayBuilder filesArrayBuilder = Json.createArrayBuilder();
279             for (File fileDb : fileList) {
280                 filesArrayBuilder.add(RestUtil.fileToJsonObjectBuilder(fileDb));
281             }
282 
283             document.add("files", filesArrayBuilder);
284         }
285 
286         return Response.ok().entity(document.build()).build();
287     }

290      * Export a document to PDF.
291      *
292      * @api {get} /document/:id/pdf Export a document to PDF
293      * @apiName GetDocumentPdf
294      * @apiGroup Document
295      * @apiParam {String} id Document ID
296      * @apiParam {String} share Share ID
297      * @apiParam {Boolean} metadata If true, export metadata
298      * @apiParam {Boolean} fitimagetopage If true, fit the images to pages
299      * @apiParam {Number} margin Margin around the pages, in millimeter
300      * @apiSuccess {String} pdf The whole response is the PDF file
301      * @apiError (client) NotFound Document not found
302      * @apiError (client) ValidationError Validation error
303      * @apiPermission none
304      * @apiVersion 1.5.0
305      *
306      * @param documentId Document ID
307      * @param shareId Share ID
308      * @param metadata Export metadata
309      * @param fitImageToPage Fit images to page
310      * @param marginStr Margins
311      * @return Response
312      */
313     @GET
314     @Path("{id: [a-z0-9\\-]+}/pdf")
315     public Response getPdf(
316             @PathParam("id") String documentId,
317             @QueryParam("share") String shareId,
318             final @QueryParam("metadata") Boolean metadata,
319             final @QueryParam("fitimagetopage") Boolean fitImageToPage,
320             @QueryParam("margin") String marginStr) {
321         authenticate();
322 
323         // Validate input
324         final int margin = ValidationUtil.validateInteger(marginStr, "margin");
325 
326         // Get document and check read permission
327         DocumentDao documentDao = new DocumentDao();
328         final DocumentDto documentDto = documentDao.getDocument(documentId, PermType.READ, getTargetIdList(shareId));
329         if (documentDto == null) {
330             throw new NotFoundException();
331         }
332 
333         // Get files
334         FileDao fileDao = new FileDao();
335         UserDao userDao = new UserDao();
336         final List<File> fileList = fileDao.getByDocumentId(null, documentId);
337         for (File file : fileList) {
338             // A file is always encrypted by the creator of it
339             // Store its private key to decrypt it
340             User user = userDao.getById(file.getUserId());
341             file.setPrivateKey(user.getPrivateKey());
342         }
343 
344         // Convert to PDF
345         StreamingOutput stream = outputStream -> {
346             try {
347                 PdfUtil.convertToPdf(documentDto, fileList, fitImageToPage, metadata, margin, outputStream);
348             } catch (Exception e) {
349                 throw new IOException(e);
350             }
351         };
352 
353         return Response.ok(stream)
354                 .header("Content-Type", MimeType.APPLICATION_PDF)
355                 .header("Content-Disposition", "inline; filename=\"" + documentDto.getTitle() + ".pdf\"")
356                 .build();
357     }

429     public Response list(
430             @QueryParam("limit") Integer limit,
431             @QueryParam("offset") Integer offset,
432             @QueryParam("sort_column") Integer sortColumn,
433             @QueryParam("asc") Boolean asc,
434             @QueryParam("search") String search,
435             @QueryParam("files") Boolean files,
436 
437             @QueryParam("search[after]") String searchCreatedAfter,
438             @QueryParam("search[before]") String searchCreatedBefore,
439             @QueryParam("search[by]") String searchBy,
440             @QueryParam("search[full]") String searchFull,
441             @QueryParam("search[lang]") String searchLang,
442             @QueryParam("search[mime]") String searchMime,
443             @QueryParam("search[shared]") Boolean searchShared,
444             @QueryParam("search[simple]") String searchSimple,
445             @QueryParam("search[tag]") String searchTag,
446             @QueryParam("search[nottag]") String searchTagNot,
447             @QueryParam("search[title]") String searchTitle,
448             @QueryParam("search[uafter]") String searchUpdatedAfter,
449             @QueryParam("search[ubefore]") String searchUpdatedBefore,
450             @QueryParam("search[searchworkflow]") String searchWorkflow
451     ) {
452         if (!authenticate()) {
453             throw new ForbiddenClientException();
454         }
455 
456         JsonObjectBuilder response = Json.createObjectBuilder();
457         JsonArrayBuilder documents = Json.createArrayBuilder();
458 
459         TagDao tagDao = new TagDao();
460         PaginatedList<DocumentDto> paginatedList = PaginatedLists.create(limit, offset);
461         List<String> suggestionList = Lists.newArrayList();
462         SortCriteria sortCriteria = new SortCriteria(sortColumn, asc);
463 
464         List<TagDto> allTagDtoList = tagDao.findByCriteria(new TagCriteria().setTargetIdList(getTargetIdList(null)), null);
465 
466         DocumentCriteria documentCriteria = DocumentSearchCriteriaUtil.parseSearchQuery(search, allTagDtoList);
467         DocumentSearchCriteriaUtil.addHttpSearchParams(
468                 documentCriteria,
469                 searchBy,
470                 searchCreatedAfter,
471                 searchCreatedBefore,
472                 searchFull,
473                 searchLang,
474                 searchMime,
475                 searchShared,
476                 searchSimple,
477                 searchTag,
478                 searchTagNot,
479                 searchTitle,
480                 searchUpdatedAfter,
481                 searchUpdatedBefore,
482                 searchWorkflow,
483                 allTagDtoList);
484 
485         documentCriteria.setTargetIdList(getTargetIdList(null));
486         try {
487             AppContext.getInstance().getIndexingHandler().findByCriteria(paginatedList, suggestionList, documentCriteria, sortCriteria);
488         } catch (Exception e) {
489             throw new ServerException("SearchError", "Error searching in documents", e);
490         }
491 
492         // Find the files of the documents
493         Iterable<String> documentsIds = CollectionUtils.collect(paginatedList.getResultList(), DocumentDto::getId);
494         FileDao fileDao = new FileDao();
495         List<File> filesList = null;
496         Map<String, Long> filesCountByDocument = null;
497         if (Boolean.TRUE == files) {
498             filesList = fileDao.getByDocumentsIds(documentsIds);
499         } else {
500             filesCountByDocument = fileDao.countByDocumentsIds(documentsIds);
501         }
502 
503         for (DocumentDto documentDto : paginatedList.getResultList()) {
504             // Get tags accessible by the current user on this document
505             List<TagDto> tagDtoList = tagDao.findByCriteria(new TagCriteria()
506                     .setTargetIdList(getTargetIdList(null))
507                     .setDocumentId(documentDto.getId()), new SortCriteria(1, true));
508 
509             Long filesCount;
510             Collection<File> filesOfDocument = null;
511             if (Boolean.TRUE == files) {
512                 // Find files matching the document
513                 filesOfDocument = CollectionUtils.select(filesList, file -> file.getDocumentId().equals(documentDto.getId()));
514                 filesCount = (long) filesOfDocument.size();
515             } else {
516                 filesCount = filesCountByDocument.getOrDefault(documentDto.getId(), 0L);
517             }
518 
519             JsonObjectBuilder documentObjectBuilder = createDocumentObjectBuilder(documentDto)
520                     .add("active_route", documentDto.isActiveRoute())
521                     .add("current_step_name", JsonUtil.nullable(documentDto.getCurrentStepName()))

888      * @apiError (client) QuotaReached Quota limit reached
889      * @apiError (server) FileError Error adding a file

892      *
893      * @param fileBodyPart File to import
894      * @return Response
895      */
896     @PUT
897     @Path("eml")
898     @Consumes("multipart/form-data")
899     public Response importEml(@FormDataParam("file") FormDataBodyPart fileBodyPart) {
900         if (!authenticate()) {
901             throw new ForbiddenClientException();
902         }
903 
904         // Validate input data
905         ValidationUtil.validateRequired(fileBodyPart, "file");
906 
907         // Save the file to a temporary file
908         java.nio.file.Path unencryptedFile;
909         try {
910             unencryptedFile = AppContext.getInstance().getFileService().createTemporaryFile();
911             Files.copy(fileBodyPart.getValueAs(InputStream.class), unencryptedFile, StandardCopyOption.REPLACE_EXISTING);
912         } catch (IOException e) {
913             throw new ServerException("StreamError", "Error reading the input file", e);
914         }
915 
916         // Read the EML file
917         Properties props = new Properties();
918         Session mailSession = Session.getDefaultInstance(props, null);
919         EmailUtil.MailContent mailContent = new EmailUtil.MailContent();
920         try (InputStream inputStream = Files.newInputStream(unencryptedFile)) {
921             Message message = new MimeMessage(mailSession, inputStream);
922             mailContent.setSubject(message.getSubject());
923             mailContent.setDate(message.getSentDate());
924             EmailUtil.parseMailContent(message, mailContent);
925         } catch (IOException | MessagingException e) {
926             throw new ServerException("StreamError", "Error reading the temporary file", e);
927         }
928 
929         // Create the document
930         Document document = new Document();
931         document.setUserId(principal.getId());
932         if (mailContent.getSubject() == null) {
933             document.setTitle("Imported email from EML file");
934         } else {
935             document.setTitle(StringUtils.abbreviate(mailContent.getSubject(), 100));
936         }
937         document.setDescription(StringUtils.abbreviate(mailContent.getMessage(), 4000));
938         document.setSubject(StringUtils.abbreviate(mailContent.getSubject(), 500));
939         document.setFormat("EML");
940         document.setSource("Email");
941         document.setLanguage(ConfigUtil.getConfigStringValue(ConfigType.DEFAULT_LANGUAGE));
942         if (mailContent.getDate() == null) {
943             document.setCreateDate(new Date());
944         } else {
945             document.setCreateDate(mailContent.getDate());
946         }
947 
948         // Save the document, create the base ACLs
949         DocumentUtil.createDocument(document, principal.getId());
950 
951         // Raise a document created event
952         DocumentCreatedAsyncEvent documentCreatedAsyncEvent = new DocumentCreatedAsyncEvent();
953         documentCreatedAsyncEvent.setUserId(principal.getId());
954         documentCreatedAsyncEvent.setDocumentId(document.getId());
955         ThreadLocalContext.get().addAsyncEvent(documentCreatedAsyncEvent);
956 
957         // Add files to the document
958         try {
959             for (EmailUtil.FileContent fileContent : mailContent.getFileContentList()) {
960                 FileUtil.createFile(fileContent.getName(), null, fileContent.getFile(), fileContent.getSize(),
961                         document.getLanguage(), principal.getId(), document.getId());
962             }
963         } catch (IOException e) {
964             throw new ClientException(e.getMessage(), e.getMessage(), e);
965         } catch (Exception e) {
966             throw new ServerException("FileError", "Error adding a file", e);
967         }
968 
969         JsonObjectBuilder response = Json.createObjectBuilder()
970                 .add("id", document.getId());
971         return Response.ok().entity(response.build()).build();
