39      /**
40       * Add an ACL.
41       *
42       * @api {put} /acl Add an ACL
43       * @apiName PutAcl
44       * @apiGroup Acl
45       * @apiParam {String} source Source ID
46       * @apiParam {String="READ","WRITE"} perm Permission
47       * @apiParam {String} target Target ID
48       * @apiParam {String="USER","GROUP","SHARE"} type Target type
49       * @apiSuccess {String} id Acl ID
50       * @apiSuccess {String} perm Permission
51       * @apiSuccess {String} name Target name
52       * @apiSuccess {String="USER","GROUP","SHARE"} type Target type
53       * @apiError (client) ForbiddenError Access denied
54       * @apiError (client) ValidationError Validation error
55       * @apiError (client) InvalidTarget This target does not exist
56       * @apiPermission user
57       * @apiVersion 1.5.0
58       *
59       * @param sourceId Source ID
60       * @param permStr Permission
61       * @param targetName Target name
62       * @param typeStr ACL type
63       * @return Response
64       */
65      @PUT
66      public Response add(@FormParam("source") String sourceId,
67              @FormParam("perm") String permStr,
68              @FormParam("target") String targetName,
69              @FormParam("type") String typeStr) {
70          if (!authenticate()) {
71              throw new ForbiddenClientException();
72          }
73          
74          // Validate input
75          ValidationUtil.validateRequired(sourceId, "source");
76          PermType perm = PermType.valueOf(ValidationUtil.validateLength(permStr, "perm", 1, 30, false));
77          AclTargetType type = AclTargetType.valueOf(ValidationUtil.validateLength(typeStr, "type", 1, 10, false));
78          targetName = ValidationUtil.validateLength(targetName, "target", 1, 50, false);
79          
80          // Does a target has been found?
81          String targetId = SecurityUtil.getTargetIdFromName(targetName, type);
82          if (targetId == null) {
83              throw new ClientException("InvalidTarget", MessageFormat.format("This target does not exist: {0}", targetName));
84          }
85          
86          // Check permission on the source by the principal
87          AclDao aclDao = new AclDao();
88          if (!aclDao.checkPermission(sourceId, PermType.WRITE, getTargetIdList(null))) {
89              throw new ForbiddenClientException();
90          }
91          
92          // Create the ACL
93          Acl acl = new Acl();
94          acl.setSourceId(sourceId);
95          acl.setPerm(perm);
96          acl.setTargetId(targetId);
97          acl.setType(AclType.USER);
98          
99          // Avoid duplicates
100         if (!aclDao.checkPermission(acl.getSourceId(), acl.getPerm(), Lists.newArrayList(acl.getTargetId()))) {
101             aclDao.create(acl, principal.getId());
102 
103             // Raise an ACL created event
104             AclCreatedAsyncEvent event = new AclCreatedAsyncEvent();
105             event.setUserId(principal.getId());
106             event.setSourceId(sourceId);
107             event.setPerm(perm);
108             event.setTargetId(targetId);
109             ThreadLocalContext.get().addAsyncEvent(event);
110 
111             // Returns the ACL
112             JsonObjectBuilder response = Json.createObjectBuilder()
113                     .add("perm", acl.getPerm().name())
114                     .add("id", acl.getTargetId())
115                     .add("name", targetName)
116                     .add("type", type.name());
117             return Response.ok().entity(response.build()).build();
118         }
119         
120         return Response.ok().entity(Json.createObjectBuilder().build()).build();
121     }

123     /**
124      * Deletes an ACL.
125      *
126      * @api {delete} /acl/:source/:perm/:target Delete an ACL
127      * @apiName DeleteAcl
128      * @apiGroup Acl
129      * @apiParam {String} source Source ID
130      * @apiParam {String="READ","WRITE"} perm Permission
131      * @apiParam {String} target Target ID
132      * @apiSuccess {String} status Status OK
133      * @apiError (client) ForbiddenError Access denied
134      * @apiError (client) ValidationError Validation error
135      * @apiError (client) AclError Cannot delete base ACL on a document or a tag
136      * @apiPermission user
137      * @apiVersion 1.5.0
138      * 
139      * @param sourceId Source ID
140      * @param permStr Permission
141      * @param targetId Target ID
142      * @return Response
143      */
144     @DELETE
145     @Path("{sourceId: [a-z0-9\\-]+}/{perm: [A-Z]+}/{targetId: [a-z0-9\\-]+}")
146     public Response delete(
147             @PathParam("sourceId") String sourceId,
148             @PathParam("perm") String permStr,
149             @PathParam("targetId") String targetId) {
150         if (!authenticate()) {
151             throw new ForbiddenClientException();
152         }
153         
154         // Validate input
155         ValidationUtil.validateRequired(sourceId, "source");
156         PermType perm = PermType.valueOf(ValidationUtil.validateLength(permStr, "perm", 1, 30, false));
157         ValidationUtil.validateRequired(targetId, "target");
158 
159         // Check permission on the source by the principal
160         AclDao aclDao = new AclDao();
161         if (!aclDao.checkPermission(sourceId, PermType.WRITE, getTargetIdList(null))) {
162             throw new ForbiddenClientException();
163         }
164         
165         // Cannot delete R/W on a source document if the target is the creator
166         DocumentDao documentDao = new DocumentDao();
167         Document document = documentDao.getById(sourceId);
168         if (document != null && document.getUserId().equals(targetId)) {
169             throw new ClientException("AclError", "Cannot delete base ACL on a document");
170         }
171 
172         // Cannot delete R/W on a source tag if the target is the creator
173         TagDao tagDao = new TagDao();
174         Tag tag = tagDao.getById(sourceId);
175         if (tag != null && tag.getUserId().equals(targetId)) {
176             throw new ClientException("AclError", "Cannot delete base ACL on a tag");
177         }
178 
179         // Delete the ACL
180         aclDao.delete(sourceId, perm, targetId, principal.getId(), AclType.USER);
181 
182         // Raise an ACL deleted event
183         AclDeletedAsyncEvent event = new AclDeletedAsyncEvent();
184         event.setUserId(principal.getId());
185         event.setSourceId(sourceId);
186         event.setPerm(perm);
187         event.setTargetId(targetId);
188         ThreadLocalContext.get().addAsyncEvent(event);
189         
190         // Always return OK
191         JsonObjectBuilder response = Json.createObjectBuilder()
192                 .add("status", "ok");
193         return Response.ok().entity(response.build()).build();
194     }

196     /**
197      * Search possible ACL target.
198      *
199      * @api {get} /acl/target/search Search in ACL targets
200      * @apiName GetAclTargetSearch
201      * @apiGroup Acl
202      * @apiParam {String} search Search query
203      * @apiSuccess {Object[]} users List of users
204      * @apiSuccess {String} users.name Username
205      * @apiSuccess {Object[]} groups List of groups
206      * @apiSuccess {String} groups.name Group name
207      * @apiError (client) ForbiddenError Access denied
208      * @apiError (client) ValidationError Validation error
209      * @apiPermission user
210      * @apiVersion 1.5.0
211      * 
212      * @param search Search query
213      * @return Response
214      */
215     @GET
216     @Path("target/search")
217     public Response targetList(@QueryParam("search") String search) {
218         if (!authenticate()) {
219             throw new ForbiddenClientException();
220         }
221         
222         // Validate input
223         search = ValidationUtil.validateLength(search, "search", 1, 50, false);
224         
225         // Search users
226         UserDao userDao = new UserDao();
227         JsonArrayBuilder users = Json.createArrayBuilder();
228         SortCriteria sortCriteria = new SortCriteria(1, true);
229         List<UserDto> userDtoList = userDao.findByCriteria(new UserCriteria().setSearch(search), sortCriteria);
230         for (UserDto userDto : userDtoList) {
231             // No need to add users who will skip ACL check anyways
232             if (!SecurityUtil.skipAclCheck(Lists.newArrayList(userDto.getId()))) {
233                 users.add(Json.createObjectBuilder()
234                         .add("name", userDto.getUsername()));
235             }
236         }
237         
238         // Search groups
239         GroupDao groupDao = new GroupDao();
240         JsonArrayBuilder groups = Json.createArrayBuilder();
241         List<GroupDto> groupDtoList = groupDao.findByCriteria(new GroupCriteria().setSearch(search), sortCriteria);
242         for (GroupDto groupDto : groupDtoList) {
243             // No need to add users who will skip ACL check anyways
244             if (!SecurityUtil.skipAclCheck(Lists.newArrayList(groupDto.getId()))) {
245                 groups.add(Json.createObjectBuilder()
246                         .add("name", groupDto.getName()));
247             }
248         }
249         
250         JsonObjectBuilder response = Json.createObjectBuilder()
251                 .add("users", users)
252                 .add("groups", groups);
253         return Response.ok().entity(response.build()).build();
254     }
