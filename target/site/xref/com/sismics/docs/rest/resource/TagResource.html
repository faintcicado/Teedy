35      /**
36       * Returns the list of all visible tags.
37       *
38       * @api {get} /tag/list Get tags
39       * @apiName GetTagList
40       * @apiGroup Tag
41       * @apiSuccess {Object[]} tags List of tags
42       * @apiSuccess {String} tags.id ID
43       * @apiSuccess {String} tags.name Name
44       * @apiSuccess {String} tags.color Color
45       * @apiSuccess {String} tags.parent Parent
46       * @apiError (client) ForbiddenError Access denied
47       * @apiPermission user
48       * @apiVersion 1.5.0
49       *
50       * @return Response
51       */
52      @GET
53      @Path("/list")
54      public Response list() {
55          if (!authenticate()) {
56              throw new ForbiddenClientException();
57          }
58          
59          TagDao tagDao = new TagDao();
60          List<TagDto> tagDtoList = tagDao.findByCriteria(new TagCriteria().setTargetIdList(getTargetIdList(null)), new SortCriteria(1, true));
61  
62          // Extract tag IDs
63          Set<String> tagIdSet = Sets.newHashSet();
64          for (TagDto tagDto : tagDtoList) {
65              tagIdSet.add(tagDto.getId());
66          }
67  
68          // Build the response
69          JsonArrayBuilder items = Json.createArrayBuilder();
70          for (TagDto tagDto : tagDtoList) {
71              JsonObjectBuilder item = Json.createObjectBuilder()
72                      .add("id", tagDto.getId())
73                      .add("name", tagDto.getName())
74                      .add("color", tagDto.getColor());
75              if (tagIdSet.contains(tagDto.getParentId())) {
76                  item.add("parent", tagDto.getParentId());
77              }
78              items.add(item);
79          }
80          
81          JsonObjectBuilder response = Json.createObjectBuilder()
82                  .add("tags", items);
83          return Response.ok().entity(response.build()).build();
84      }

86      /**
87       * Returns a tag.
88       *
89       * @api {get} /tag/:id Get a tag
90       * @apiName GetTag
91       * @apiGroup Tag
92       * @apiSuccess {String} id ID
93       * @apiSuccess {String} name Name
94       * @apiSuccess {String} creator Username of the creator
95       * @apiSuccess {String} color Color
96       * @apiSuccess {String} parent Parent
97       * @apiSuccess {Boolean} writable True if the tag is writable by the current user
98       * @apiSuccess {Object[]} acls List of ACL
99       * @apiSuccess {String} acls.id ID
100      * @apiSuccess {String="READ","WRITE"} acls.perm Permission
101      * @apiSuccess {String} acls.name Target name
102      * @apiSuccess {String="USER","GROUP","SHARE"} acls.type Target type
103      * @apiError (client) ForbiddenError Access denied
104      * @apiError (client) NotFound Tag not found
105      * @apiPermission user
106      * @apiVersion 1.5.0
107      *
108      * @param id Tag ID
109      * @return Response
110      */
111     @GET
112     @Path("{id: [a-z0-9\\-]+}")
113     public Response get(@PathParam("id") String id) {
114         if (!authenticate()) {
115             throw new ForbiddenClientException();
116         }
117 
118         TagDao tagDao = new TagDao();
119         List<TagDto> tagDtoList = tagDao.findByCriteria(new TagCriteria().setTargetIdList(getTargetIdList(null)).setId(id), null);
120         if (tagDtoList.isEmpty()) {
121             throw new NotFoundException();
122         }
123 
124         // Add tag informatiosn
125         TagDto tagDto = tagDtoList.get(0);
126         JsonObjectBuilder tag = Json.createObjectBuilder()
127                 .add("id", tagDto.getId())
128                 .add("creator", tagDto.getCreator())
129                 .add("name", tagDto.getName())
130                 .add("color", tagDto.getColor());
131 
132         // Add the parent if its visible
133         if (tagDto.getParentId() != null) {
134             AclDao aclDao = new AclDao();
135             if (aclDao.checkPermission(tagDto.getParentId(), PermType.READ, getTargetIdList(null))) {
136                 tag.add("parent", tagDto.getParentId());
137             }
138         }
139 
140         // Add ACL
141         AclUtil.addAcls(tag, id, getTargetIdList(null));
142 
143         return Response.ok().entity(tag.build()).build();
144     }

146     /**
147      * Creates a new tag.
148      *
149      * @api {put} /tag Create a tag
150      * @apiName PutTag
151      * @apiGroup Tag
152      * @apiParam {String} name Name
153      * @apiParam {String} color Color
154      * @apiParam {String} parent Parent ID
155      * @apiSuccess {String} id Tag ID
156      * @apiError (client) ForbiddenError Access denied
157      * @apiError (client) ValidationError Validation error
158      * @apiError (client) IllegalTagName Spaces and colons are not allowed in tag name
159      * @apiError (client) ParentNotFound Parent not found
160      * @apiPermission user
161      * @apiVersion 1.5.0
162      *
163      * @param name Name
164      * @param color Color
165      * @param parentId Parent ID
166      * @return Response
167      */
168     @PUT
169     public Response add(
170             @FormParam("name") String name,
171             @FormParam("color") String color,
172             @FormParam("parent") String parentId) {
173         if (!authenticate()) {
174             throw new ForbiddenClientException();
175         }
176         
177         // Validate input data
178         name = ValidationUtil.validateLength(name, "name", 1, 36, false);
179         ValidationUtil.validateHexColor(color, "color", true);
180         ValidationUtil.validateTagName(name);
181 
182         // Check the parent
183         if (StringUtils.isEmpty(parentId)) {
184             parentId = null;
185         } else {
186             AclDao aclDao = new AclDao();
187             if (!aclDao.checkPermission(parentId, PermType.READ, getTargetIdList(null))) {
188                 throw new ClientException("ParentNotFound", MessageFormat.format("Parent not found: {0}", parentId));
189             }
190         }
191 
192         // Create the tag
193         TagDao tagDao = new TagDao();
194         Tag tag = new Tag();
195         tag.setName(name);
196         tag.setColor(color);
197         tag.setUserId(principal.getId());
198         tag.setParentId(parentId);
199         String id = tagDao.create(tag, principal.getId());
200 
201         // Create read ACL
202         AclDao aclDao = new AclDao();
203         Acl acl = new Acl();
204         acl.setPerm(PermType.READ);
205         acl.setType(AclType.USER);
206         acl.setSourceId(id);
207         acl.setTargetId(principal.getId());
208         aclDao.create(acl, principal.getId());
209 
210         // Create write ACL
211         acl = new Acl();
212         acl.setPerm(PermType.WRITE);
213         acl.setType(AclType.USER);
214         acl.setSourceId(id);
215         acl.setTargetId(principal.getId());
216         aclDao.create(acl, principal.getId());
217         
218         JsonObjectBuilder response = Json.createObjectBuilder()
219                 .add("id", id);
220         return Response.ok().entity(response.build()).build();
221     }

223     /**
224      * Update a tag.
225      *
226      * @api {post} /tag/:id Update a tag
227      * @apiName PostTag
228      * @apiGroup Tag
229      * @apiParam {String} id Tag ID
230      * @apiParam {String} name Name
231      * @apiParam {String} color Color
232      * @apiParam {String} parent Parent ID
233      * @apiSuccess {String} id Tag ID
234      * @apiError (client) ForbiddenError Access denied
235      * @apiError (client) ValidationError Validation error
236      * @apiError (client) IllegalTagName Spaces and colons are not allowed in tag name
237      * @apiError (client) ParentNotFound Parent not found
238      * @apiError (client) CircularReference Circular reference in parent tag
239      * @apiError (client) NotFound Tag not found
240      * @apiPermission user
241      * @apiVersion 1.5.0
242      *
243      * @param name Name
244      * @param color Color
245      * @param parentId Parent ID
246      * @return Response
247      */
248     @POST
249     @Path("{id: [a-z0-9\\-]+}")
250     public Response update(
251             @PathParam("id") String id,
252             @FormParam("name") String name,
253             @FormParam("color") String color,
254             @FormParam("parent") String parentId) {
255         if (!authenticate()) {
256             throw new ForbiddenClientException();
257         }
258         
259         // Validate input data
260         name = ValidationUtil.validateLength(name, "name", 1, 36, true);
261         ValidationUtil.validateHexColor(color, "color", true);
262         ValidationUtil.validateTagName(name);
263 
264         // Check permission
265         AclDao aclDao = new AclDao();
266         if (!aclDao.checkPermission(id, PermType.WRITE, getTargetIdList(null))) {
267             throw new NotFoundException();
268         }
269         
270         // Check the parent
271         TagDao tagDao = new TagDao();
272         if (StringUtils.isEmpty(parentId)) {
273             parentId = null;
274         } else {
275             if (!aclDao.checkPermission(parentId, PermType.READ, getTargetIdList(null))) {
276                 throw new ClientException("ParentNotFound", MessageFormat.format("Parent not found: {0}", parentId));
277             }
278 
279             String parentTagId = parentId;
280             do {
281                 Tag parentTag = tagDao.getById(parentTagId);
282                 parentTagId = parentTag.getParentId();
283                 if (parentTag.getId().equals(id)) {
284                     throw new ClientException("CircularReference", "Circular reference in parent tag");
285                 }
286             } while (parentTagId != null);
287         }
288 
289         // Update the tag
290         Tag tag = tagDao.getById(id);
291         if (!StringUtils.isEmpty(name)) {
292             tag.setName(name);
293         }
294         if (!StringUtils.isEmpty(color)) {
295             tag.setColor(color);
296         }
297         // Parent tag is always updated to have the possibility to delete it
298         tag.setParentId(parentId);
299         
300         tagDao.update(tag, principal.getId());
301         
302         JsonObjectBuilder response = Json.createObjectBuilder()
303                 .add("id", id);
304         return Response.ok().entity(response.build()).build();
305     }

307     /**
308      * Delete a tag.
309      *
310      * @api {delete} /tag/:id Delete a tag
311      * @apiName DeleteTag
312      * @apiGroup Tag
313      * @apiParam {String} id Tag ID
314      * @apiSuccess {String} status Status OK
315      * @apiError (client) ForbiddenError Access denied
316      * @apiError (client) NotFound Tag not found
317      * @apiPermission user
318      * @apiVersion 1.5.0
319      * 
320      * @param id Tag ID
321      * @return Response
322      */
323     @DELETE
324     @Path("{id: [a-z0-9\\-]+}")
325     public Response delete(
326             @PathParam("id") String id) {
327         if (!authenticate()) {
328             throw new ForbiddenClientException();
329         }
330         
331         // Get the tag
332         AclDao aclDao = new AclDao();
333         if (!aclDao.checkPermission(id, PermType.WRITE, getTargetIdList(null))) {
334             throw new NotFoundException();
335         }
336 
337         // Delete the tag
338         TagDao tagDao = new TagDao();
339         tagDao.delete(id, principal.getId());
340         
341         // Always return OK
342         JsonObjectBuilder response = Json.createObjectBuilder()
343                 .add("status", "ok");
344         return Response.ok().entity(response.build()).build();
345     }
