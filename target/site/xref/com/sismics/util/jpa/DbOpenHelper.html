30  /**
31   * A helper to update the database incrementally.
32   *
33   * @author jtremeaux
34   */
35  abstract class DbOpenHelper {
36      /**
37       * Logger.
38       */
39      private static final Logger log = LoggerFactory.getLogger(DbOpenHelper.class);
40  
41      private final JdbcConnectionAccess jdbcConnectionAccess;
42  
43      private final List<Exception> exceptions = new ArrayList<>();
44  
45      private Formatter formatter;
46  
47      private Statement stmt;
48  
49      DbOpenHelper(ServiceRegistry serviceRegistry) throws HibernateException {
50          final JdbcServices jdbcServices = serviceRegistry.getService(JdbcServices.class);
51          SqlStatementLogger sqlStatementLogger = jdbcServices.getSqlStatementLogger();
52          jdbcConnectionAccess = jdbcServices.getBootstrapJdbcConnectionAccess();
53          formatter = (sqlStatementLogger.isFormat() ? FormatStyle.DDL : FormatStyle.NONE).getFormatter();
54      }
55  
56      public void open() {
57          log.info("Opening database and executing incremental updates");
58  
59          Connection connection = null;
60  
61          exceptions.clear();
62  
63          try {
64              try {
65                  connection = jdbcConnectionAccess.obtainConnection();
66              } catch (SQLException sqle) {
67                  exceptions.add(sqle);
68                  log.error("Unable to get database metadata", sqle);
69                  throw sqle;
70              }
71  
72              // Check if database is already created
73              Integer oldVersion = null;
74              try {
75                  stmt = connection.createStatement();
76                  ResultSet result = stmt.executeQuery("select c.CFG_VALUE_C from T_CONFIG c where c.CFG_ID_C='DB_VERSION'");
77                  if (result.next()) {
78                      String oldVersionStr = result.getString(1);
79                      oldVersion = Integer.parseInt(oldVersionStr);
80                  }
81              } catch (Exception e) {
82                  if (DialectUtil.isObjectNotFound(e.getMessage())) {
83                      log.info("Unable to get database version: Table T_CONFIG not found");
84                  } else {
85                      log.error("Unable to get database version", e);
86                  }
87              } finally {
88                  connection.commit();
89                  if (stmt != null) {
90                      stmt.close();
91                      stmt = null;
92                  }
93              }
94  
95              stmt = connection.createStatement();
96              if (oldVersion == null) {
97                  // Execute creation script
98                  log.info("Executing initial schema creation script");
99                  onCreate();
100                 oldVersion = 0;
101             }
102 
103             // Execute update script
104             ResourceBundle configBundle = ConfigUtil.getConfigBundle();
105             Integer currentVersion = Integer.parseInt(configBundle.getString("db.version"));
106             log.info(MessageFormat.format("Found database version {0}, new version is {1}, executing database incremental update scripts", oldVersion, currentVersion));
107             onUpgrade(oldVersion, currentVersion);
108             log.info("Database upgrade complete");
109         } catch (Exception e) {
110             exceptions.add(e);
111             log.error("Unable to complete schema update", e);
112         } finally {
113             try {
114                 connection.commit();
115                 if (stmt != null) {
116                     stmt.close();
117                     stmt = null;
118                 }
119                 jdbcConnectionAccess.releaseConnection(connection);
120             } catch (Exception e) {
121                 exceptions.add(e);
122                 log.error("Unable to close connection", e);
123             }
124         }
125     }

127     /**
128      * Execute all upgrade scripts in ascending order for a given version.
129      *
130      * @param version Version number
131      * @throws Exception e
132      */
133     void executeAllScript(final int version) throws Exception {
134         List<String> fileNameList = ResourceUtil.list(getClass(), "/db/update/", (dir, name) -> {
135             String versionString = String.format("%03d", version);
136             return name.matches("dbupdate-" + versionString + "-\\d+\\.sql");
137         });
138         Collections.sort(fileNameList);
139 
140         for (String fileName : fileNameList) {
141             if (log.isInfoEnabled()) {
142                 log.info(MessageFormat.format("Executing script: {0}", fileName));
143             }
144             InputStream is = getClass().getResourceAsStream("/db/update/" + fileName);
145             executeScript(is);
146         }
147     }
148 
149     /**
150      * Execute a SQL script. All statements must be one line only.
151      *
152      * @param inputScript Script to execute
153      * @throws IOException e
154      */
155     private void executeScript(InputStream inputScript) throws IOException {
156         List<String> lines = CharStreams.readLines(new InputStreamReader(inputScript));
157 
158         for (String sql : lines) {
159             if (Strings.isNullOrEmpty(sql) || sql.startsWith("--")) {
160                 continue;
161             }
162 
163             String transformed = DialectUtil.transform(sql);
164             if (transformed != null) {
165                 String formatted = formatter.format(transformed);
166                 try {
167                     log.debug(formatted);
168                     stmt.executeUpdate(formatted);
169                 } catch (SQLException e) {
170                     exceptions.add(e);
171                     if (log.isErrorEnabled()) {
172                         log.error("Error executing SQL statement: {}", sql);
173                         log.error(e.getMessage());
174                     }
175                 }
176             }
177         }
178     }

188      */

190         return exceptions;
191     }

194      * Format the output SQL statements.
195      *
196      * @param format True to format
197      */
198     public void setFormat(boolean format) {
199         this.formatter = (format ? FormatStyle.DDL : FormatStyle.NONE).getFormatter();
200     }
