4    * Dialect utilities.

6    * @author jtremeaux

8   public class DialectUtil {
9       /**
10       * Checks if the error from the drivers relates to an object not found.
11       *
12       * @param message Error message
13       * @return Object not found
14       */
15      public static boolean isObjectNotFound(String message) {
16          return EMF.isDriverH2() && message.contains("not found") ||
17                  EMF.isDriverPostgresql() && message.contains("does not exist");
18      }
19  
20  
21      /**
22       * Transform SQL dialect to current dialect.
23       *
24       * @param sql SQL to transform
25       * @return Transformed SQL
26       */
27      public static String transform(String sql) {
28          if (sql.startsWith("!PGSQL!")) {
29              return EMF.isDriverH2() ? null : sql.substring(7);
30          }
31          if (sql.startsWith("!H2!")) {
32              return EMF.isDriverPostgresql() ? null : sql.substring(4);
33          }
34  
35          if (EMF.isDriverPostgresql()) {
36              sql = transformToPostgresql(sql);
37          }
38          return sql;
39      }
40  
41      /**
42       * Transform SQL from HSQLDB dialect to current dialect.
43       *
44       * @param sql SQL to transform
45       * @return Transformed SQL
46       */
47      public static String transformToPostgresql(String sql) {
48          sql = sql.replaceAll("(cached|memory) table", "table");
49          sql = sql.replaceAll("datetime", "timestamp");
50          sql = sql.replaceAll("longvarchar", "text");
51          sql = sql.replaceAll("bit default 1", "bool default true");
52          sql = sql.replaceAll("bit default 0", "bool default false");
53          sql = sql.replaceAll("bit not null default 1", "bool not null default true");
54          sql = sql.replaceAll("bit not null default 0", "bool not null default false");
55          sql = sql.replaceAll("bit not null", "bool not null");
56          return sql;
57      }
