24  /**
25   * Filter used to process a couple things in the request context.
26   * 
27   * @author jtremeaux
28   */
29  public class RequestContextFilter implements Filter {
30      /**
31       * Logger.
32       */
33      private static final Logger log = LoggerFactory.getLogger(RequestContextFilter.class);
34  
35      @Override
36      public void init(FilterConfig filterConfig) {
37          // Initialize the app directory
38          if (!filterConfig.getServletContext().getServerInfo().startsWith("Grizzly")) {
39              EnvironmentUtil.setWebappContext(true);
40          }
41          try {
42              if (log.isInfoEnabled()) {
43                  log.info(MessageFormat.format("Using base data directory: {0}", DirectoryUtil.getBaseDataDirectory()));
44              }
45          } catch (Exception e) {
46              log.error("Error initializing base data directory", e);
47          }
48          
49          
50          // Initialize file logger
51          RollingFileAppender fileAppender = new RollingFileAppender();
52          fileAppender.setName("FILE");
53          fileAppender.setFile(DirectoryUtil.getLogDirectory().resolve("docs.log").toString());
54          fileAppender.setLayout(new PatternLayout("%d{DATE} %p %l %m %n"));
55          fileAppender.setThreshold(Level.INFO);
56          fileAppender.setAppend(true);
57          fileAppender.setMaxFileSize("5MB");
58          fileAppender.setMaxBackupIndex(5);
59          fileAppender.activateOptions();
60          org.apache.log4j.Logger.getRootLogger().addAppender(fileAppender);
61          SLF4JBridgeHandler.removeHandlersForRootLogger();
62          SLF4JBridgeHandler.install();
63          
64          // Initialize the application context
65          TransactionUtil.handle(AppContext::getInstance);
66      }
67  
68      @Override
69      public void destroy() {
70          AppContext.getInstance().shutDown();
71      }
72  
73      @Override
74      public void doFilter(ServletRequest request, ServletResponse response, FilterChain filterChain) throws IOException, ServletException {
75          EntityManager em;
76          
77          try {
78              em = EMF.get().createEntityManager();
79          } catch (Exception e) {
80              throw new ServletException("Cannot create entity manager", e);
81          }
82          ThreadLocalContext context = ThreadLocalContext.get();
83          context.setEntityManager(em);
84          EntityTransaction tx = em.getTransaction();
85          tx.begin();
86          
87          try {
88              addCacheHeaders(response);
89              filterChain.doFilter(request, response);
90          } catch (Exception e) {
91              ThreadLocalContext.cleanup();
92              
93              // IOException are thrown if the client closes the connection before completion
94              if (!(e instanceof IOException)) {
95                  log.error("An exception occured, rolling back current transaction", e);
96  
97                  // If an unprocessed error comes up from the application layers (Jersey...), rollback the transaction (should not happen)
98                  if (em.isOpen()) {
99                      if (em.getTransaction() != null && em.getTransaction().isActive()) {
100                         em.getTransaction().rollback();
101                     }
102                     
103                     try {
104                         em.close();
105                     } catch (Exception ce) {
106                         log.error("Error closing entity manager", ce);
107                     }
108                 }
109                 throw new ServletException(e);
110             }
111         }
112 
113         // No error processing the request : commit / rollback the current transaction depending on the HTTP code
114         if (em.isOpen()) {
115             if (em.getTransaction() != null && em.getTransaction().isActive()) {
116                 HttpServletResponse r = (HttpServletResponse) response;
117                 int statusClass = r.getStatus() / 100;
118                 if (statusClass == 2 || statusClass == 3) {
119                     try {
120                         em.getTransaction().commit();
121                     } catch (Exception e) {
122                         log.error("Error during commit", e);
123                         r.sendError(500);
124                     }
125                 } else {
126                     em.getTransaction().rollback();
127                 }
128                 
129                 try {
130                     em.close();
131                 } catch (Exception e) {
132                     log.error("Error closing entity manager", e);
133                 }
134             }
135         }
136 
137         // Fire all pending async events after request transaction commit.
138         // This way, all modifications done during this request are available in the listeners.
139         context.fireAllAsyncEvents();
140 
141         ThreadLocalContext.cleanup();
142     }
143 
144     /**
145      * Add no-cache header.
146      *
147      * @param response Response
148      */
149     private void addCacheHeaders(ServletResponse response) {
150         HttpServletResponse r = (HttpServletResponse) response;
151         r.addHeader(HttpHeaders.CACHE_CONTROL, "no-cache");
152         r.addHeader(HttpHeaders.EXPIRES, "0");
153     }
