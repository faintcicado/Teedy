36  /**
37   * This filter is used to authenticate the user having an active session by validating a jwt token.
38   * The filter extracts the jwt token stored from Authorization header.
39   * It validates the token by calling an Identity Broker like KeyCloak.
40   * If validated, the user is retrieved, and the filter injects a UserPrincipal into the request attribute.
41   *
42   * @author smitra
43   */
44  public class JwtBasedSecurityFilter extends SecurityFilter {
45      private static final Logger log = LoggerFactory.getLogger(JwtBasedSecurityFilter.class);
46      private static final okhttp3.OkHttpClient client = new okhttp3.OkHttpClient();
47      /**
48       * Name of the header used to store the authentication token.
49       */
50      public static final String HEADER_NAME = "Authorization";
51      /**
52       * True if this authentication method is enabled.
53       */
54      private boolean enabled;
55  
56      @Override
57      public void init(FilterConfig filterConfig) {
58          enabled = Boolean.parseBoolean(filterConfig.getInitParameter("enabled"))
59                  || Boolean.parseBoolean(System.getProperty("docs.jwt_authentication"));
60      }
61  
62      @Override
63      protected User authenticate(final HttpServletRequest request) {
64          if (!enabled) {
65              return null;
66          }
67          log.info("Jwt authentication started");
68          User user = null;
69          String token = extractAuthToken(request).replace("Bearer ", "");
70          DecodedJWT jwt = JWT.decode(token);
71          if (verifyJwt(jwt, token)) {
72              String email = jwt.getClaim("preferred_username").toString();
73              UserDao userDao = new UserDao();
74              user = userDao.getActiveByUsername(email);
75              if (user == null) {
76                  user = new User();
77                  user.setRoleId(Constants.DEFAULT_USER_ROLE);
78                  user.setUsername(email);
79                  user.setEmail(email);
80                  user.setStorageQuota(Long.parseLong(ofNullable(System.getenv(Constants.GLOBAL_QUOTA_ENV))
81                          .orElse("1073741824")));
82                  user.setPassword(UUID.randomUUID().toString());
83                  try {
84                      userDao.create(user, email);
85                      log.info("user created");
86                  } catch (Exception e) {
87                      log.info("Error:" + e.getMessage());
88                      return null;
89                  }
90              }
91          }
92          return user;
93      }
94  
95      private boolean verifyJwt(final DecodedJWT jwt, final String token) {
96  
97          try {
98              buildJWTVerifier(jwt).verify(token);
99              // if token is valid no exception will be thrown
100             log.info("Valid TOKEN");
101             return Boolean.TRUE;
102         } catch (CertificateException e) {
103             //if CertificateException comes from buildJWTVerifier()
104             log.info("InValid TOKEN: " + e.getMessage());
105             return Boolean.FALSE;
106         } catch (JWTVerificationException e) {
107             // if JWT Token in invalid
108             log.info("InValid TOKEN: " + e.getMessage() );
109             return Boolean.FALSE;
110         } catch (Exception e) {
111             // If any other exception comes
112             log.info("InValid TOKEN, Exception Occurred: " + e.getMessage());
113             return Boolean.FALSE;
114         }
115     }
116 
117     private String extractAuthToken(final HttpServletRequest request) {
118         return ofNullable(request.getHeader("Authorization")).orElse("");
119     }
120 
121     private RSAPublicKey getPublicKey(DecodedJWT jwt) {
122         String jwtIssuerCerts = jwt.getIssuer() + "/protocol/openid-connect/certs";
123         String publicKey = "";
124         RSAPublicKey rsaPublicKey = null;
125         Request request = new Request.Builder()
126                 .url(jwtIssuerCerts)
127                 .get()
128                 .build();
129         try (Response response = client.newCall(request).execute()) {
130             log.info("Successfully called the jwt issuer at: " + jwtIssuerCerts + " - " + response.code());
131             assert response.body() != null;
132             if (response.isSuccessful()) {
133                 try (Reader reader = response.body().charStream()) {
134                     try (JsonReader jsonReader = Json.createReader(reader)) {
135                         JsonObject jwks = jsonReader.readObject();
136                         JsonArray keys = jwks.getJsonArray("keys");
137                         publicKey = keys.stream().filter(key -> Objects.equals(key.asJsonObject().getString("kid"),
138                                         jwt.getKeyId()))
139                                 .findFirst()
140                                 .map(k -> k.asJsonObject().getJsonArray("x5c").getString(0))
141                                 .orElse("");
142                         var decode = Base64.getDecoder().decode(publicKey);
143                         var certificate = CertificateFactory.getInstance("X.509")
144                                 .generateCertificate(new ByteArrayInputStream(decode));
145                         rsaPublicKey = (RSAPublicKey) certificate.getPublicKey();
146                     }
147                 }
148             }
149         } catch (IOException e) {
150             log.error("Error calling the jwt issuer at: " + jwtIssuerCerts, e);
151         } catch (CertificateException e) {
152             log.error("Error in getting the certificate: ", e);
153         }
154         return rsaPublicKey;
155     }
156 
157     private JWTVerifier buildJWTVerifier(DecodedJWT jwt) throws CertificateException {
158         var algo = Algorithm.RSA256(getPublicKey(jwt), null);
159         return JWT.require(algo).build();
160     }
