24      /**
25       * Write a high quality JPEG.
26       * 
27       * @param image Image
28       * @param outputStream Output stream
29       * @throws IOException e
30       */
31      public static void writeJpeg(BufferedImage image, OutputStream outputStream) throws IOException {
32          Iterator<ImageWriter> iter = ImageIO.getImageWritersByFormatName("jpeg");
33          ImageWriter writer = null;
34          ImageOutputStream imageOutputStream = null;
35          try {
36              writer = iter.next();
37              ImageWriteParam iwp = writer.getDefaultWriteParam();
38              iwp.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
39              iwp.setCompressionQuality(1.f);
40              imageOutputStream = ImageIO.createImageOutputStream(outputStream);
41              writer.setOutput(imageOutputStream);
42  
43              if (image.getColorModel().hasAlpha()) {
44                  // Strip alpha channel
45                  BufferedImage noAlphaImage = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.TYPE_INT_RGB);
46                  Graphics graphics = noAlphaImage.getGraphics();
47                  graphics.setColor(Color.WHITE);
48                  graphics.fillRect(0, 0, image.getWidth(), image.getHeight());
49                  graphics.drawImage(image, 0, 0, null);
50                  image = noAlphaImage;
51              }
52  
53              IIOImage iioImage = new IIOImage(image, null, null);
54              writer.write(null, iioImage, iwp);
55          } finally {
56              if (imageOutputStream != null) {
57                  try {
58                      imageOutputStream.close();
59                  } catch (Exception inner) {
60                      // NOP
61                  }
62              }
63              if (writer != null) {
64                  writer.dispose();
65              }
66          }
67      }

69      /**
70       * Compute Gravatar hash.
71       * See https://en.gravatar.com/site/implement/hash/.
72       * 
73       * @param email Email
74       * @return Gravatar hash
75       */
76      @SuppressWarnings("deprecation") // Gravatar uses MD5, nothing we can do about it
77      public static String computeGravatar(String email) {
78          if (email == null) {
79              return null;
80          }
81  
82          return Hashing.md5().hashString(
83                  email.trim().toLowerCase(), StandardCharsets.UTF_8)
84                  .toString();
85      }

87      /**
88       * Return true if a pixel is black.
89       *
90       * @param image Image
91       * @param x X
92       * @param y Y
93       * @return True if black
94       */
95      public static boolean isBlack(BufferedImage image, int x, int y) {
96          if (image.getType() == BufferedImage.TYPE_BYTE_BINARY) {
97              WritableRaster raster = image.getRaster();
98              int pixelRGBValue = raster.getSample(x, y, 0);
99              return pixelRGBValue == 0;
100         }
101 
102         int luminanceValue = 140;
103         return isBlack(image, x, y, luminanceValue);
104     }
105 
106     /**
107      * Return true if a pixel is black.
108      *
109      * @param image Image
110      * @param x X
111      * @param y Y
112      * @param luminanceCutOff Luminance cutoff
113      * @return True if black
114      */
115     private static boolean isBlack(BufferedImage image, int x, int y, int luminanceCutOff) {
116         int pixelRGBValue;
117         int r;
118         int g;
119         int b;
120         double luminance = 0.0;
121 
122         // return white on areas outside of image boundaries
123         if (x < 0 || y < 0 || x > image.getWidth() || y > image.getHeight()) {
124             return false;
125         }
126 
127         try {
128             pixelRGBValue = image.getRGB(x, y);
129             r = (pixelRGBValue >> 16) & 0xff;
130             g = (pixelRGBValue >> 8) & 0xff;
131             b = (pixelRGBValue) & 0xff;
132             luminance = (r * 0.299) + (g * 0.587) + (b * 0.114);
133         } catch (Exception e) {
134             // NOP
135         }
136 
137         return luminance < luminanceCutOff;
138     }
