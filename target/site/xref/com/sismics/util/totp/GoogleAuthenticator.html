42  /**
43   * This class implements the functionality described in RFC 6238 (TOTP: Time
44   * based one-time password algorithm) and has been tested again Google's
45   * implementation of such algorithm in its Google Authenticator application.
46   * <p/>
47   * This class lets users create a new 16-bit base32-encoded secret key with the
48   * validation code calculated at {@code time = 0} (the UNIX epoch) and the URL
49   * of a Google-provided QR barcode to let an user load the generated information
50   * into Google Authenticator.
51   * <p/>
52   * The random number generator used by this class uses the default algorithm and
53   * provider. Users can override them by setting the following system properties
54   * to the algorithm and provider name of their choice:
55   * <ul>
56   * <li>{@link #RNG_ALGORITHM}.</li>
57   * <li>{@link #RNG_ALGORITHM_PROVIDER}.</li>
58   * </ul>
59   * <p/>
60   * This class does not store in any way either the generated keys nor the keys
61   * passed during the authorization process.
62   * <p/>
63   * Java Server side class for Google Authenticator's TOTP generator was inspired
64   * by an author's blog post.
65   *
66   * @author Enrico M. Crisostomo
67   * @author Warren Strange
68   * @version 0.5.0
69   * @see <a href=
70   *      "http://thegreyblog.blogspot.com/2011/12/google-authenticator-using-it-in-your.html"
71   *      />
72   * @see <a href="http://code.google.com/p/google-authenticator" />
73   * @see <a href="http://tools.ietf.org/id/draft-mraihi-totp-timebased-06.txt" />
74   * @since 0.3.0
75   */

93      /**
94       * The number of bits of a secret key in binary form. Since the Base32
95       * encoding with 8 bit characters introduces an 160% overhead, we just need
96       * 80 bits (10 bytes) to generate a 16 bytes Base32-encoded secret key.

100     /**
101      * Number of scratch codes to generate during the key generation. We are
102      * using Google's default of providing 5 scratch codes.

104     private static final int SCRATCH_CODES = 5;

106     /**
107      * Number of digits of a scratch code represented as a decimal integer.

111     /**
112      * Modulus used to truncate the scratch code.
113      */
114     private static final int SCRATCH_CODE_MODULUS = (int) Math.pow(10, SCRATCH_CODE_LENGTH);

116     /**
117      * Magic number representing an invalid scratch code.
118      */
119     private static final int SCRATCH_CODE_INVALID = -1;

121     /**
122      * Length in bytes of each scratch code. We're using Google's default of
123      * using 4 bytes per scratch code.

185     public int calculateCode(String secret, long tm) {
186         return calculateCode(decodeKey(secret), tm);
187     }

189     /**
190      * Decode a secret key in raw bytes.
191      * 
192      * @param secret Secret key
193      * @return Raw bytes
194      */
195     private byte[] decodeKey(String secret) {
196         switch (config.getKeyRepresentation()) {
197         case BASE32:
198             Base32 codec32 = new Base32();
199             return codec32.decode(secret);
200         case BASE64:
201             Base64 codec64 = new Base64();
202             return  codec64.decode(secret);
203         default:
204             throw new IllegalArgumentException("Unknown key representation type.");
205         }
206     }
207     
208     /**
209      * Calculates the verification code of the provided key at the specified
210      * instant of time using the algorithm specified in RFC 6238.
211      *
212      * @param key the secret key in binary format.
213      * @param tm the instant of time.
214      * @return the validation code for the provided key at the specified instant
215      *         of time.
216      */
217     private int calculateCode(byte[] key, long tm) {
218         // Allocating an array of bytes to represent the specified instant
219         // of time.
220         byte[] data = new byte[8];
221         long value = tm;
222 
223         // Converting the instant of time from the long representation to a
224         // big-endian array of bytes (RFC4226, 5.2. Description).
225         for (int i = 8; i-- > 0; value >>>= 8) {
226             data[i] = (byte) value;
227         }
228 
229         // Building the secret key specification for the HmacSHA1 algorithm.
230         SecretKeySpec signKey = new SecretKeySpec(key, HMAC_HASH_FUNCTION);
231 
232         try {
233             // Getting an HmacSHA1 algorithm implementation from the JCE.
234             Mac mac = Mac.getInstance(HMAC_HASH_FUNCTION);
235 
236             // Initializing the MAC algorithm.
237             mac.init(signKey);
238 
239             // Processing the instant of time and getting the encrypted data.
240             byte[] hash = mac.doFinal(data);
241 
242             // Building the validation code performing dynamic truncation
243             // (RFC4226, 5.3. Generating an HOTP value)
244             int offset = hash[hash.length - 1] & 0xF;
245 
246             // We are using a long because Java hasn't got an unsigned integer
247             // type
248             // and we need 32 unsigned bits).
249             long truncatedHash = 0;
250 
251             for (int i = 0; i < 4; ++i) {
252                 truncatedHash <<= 8;
253 
254                 // Java bytes are signed but we need an unsigned integer:
255                 // cleaning off all but the LSB.
256                 truncatedHash |= (hash[offset + i] & 0xFF);
257             }
258 
259             // Clean bits higher than the 32nd (inclusive) and calculate the
260             // module with the maximum validation code value.
261             truncatedHash &= 0x7FFFFFFF;
262             truncatedHash %= config.getKeyModulus();
263 
264             // Returning the validation code to the caller.
265             return (int) truncatedHash;
266         } catch (NoSuchAlgorithmException | InvalidKeyException ex) {
267             // We're not disclosing internal error details to our clients.
268             throw new GoogleAuthenticatorException("The operation cannot be performed now.", ex);
269         }
270     }
271 
272     /**
273      * This method implements the algorithm specified in RFC 6238 to check if a
274      * validation code is valid in a given instant of time for the given secret
275      * key.
276      *
277      * @param secret the Base32 encoded secret key.
278      * @param code the code to validate.
279      * @param timestamp the instant of time to use during the validation process.
280      * @param window the window size to use during the validation process.
281      * @return <code>true</code> if the validation code is valid,
282      *         <code>false</code> otherwise.
283      */
284     private boolean checkCode(String secret, long code, long timestamp, int window) {
285         // Decoding the secret key to get its raw byte representation.
286         byte[] decodedKey = decodeKey(secret);
287 
288         // convert unix time into a 30 second "window" as specified by the
289         // TOTP specification. Using Google's default interval of 30 seconds.
290         final long timeWindow = timestamp / this.config.getTimeStepSizeInMillis();
291 
292         // Calculating the verification code of the given key in each of the
293         // time intervals and returning true if the provided code is equal to
294         // one of them.
295         for (int i = -((window - 1) / 2); i <= window / 2; ++i) {
296             // Calculating the verification code for the current time interval.
297             long hash = calculateCode(decodedKey, timeWindow + i);
298 
299             // Checking if the provided code is equal to the calculated one.
300             if (hash == code) {
301                 // The verification code is valid.
302                 return true;
303             }
304         }
305 
306         // The verification code is invalid.
307         return false;
308     }

310     public GoogleAuthenticatorKey createCredentials() {
311         // Allocating a buffer sufficiently large to hold the bytes required by
312         // the secret key and the scratch codes.
313         byte[] buffer = new byte[SECRET_BITS / 8 + SCRATCH_CODES * BYTES_PER_SCRATCH_CODE];
314 
315         secureRandom.nextBytes(buffer);
316 
317         // Extracting the bytes making up the secret key.
318         byte[] secretKey = Arrays.copyOf(buffer, SECRET_BITS / 8);
319         String generatedKey = calculateSecretKey(secretKey);
320 
321         // Generating the verification code at time = 0.
322         int validationCode = calculateValidationCode(secretKey);
323 
324         // Calculate scratch codes
325         List<Integer> scratchCodes = calculateScratchCodes(buffer);
326 
327         return new GoogleAuthenticatorKey(generatedKey, validationCode, scratchCodes);
328     }
329 
330     private List<Integer> calculateScratchCodes(byte[] buffer) {
331         List<Integer> scratchCodes = new ArrayList<>();
332 
333         while (scratchCodes.size() < SCRATCH_CODES) {
334             byte[] scratchCodeBuffer = Arrays.copyOfRange(buffer, SECRET_BITS / 8 + BYTES_PER_SCRATCH_CODE * scratchCodes.size(), SECRET_BITS / 8 + BYTES_PER_SCRATCH_CODE * scratchCodes.size() + BYTES_PER_SCRATCH_CODE);
335 
336             int scratchCode = calculateScratchCode(scratchCodeBuffer);
337 
338             if (scratchCode != SCRATCH_CODE_INVALID) {
339                 scratchCodes.add(scratchCode);
340             } else {
341                 scratchCodes.add(generateScratchCode());
342             }
343         }
344 
345         return scratchCodes;
346     }

348     /**
349      * This method calculates a scratch code from a random byte buffer of
350      * suitable size <code>#BYTES_PER_SCRATCH_CODE</code>.
351      *
352      * @param scratchCodeBuffer a random byte buffer whose minimum size is <code>#BYTES_PER_SCRATCH_CODE</code>.
353      * @return the scratch code.
354      */
355     private int calculateScratchCode(byte[] scratchCodeBuffer) {
356         if (scratchCodeBuffer.length < BYTES_PER_SCRATCH_CODE) {
357             throw new IllegalArgumentException(String.format("The provided random byte buffer is too small: %d.", scratchCodeBuffer.length));
358         }
359 
360         int scratchCode = 0;
361 
362         for (int i = 0; i < BYTES_PER_SCRATCH_CODE; ++i) {
363             scratchCode = (scratchCode << 8) + (scratchCodeBuffer[i] & 0xff);
364         }
365 
366         scratchCode = (scratchCode & 0x7FFFFFFF) % SCRATCH_CODE_MODULUS;
367 
368         // Accept the scratch code only if it has exactly
369         // SCRATCH_CODE_LENGTH digits.
370         if (scratchCode >= SCRATCH_CODE_MODULUS / 10) {
371             return scratchCode;
372         } else {
373             return SCRATCH_CODE_INVALID;
374         }
375     }
376 
377     /**
378      * This method creates a new random byte buffer from which a new scratch
379      * code is generated. This function is invoked if a scratch code generated
380      * from the main buffer is invalid because it does not satisfy the scratch
381      * code restrictions.
382      *
383      * @return A valid scratch code.
384      */
385     private int generateScratchCode() {
386         while (true) {
387             byte[] scratchCodeBuffer = new byte[BYTES_PER_SCRATCH_CODE];
388             secureRandom.nextBytes(scratchCodeBuffer);
389 
390             int scratchCode = calculateScratchCode(scratchCodeBuffer);
391 
392             if (scratchCode != SCRATCH_CODE_INVALID) {
393                 return scratchCode;
394             }
395         }
396     }

398     /**
399      * This method calculates the validation code at time 0.
400      *
401      * @param secretKey The secret key to use.
402      * @return the validation code at time 0.
403      */
404     private int calculateValidationCode(byte[] secretKey) {
405         return calculateCode(secretKey, 0);
406     }

408     /**
409      * This method calculates the secret key given a random byte buffer.
410      *
411      * @param secretKey a random byte buffer.
412      * @return the secret key.
413      */
414     private String calculateSecretKey(byte[] secretKey) {
415         switch (config.getKeyRepresentation()) {
416         case BASE32:
417             return new Base32().encodeToString(secretKey);
418         case BASE64:
419             return new Base64().encodeToString(secretKey);
420         default:
421             throw new IllegalArgumentException("Unknown key representation type.");
422         }
423     }

425     public boolean authorize(String secret, int verificationCode) throws GoogleAuthenticatorException {
426         return authorize(secret, verificationCode, new Date().getTime());
427     }
428 
429     private boolean authorize(String secret, int verificationCode, long time) throws GoogleAuthenticatorException {
430         // Checking user input and failing if the secret key was not provided.
431         if (secret == null) {
432             throw new IllegalArgumentException("Secret cannot be null.");
433         }
434 
435         // Checking if the verification code is between the legal bounds.
436         if (verificationCode <= 0 || verificationCode >= this.config.getKeyModulus()) {
437             return false;
438         }
439 
440         // Checking the validation code using the current UNIX time.
441         return checkCode(secret, verificationCode, time, this.config.getWindowSize());
442     }
