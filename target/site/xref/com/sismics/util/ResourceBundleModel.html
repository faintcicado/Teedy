5   import freemarker.template.TemplateModel;

8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.MissingResourceException;
11  import java.util.ResourceBundle;
12  
13  /**
14   * Override of {@link freemarker.ext.beans.ResourceBundleModel}
15   * to threat single quotes uniformely.
16   * 
17   * @author bgamard
18   */
19  public class ResourceBundleModel extends freemarker.ext.beans.ResourceBundleModel {
20  
21      /**
22       * Default constructor.
23       * 
24       * @param bundle Resource bundle
25       * @param wrapper Beans wrapper
26       */
27      public ResourceBundleModel(ResourceBundle bundle, BeansWrapper wrapper) {
28          super(bundle, wrapper);
29      }
30  
31      @SuppressWarnings("rawtypes")
32      @Override
33      public Object exec(List arguments) throws TemplateModelException {
34          // Must have at least one argument - the key
35          if (arguments.size() < 1)
36              throw new TemplateModelException("No message key was specified");
37          // Read it
38          Iterator it = arguments.iterator();
39          String key = unwrap((TemplateModel) it.next()).toString();
40          try {
41              // Copy remaining arguments into an Object[]
42              int args = arguments.size() - 1;
43              Object[] params = new Object[args];
44              for (int i = 0; i < args; ++i)
45                  params[i] = unwrap((TemplateModel) it.next());
46  
47              // Invoke format
48              return new StringModel(format(key, params), wrapper);
49          } catch (MissingResourceException e) {
50              throw new TemplateModelException("No such key: " + key);
51          } catch (Exception e) {
52              throw new TemplateModelException(e.getMessage());
53          }
54      }
55  }
