1   package com.sismics.rest.exception;
2   
3   import jakarta.ws.rs.WebApplicationException;
4   import jakarta.ws.rs.core.Response;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   import jakarta.json.Json;
9   
10  /**
11   * Jersey exception encapsulating an error from the client (BAD_REQUEST).
12   *
13   * @author jtremeaux
14   */
15  public class ClientException extends WebApplicationException {
16      /**
17       * Serial UID.
18       */
19      private static final long serialVersionUID = 1L;
20  
21      /**
22       * Logger.
23       */
24      private static final Logger log = LoggerFactory.getLogger(ClientException.class);
25  
26      /**
27       * Constructor of ClientException.
28       * 
29       * @param type Error type (e.g. AlreadyExistingEmail, ValidationError)
30       * @param message Human readable error message
31       * @param e Readable error message
32       */
33      public ClientException(String type, String message, Exception e) {
34          this(type, message);
35          log.error(type + ": " + message, e);
36      }
37  
38      /**
39       * Constructor of ClientException.
40       * 
41       * @param type Error type (e.g. AlreadyExistingEmail, ValidationError)
42       * @param message Human readable error message
43       */
44      public ClientException(String type, String message) {
45          super(Response.status(Response.Status.BAD_REQUEST).entity(Json.createObjectBuilder()
46              .add("type", type)
47              .add("message", message).build()).build());
48      }
49  }
