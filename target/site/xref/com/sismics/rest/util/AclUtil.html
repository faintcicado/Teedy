14  /**
15   * Acl utilities.
16   *
17   * @author bgamard
18   */
19  public class AclUtil {
20      /**
21       * Add ACLs to a JSON response.
22       *
23       * @param json JSON
24       * @param sourceId Source ID
25       * @param targetIdList List of target ID
26       */
27      public static void addAcls(JsonObjectBuilder json, String sourceId, List<String> targetIdList) {
28          AclDao aclDao = new AclDao();
29          List<AclDto> aclDtoList = aclDao.getBySourceId(sourceId, AclType.USER);
30          JsonArrayBuilder aclList = Json.createArrayBuilder();
31          for (AclDto aclDto : aclDtoList) {
32              aclList.add(Json.createObjectBuilder()
33                      .add("perm", aclDto.getPerm().name())
34                      .add("id", aclDto.getTargetId())
35                      .add("name", JsonUtil.nullable(aclDto.getTargetName()))
36                      .add("type", aclDto.getTargetType()));
37          }
38          json.add("acls", aclList)
39                  .add("writable", aclDao.checkPermission(sourceId, PermType.WRITE, targetIdList));
40      }
41  }
