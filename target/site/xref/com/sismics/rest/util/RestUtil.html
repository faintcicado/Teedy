15  /**
16   * Rest utilities.
17   *
18   * @author bgamard

20  public class RestUtil {
21      /**
22       * Transform a File into its JSON representation.
23       * If the file size it is not stored in the database the size can be wrong
24       * because the encrypted file size is used.
25       * @param fileDb a file
26       * @return the JSON
27       */
28      public static JsonObjectBuilder fileToJsonObjectBuilder(File fileDb) {
29          try {
30              long fileSize = fileDb.getSize().equals(File.UNKNOWN_SIZE) ? Files.size(DirectoryUtil.getStorageDirectory().resolve(fileDb.getId())) : fileDb.getSize();
31              return Json.createObjectBuilder()
32                      .add("id", fileDb.getId())
33                      .add("processing", FileUtil.isProcessingFile(fileDb.getId()))
34                      .add("name", JsonUtil.nullable(fileDb.getName()))
35                      .add("version", fileDb.getVersion())
36                      .add("mimetype", fileDb.getMimeType())
37                      .add("document_id", JsonUtil.nullable(fileDb.getDocumentId()))
38                      .add("create_date", fileDb.getCreateDate().getTime())
39                      .add("size", fileSize);
40          } catch (IOException e) {
41              throw new ServerException("FileError", "Unable to get the size of " + fileDb.getId(), e);
42          }
43      }
44  }
