14   * 
15   * @author bgamard

17  public class TestMimeTypeUtil extends BaseTest {
18      @Test
19      public void test() throws Exception {
20          // Detect ODT files
21          Path path = Paths.get(getResource(FILE_ODT).toURI());
22          Assert.assertEquals(MimeType.OPEN_DOCUMENT_TEXT, MimeTypeUtil.guessMimeType(path, FILE_ODT));
23  
24          // Detect DOCX files
25          path = Paths.get(getResource(FILE_DOCX).toURI());
26          Assert.assertEquals(MimeType.OFFICE_DOCUMENT, MimeTypeUtil.guessMimeType(path, FILE_ODT));
27  
28          // Detect PPTX files
29          path = Paths.get(getResource(FILE_PPTX).toURI());
30          Assert.assertEquals(MimeType.OFFICE_PRESENTATION, MimeTypeUtil.guessMimeType(path, FILE_PPTX));
31  
32          // Detect XLSX files
33          path = Paths.get(getResource(FILE_XLSX).toURI());
34          Assert.assertEquals(MimeType.OFFICE_SHEET, MimeTypeUtil.guessMimeType(path, FILE_XLSX));
35  
36          // Detect TXT files
37          path = Paths.get(getResource(FILE_TXT).toURI());
38          Assert.assertEquals(MimeType.TEXT_PLAIN, MimeTypeUtil.guessMimeType(path, FILE_TXT));
39  
40          // Detect CSV files
41          path = Paths.get(getResource(FILE_CSV).toURI());
42          Assert.assertEquals(MimeType.TEXT_CSV, MimeTypeUtil.guessMimeType(path, FILE_CSV));
43  
44          // Detect PDF files
45          path = Paths.get(getResource(FILE_PDF).toURI());
46          Assert.assertEquals(MimeType.APPLICATION_PDF, MimeTypeUtil.guessMimeType(path, FILE_PDF));
47  
48          // Detect JPEG files
49          path = Paths.get(getResource(FILE_JPG).toURI());
50          Assert.assertEquals(MimeType.IMAGE_JPEG, MimeTypeUtil.guessMimeType(path, FILE_JPG));
51  
52          // Detect GIF files
53          path = Paths.get(getResource(FILE_GIF).toURI());
54          Assert.assertEquals(MimeType.IMAGE_GIF, MimeTypeUtil.guessMimeType(path, FILE_GIF));
55  
56          // Detect PNG files
57          path = Paths.get(getResource(FILE_PNG).toURI());
58          Assert.assertEquals(MimeType.IMAGE_PNG, MimeTypeUtil.guessMimeType(path, FILE_PNG));
59  
60          // Detect ZIP files
61          path = Paths.get(getResource(FILE_ZIP).toURI());
62          Assert.assertEquals(MimeType.APPLICATION_ZIP, MimeTypeUtil.guessMimeType(path, FILE_ZIP));
63  
64          // Detect WEBM files
65          path = Paths.get(getResource(FILE_WEBM).toURI());
66          Assert.assertEquals(MimeType.VIDEO_WEBM, MimeTypeUtil.guessMimeType(path, FILE_WEBM));
67  
68          // Detect MP4 files
69          path = Paths.get(getResource(FILE_MP4).toURI());
70          Assert.assertEquals(MimeType.VIDEO_MP4, MimeTypeUtil.guessMimeType(path, FILE_MP4));
71      }
