23  /**
24   * Exhaustive test of the share resource.
25   * 
26   * @author bgamard
27   */
28  public class TestShareResource extends BaseJerseyTest {
29      /**
30       * Test the share resource.
31       * 
32       * @throws Exception
33       */
34      @Test
35      public void testShareResource() throws Exception {
36          // Login share1
37          clientUtil.createUser("share1");
38          String share1Token = clientUtil.login("share1");
39          
40          // Create a document
41          JsonObject json = target().path("/document").request()
42                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share1Token)
43                  .put(Entity.form(new Form()
44                          .param("title", "File test document 1")
45                          .param("language", "eng")), JsonObject.class);
46          String document1Id = json.getString("id");
47          Assert.assertNotNull(document1Id);
48          
49          // Add a file
50          String file1Id;
51          try (InputStream is = Resources.getResource("file/PIA00452.jpg").openStream()) {
52              StreamDataBodyPart streamDataBodyPart = new StreamDataBodyPart("file", is, "PIA00452.jpg");
53              try (FormDataMultiPart multiPart = new FormDataMultiPart()) {
54                  json = target()
55                          .register(MultiPartFeature.class)
56                          .path("/file").request()
57                          .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share1Token)
58                          .put(Entity.entity(multiPart.field("id", document1Id).bodyPart(streamDataBodyPart),
59                                  MediaType.MULTIPART_FORM_DATA_TYPE), JsonObject.class);
60                  file1Id = json.getString("id");
61              }
62          }
63          
64          // Share this document
65          json = target().path("/share").request()
66                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share1Token)
67                  .put(Entity.form(new Form()
68                          .param("id", document1Id)
69                          .param("name", "4 All")), JsonObject.class);
70          String share1Id = json.getString("id");
71          
72          // Get the document anonymously
73          json = target().path("/document/" + document1Id)
74                  .queryParam("share", share1Id)
75                  .request()
76                  .get(JsonObject.class);
77          Assert.assertEquals(document1Id, json.getString("id"));
78          Assert.assertEquals(3, json.getJsonArray("acls").size()); // 2 for the creator, 1 for the share
79  
80          // Get all comments from this document anonymously
81          json = target().path("/comment/" + document1Id)
82                  .queryParam("share", share1Id)
83                  .request()
84                  .get(JsonObject.class);
85          JsonArray comments = json.getJsonArray("comments");
86          Assert.assertEquals(0, comments.size());
87          
88          // Get all files from this document anonymously
89          json = target().path("/file/list")
90                  .queryParam("id", document1Id)
91                  .queryParam("share", share1Id)
92                  .request()
93                  .get(JsonObject.class);
94          JsonArray files = json.getJsonArray("files");
95          Assert.assertEquals(1, files.size());
96          
97          // Get the file data anonymously
98          Response response = target().path("/file/" + file1Id + "/data")
99                  .queryParam("thumbnail", false)
100                 .queryParam("share", share1Id)
101                 .request()
102                 .get();
103         InputStream is = (InputStream) response.getEntity();
104         byte[] fileBytes = ByteStreams.toByteArray(is);
105         Assert.assertEquals(163510, fileBytes.length);
106         
107         // Deletes the share (not allowed)
108         clientUtil.createUser("share2");
109         String share2Token = clientUtil.login("share2");
110         response = target().path("/share/" + share1Id).request()
111                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share2Token)
112                 .delete();
113         Assert.assertEquals(Status.BAD_REQUEST, Status.fromStatusCode(response.getStatus()));
114         json = response.readEntity(JsonObject.class);
115         Assert.assertEquals("DocumentNotFound", json.getString("type"));
116         
117         // Deletes the share
118         json = target().path("/share/" + share1Id).request()
119                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share1Token)
120                 .delete(JsonObject.class);
121         Assert.assertEquals("ok", json.getString("status"));
122 
123         // Deletes the share again
124         response = target().path("/share/" + share1Id).request()
125                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, share1Token)
126                 .delete();
127         Assert.assertEquals(Status.BAD_REQUEST, Status.fromStatusCode(response.getStatus()));
128         json = response.readEntity(JsonObject.class);
129         Assert.assertEquals("ShareNotFound", json.getString("type"));
130 
131         // Deletes share1 and share 2
132         String adminToken = adminToken();
133         target().path("/user/share1").request()
134                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
135                 .delete();
136         target().path("/user/share2").request()
137                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
138                 .delete();
139     }
