6   import org.junit.Test;

9   import jakarta.json.JsonObject;
10  import jakarta.ws.rs.client.Entity;
11  import jakarta.ws.rs.core.Form;
12  import java.util.Date;
13  
14  
15  /**
16   * Test the webhook resource.
17   * 
18   * @author bgamard
19   */
20  public class TestWebhookResource extends BaseJerseyTest {
21      /**
22       * Test the webhook resource.
23       */
24      @Test
25      public void testWebhookResource() {
26          // Login admin
27          String adminToken = adminToken();
28  
29          // Login webhook1
30          clientUtil.createUser("webhook1");
31          String webhook1Token = clientUtil.login("webhook1");
32          
33          // Get all webhooks
34          JsonObject json = target().path("/webhook")
35                  .request()
36                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
37                  .get(JsonObject.class);
38          JsonArray webhooks = json.getJsonArray("webhooks");
39          Assert.assertEquals(0, webhooks.size());
40  
41          // Create a webhook
42          target().path("/webhook").request()
43                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
44                  .put(Entity.form(new Form()
45                          .param("event", "DOCUMENT_CREATED")
46                          .param("url", "http://localhost:" + getPort() + "/docs/thirdpartywebhook")), JsonObject.class);
47  
48          // Create a document
49          json = target().path("/document").request()
50                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, webhook1Token)
51                  .put(Entity.form(new Form()
52                          .param("title", "Webhook document 1")
53                          .param("language", "eng")
54                          .param("create_date", Long.toString(new Date().getTime()))), JsonObject.class);
55          String document1Id = json.getString("id");
56  
57          // Check the webhook payload
58          JsonObject payload = ThirdPartyWebhookResource.getLastPayload();
59          Assert.assertEquals("DOCUMENT_CREATED", payload.getString("event"));
60          Assert.assertEquals(document1Id, payload.getString("id"));
61  
62          // Get all webhooks
63          json = target().path("/webhook")
64                  .request()
65                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
66                  .get(JsonObject.class);
67          webhooks = json.getJsonArray("webhooks");
68          Assert.assertEquals(1, webhooks.size());
69          JsonObject webhook = webhooks.getJsonObject(0);
70          String webhookId = webhook.getString("id");
71          Assert.assertEquals("DOCUMENT_CREATED", webhook.getString("event"));
72          Assert.assertEquals("http://localhost:" + getPort() + "/docs/thirdpartywebhook", webhook.getString("url"));
73          Assert.assertNotNull(webhook.getJsonNumber("create_date"));
74  
75          // Delete a webhook
76          target().path("/webhook/" + webhookId).request()
77                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
78                  .delete(JsonObject.class);
79  
80          // Get all webhooks
81          json = target().path("/webhook")
82                  .request()
83                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
84                  .get(JsonObject.class);
85          webhooks = json.getJsonArray("webhooks");
86          Assert.assertEquals(0, webhooks.size());
87  
88          // Deletes webhook1
89          target().path("/user/webhook1").request()
90                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
91                  .delete();
92      }
93  }
