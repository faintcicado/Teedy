1   package com.sismics.docs.rest;
2   
3   import java.util.Date;
4   
5   import jakarta.json.JsonObject;
6   import jakarta.ws.rs.client.Entity;
7   import jakarta.ws.rs.core.Form;
8   import jakarta.ws.rs.core.Response;
9   import jakarta.ws.rs.core.Response.Status;
10  
11  import org.junit.Assert;
12  import org.junit.Test;
13  
14  import com.sismics.util.filter.TokenBasedSecurityFilter;
15  
16  /**
17   * Exhaustive test of the comment resource.
18   * 
19   * @author bgamard
20   */
21  public class TestCommentResource extends BaseJerseyTest {
22      /**
23       * Test the comment resource.
24       */
25      @Test
26      public void testCommentResource() {
27          // Login comment1
28          clientUtil.createUser("comment1");
29          String comment1Token = clientUtil.login("comment1");
30          
31          // Login comment2
32          clientUtil.createUser("comment2");
33          String comment2Token = clientUtil.login("comment2");
34          
35          // Create a document with comment1
36          long create1Date = new Date().getTime();
37          JsonObject json = target().path("/document").request()
38                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
39                  .put(Entity.form(new Form()
40                          .param("title", "My super title document 1")
41                          .param("description", "My super description for document 1")
42                          .param("language", "eng")
43                          .param("create_date", Long.toString(create1Date))), JsonObject.class);
44          String document1Id = json.getString("id");
45          Assert.assertNotNull(document1Id);
46          
47          // Create a comment with comment2 (fail, no read access)
48          Response response = target().path("/comment").request()
49                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
50                  .put(Entity.form(new Form()
51                          .param("id", document1Id)
52                          .param("content", "Comment by comment2")));
53          Assert.assertEquals(Status.NOT_FOUND, Status.fromStatusCode(response.getStatus()));
54          
55          // Read comments with comment2 (fail, no read access)
56          response = target().path("/comment/" + document1Id).request()
57                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
58                  .get();
59          Assert.assertEquals(Status.NOT_FOUND, Status.fromStatusCode(response.getStatus()));
60          
61          // Read comments with comment 1
62          json = target().path("/comment/" + document1Id).request()
63                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
64                  .get(JsonObject.class);
65          Assert.assertEquals(0, json.getJsonArray("comments").size());
66          
67          // Create a comment with comment1
68          json = target().path("/comment").request()
69                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
70                  .put(Entity.form(new Form()
71                          .param("id", document1Id)
72                          .param("content", "Comment by comment1")), JsonObject.class);
73          String comment1Id = json.getString("id");
74          Assert.assertNotNull(comment1Id);
75          Assert.assertEquals("Comment by comment1", json.getString("content"));
76          Assert.assertEquals("comment1", json.getString("creator"));
77          Assert.assertNotNull(json.getJsonNumber("create_date"));
78          
79          // Read comments with comment1
80          json = target().path("/comment/" + document1Id).request()
81                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
82                  .get(JsonObject.class);
83          Assert.assertEquals(1, json.getJsonArray("comments").size());
84          Assert.assertEquals(comment1Id, json.getJsonArray("comments").getJsonObject(0).getString("id"));
85          
86          // Delete a comment with comment2 (fail, no write access)
87          response = target().path("/comment/" + comment1Id).request()
88                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
89                  .delete();
90          Assert.assertEquals(Status.NOT_FOUND, Status.fromStatusCode(response.getStatus()));
91          
92          // Delete a comment with comment1
93          json = target().path("/comment/" + comment1Id).request()
94                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
95                  .delete(JsonObject.class);
96          
97          // Read comments with comment1
98          json = target().path("/comment/" + document1Id).request()
99                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
100                 .get(JsonObject.class);
101         Assert.assertEquals(0, json.getJsonArray("comments").size());
102         
103         // Add an ACL READ for comment2 with comment1
104         json = target().path("/acl").request()
105                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment1Token)
106                 .put(Entity.form(new Form()
107                         .param("source", document1Id)
108                         .param("perm", "READ")
109                         .param("target", "comment2")
110                         .param("type", "USER")), JsonObject.class);
111         
112         // Create a comment with comment2
113         json = target().path("/comment").request()
114                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
115                 .put(Entity.form(new Form()
116                         .param("id", document1Id)
117                         .param("content", "Comment by comment2")), JsonObject.class);
118         String comment2Id = json.getString("id");
119         
120         // Read comments with comment2
121         json = target().path("/comment/" + document1Id).request()
122                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
123                 .get(JsonObject.class);
124         Assert.assertEquals(1, json.getJsonArray("comments").size());
125         JsonObject comment = json.getJsonArray("comments").getJsonObject(0);
126         Assert.assertEquals(comment2Id, comment.getString("id"));
127         Assert.assertEquals("Comment by comment2", comment.getString("content"));
128         Assert.assertEquals("comment2", comment.getString("creator"));
129         Assert.assertEquals("d6e56c42f61983bba80d370138763420", comment.getString("creator_gravatar"));
130         Assert.assertNotNull(comment.getJsonNumber("create_date"));
131         
132         // Delete a comment with comment2
133         json = target().path("/comment/" + comment2Id).request()
134                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
135                 .delete(JsonObject.class);
136         
137         // Read comments with comment2
138         json = target().path("/comment/" + document1Id).request()
139                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, comment2Token)
140                 .get(JsonObject.class);
141         Assert.assertEquals(0, json.getJsonArray("comments").size());
142     }
