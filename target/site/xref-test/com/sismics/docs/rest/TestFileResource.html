65                  .request()
66                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)

69          is = (InputStream) response.getEntity();

71          Assert.assertTrue(fileBytes.length > 0);
72          
73          // Get the content data
74          response = target().path("/file/" + file1Id + "/data")
75                  .queryParam("size", "content")
76                  .request()
77                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)

81          // Get the web data
82          response = target().path("/file/" + file1Id + "/data")

84                  .request()
85                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)

90          Assert.assertTrue(fileBytes.length > 0);

129         Assert.assertEquals("Pale Blue Dot", files.getJsonObject(0).getString("name"));
130

133                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
134                 .post(Entity.form(new Form()
135                         .param("id", document1Id)
136                         .param("order", file2Id)
137                         .param("order", file1Id)), JsonObject.class);
138 
139         // Get all files from a document
140         json = target().path("/file/list")
141                 .queryParam("id", document1Id)
142                 .request()
143                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
144                 .get(JsonObject.class);
145         files = json.getJsonArray("files");
146         Assert.assertEquals(2, files.size());
147         Assert.assertEquals(file2Id, files.getJsonObject(0).getString("id"));
148         Assert.assertEquals(file1Id, files.getJsonObject(1).getString("id"));
149         
150         // Get a ZIP from all files
151         response = target().path("/file/zip")
152                 .queryParam("id", document1Id)
153                 .request()
154                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
155                 .get();
156         Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
157 
158         // Deletes a file
159         json = target().path("/file/" + file1Id).request()
160                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
161                 .delete(JsonObject.class);
162         Assert.assertEquals("ok", json.getString("status"));
163         
164         // Get the file data (not found)
165         response = target().path("/file/" + file1Id + "/data").request()
166                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
167                 .get();
168         Assert.assertEquals(Status.NOT_FOUND, Status.fromStatusCode(response.getStatus()));
169         
170         // Check that files are deleted from FS
171         storedFile = DirectoryUtil.getStorageDirectory().resolve(file1Id);
172         Path webFile = DirectoryUtil.getStorageDirectory().resolve(file1Id + "_web");
173         Path thumbnailFile = DirectoryUtil.getStorageDirectory().resolve(file1Id + "_thumb");
174         Assert.assertFalse(Files.exists(storedFile));
175         Assert.assertFalse(Files.exists(webFile));
176         Assert.assertFalse(Files.exists(thumbnailFile));
177         
178         // Get all files from a document
179         json = target().path("/file/list")
180                 .queryParam("id", document1Id)
181                 .request()
182                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
183                 .get(JsonObject.class);
184         files = json.getJsonArray("files");
185         Assert.assertEquals(1, files.size());
186 
187         // Process a file
188         target().path("/file/" + file2Id + "/process").request()
189                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
190                 .post(Entity.form(new Form()), JsonObject.class);
191 
192         // Get all versions from a file
193         json = target().path("/file/" + file2Id + "/versions")
194                 .request()
195                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
196                 .get(JsonObject.class);
197         files = json.getJsonArray("files");
198         Assert.assertEquals(1, files.size());
199         JsonObject file = files.getJsonObject(0);
200         Assert.assertEquals(file2Id, file.getString("id"));
201         Assert.assertEquals("PIA00452.jpg", file.getString("name"));
202         Assert.assertEquals("image/jpeg", file.getString("mimetype"));
203         Assert.assertEquals(0, file.getInt("version"));
204 
205         // Add a new version to a file
206         String file3Id;
207         try (InputStream is0 = Resources.getResource("file/document.txt").openStream()) {
208             StreamDataBodyPart streamDataBodyPart = new StreamDataBodyPart("file", is0, "document.txt");
209             try (FormDataMultiPart multiPart = new FormDataMultiPart()) {
210                 json = target()
211                         .register(MultiPartFeature.class)
212                         .path("/file").request()
213                         .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
214                         .put(Entity.entity(
215                                 multiPart
216                                         .field("id", document1Id)
217                                         .field("previousFileId", file2Id)
218                                         .bodyPart(streamDataBodyPart),
219                                 MediaType.MULTIPART_FORM_DATA_TYPE), JsonObject.class);
220                 file3Id = json.getString("id");
221                 Assert.assertNotNull(file2Id);
222             }
223         }
224 
225         // Get all versions from a file
226         json = target().path("/file/" + file2Id + "/versions")
227                 .request()
228                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
229                 .get(JsonObject.class);
230         files = json.getJsonArray("files");
231         Assert.assertEquals(2, files.size());
232         file = files.getJsonObject(1);
233         Assert.assertEquals(file3Id, file.getString("id"));
234         Assert.assertEquals("document.txt", file.getString("name"));
235         Assert.assertEquals("text/plain", file.getString("mimetype"));
236         Assert.assertEquals(1, file.getInt("version"));
237 
238         // Delete the previous version
239         json = target().path("/file/" + file2Id).request()
240                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
241                 .delete(JsonObject.class);
242         Assert.assertEquals("ok", json.getString("status"));
243 
244         // Check the newly created version
245         json = target().path("/file/list")
246                 .queryParam("id", document1Id)
247                 .request()
248                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
249                 .get(JsonObject.class);
250         files = json.getJsonArray("files");
251         Assert.assertEquals(1, files.size());
252         Assert.assertEquals(file3Id, files.getJsonObject(0).getString("id"));
253         Assert.assertEquals("document.txt", files.getJsonObject(0).getString("name"));
254         Assert.assertEquals(1, files.getJsonObject(0).getInt("version"));
255     }

258     public void testFileResourceZip() throws Exception {
259         // Login file_resources
260         clientUtil.createUser("file_resources_zip");
261         String file1Token = clientUtil.login("file_resources_zip");
262 
263         // Create a document
264         String document1Id = clientUtil.createDocument(file1Token);
265 
266         // Add a file
267         String file1Id = clientUtil.addFileToDocument(FILE_PIA_00452_JPG, file1Token, document1Id);
268 
269         // Get a ZIP from all files of the document
270         Response response = target().path("/file/zip")
271                 .queryParam("id", document1Id)
272                 .request()
273                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
274                 .get();
275         Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
276         InputStream is = (InputStream) response.getEntity();
277         ZipInputStream zipInputStream = new ZipInputStream(is);
278         Assert.assertEquals(zipInputStream.getNextEntry().getName(), "0-PIA00452.jpg");
279         Assert.assertNull(zipInputStream.getNextEntry());
280 
281         // Fail if we don't have access to the document
282         response = target().path("/file/zip")
283                 .queryParam("id", document1Id)
284                 .request()
285                 .get();
286         Assert.assertEquals(Status.NOT_FOUND, Status.fromStatusCode(response.getStatus()));
287 
288         // Create a document
289         String document2Id = clientUtil.createDocument(file1Token);
290 
291         // Add a file
292         String file2Id = clientUtil.addFileToDocument(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG, file1Token, document2Id);
293 
294         // Get a ZIP from both files
295         response = target().path("/file/zip")
296                 .request()
297                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, file1Token)
298                 .post(Entity.form(new Form()
299                         .param("files", file1Id)
300                         .param("files", file2Id)));
301         Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
302         is = (InputStream) response.getEntity();
303         zipInputStream = new ZipInputStream(is);
304         Assert.assertNotNull(zipInputStream.getNextEntry().getName());
305         Assert.assertNotNull(zipInputStream.getNextEntry().getName());
306         Assert.assertNull(zipInputStream.getNextEntry());
307         
308         // Fail if we don't have access to the files
309         response = target().path("/file/zip")
310                 .request()
311                 .post(Entity.form(new Form()
312                         .param("files", file1Id)
313                         .param("files", file2Id)));
314         Assert.assertEquals(Status.FORBIDDEN, Status.fromStatusCode(response.getStatus()));
315     }

404      * Test user quota.

408     @Test
409     public void testQuota() throws Exception {
410         // Login file_quota
411         clientUtil.createUser("file_quota");
412         String fileQuotaToken = clientUtil.login("file_quota");
413         
414         // Add a file (292641 bytes large)
415         String file1Id = clientUtil.addFileToDocument(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG, fileQuotaToken, null);
416 
417         // Check current quota
418         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE, getUserQuota(fileQuotaToken));
419         
420         // Add a file (292641 bytes large)
421         clientUtil.addFileToDocument(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG, fileQuotaToken, null);
422         
423         // Check current quota
424         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE * 2, getUserQuota(fileQuotaToken));
425         
426         // Add a file (292641 bytes large)
427         clientUtil.addFileToDocument(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG, fileQuotaToken, null);
428         
429         // Check current quota
430         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE * 3, getUserQuota(fileQuotaToken));
431         
432         // Add a file (292641 bytes large)
433         try {
434             clientUtil.addFileToDocument(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG, fileQuotaToken, null);
435             Assert.fail();
436         } catch (jakarta.ws.rs.BadRequestException ignored) {
437         }
438         
439         // Deletes a file
440         JsonObject json = target().path("/file/" + file1Id).request()
441                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, fileQuotaToken)
442                 .delete(JsonObject.class);
443         Assert.assertEquals("ok", json.getString("status"));
444         
445         // Check current quota
446         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE * 2, getUserQuota(fileQuotaToken));
447 
448         // Create a document
449         long create1Date = new Date().getTime();
450         json = target().path("/document").request()
451                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, fileQuotaToken)
452                 .put(Entity.form(new Form()
453                         .param("title", "File test document 1")
454                         .param("language", "eng")
455                         .param("create_date", Long.toString(create1Date))), JsonObject.class);
456         String document1Id = json.getString("id");
457         Assert.assertNotNull(document1Id);
458 
459         // Add a file to this document (163510 bytes large)
460         clientUtil.addFileToDocument(FILE_PIA_00452_JPG, fileQuotaToken, document1Id);
461 
462         // Check current quota
463         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE * 2 + FILE_PIA_00452_JPG_SIZE, getUserQuota(fileQuotaToken));
464 
465         // Deletes the document
466         json = target().path("/document/" + document1Id).request()
467                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, fileQuotaToken)
468                 .delete(JsonObject.class);
469         Assert.assertEquals("ok", json.getString("status"));
470 
471         // Check current quota
472         Assert.assertEquals(FILE_EINSTEIN_ROOSEVELT_LETTER_PNG_SIZE * 2, getUserQuota(fileQuotaToken));
473     }

475     private long getUserQuota(String userToken) {
476         return target().path("/user").request()
477                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, userToken)
478                 .get(JsonObject.class).getJsonNumber("storage_current").longValue();
479     }
