18  public class TestAuditLogResource extends BaseJerseyTest {
19      /**
20       * Test the audit log resource.
21       *
22       * @throws Exception e
23       */
24      @Test
25      public void testAuditLogResource() throws Exception {
26          // Login auditlog1
27          clientUtil.createUser("auditlog1");
28          String auditlog1Token = clientUtil.login("auditlog1");
29          
30          // Create a tag
31          JsonObject json = target().path("/tag").request()
32                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
33                  .put(Entity.form(new Form()
34                          .param("name", "SuperTag")
35                          .param("color", "#ffff00")), JsonObject.class);
36          String tag1Id = json.getString("id");
37          Assert.assertNotNull(tag1Id);
38  
39          // Create a document
40          long create1Date = new Date().getTime();
41          json = target().path("/document").request()
42                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
43                  .put(Entity.form(new Form()
44                          .param("title", "My super title document 1")
45                          .param("description", "My super description for document 1")
46                          .param("tags", tag1Id)
47                          .param("language", "eng")
48                          .param("create_date", Long.toString(create1Date))), JsonObject.class);
49          String document1Id = json.getString("id");
50          Assert.assertNotNull(document1Id);
51          
52          // Get all logs for the document
53          json = target().path("/auditlog")
54                  .queryParam("document", document1Id)
55                  .request()
56                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
57                  .get(JsonObject.class);
58          JsonArray logs = json.getJsonArray("logs");
59          Assert.assertEquals(3, logs.size());
60          Assert.assertEquals(countByClass(logs, "Document"), 1);
61          Assert.assertEquals(countByClass(logs, "Acl"), 2);
62          Assert.assertEquals("auditlog1", logs.getJsonObject(0).getString("username"));
63          Assert.assertNotNull(logs.getJsonObject(0).getString("id"));
64          Assert.assertNotNull(logs.getJsonObject(0).getString("target"));
65          Assert.assertNotNull(logs.getJsonObject(0).getString("type"));
66          Assert.assertNotNull(logs.getJsonObject(0).getString("message"));
67          Assert.assertNotNull(logs.getJsonObject(0).getJsonNumber("create_date"));
68          Assert.assertEquals("auditlog1", logs.getJsonObject(1).getString("username"));
69          Assert.assertEquals("auditlog1", logs.getJsonObject(2).getString("username"));
70          
71          // Get all logs for the current user
72          json = target().path("/auditlog").request()
73                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
74                  .get(JsonObject.class);
75          logs = json.getJsonArray("logs");
76          Assert.assertEquals(2, logs.size());
77          Assert.assertEquals(countByClass(logs, "Document"), 1);
78          Assert.assertEquals(countByClass(logs, "Tag"), 1);
79          
80          // Deletes a tag
81          json = target().path("/tag/" + tag1Id).request()
82                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
83                  .delete(JsonObject.class);
84          Assert.assertEquals("ok", json.getString("status"));
85          
86          // Get all logs for the current user
87          json = target().path("/auditlog").request()
88                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
89                  .get(JsonObject.class);
90          logs = json.getJsonArray("logs");
91          Assert.assertEquals(3, logs.size());
92          Assert.assertEquals(countByClass(logs, "Document"), 1);
93          Assert.assertEquals(countByClass(logs, "Tag"), 2);
94  
95          // Get document 1
96          json = target().path("/document/" + document1Id).request()
97                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
98                  .get(JsonObject.class);
99          long update1Date = json.getJsonNumber("update_date").longValue();
100 
101         // Add a file to the document
102         clientUtil.addFileToDocument(FILE_WIKIPEDIA_PDF, auditlog1Token, document1Id);
103 
104         // Get document 1
105         json = target().path("/document/" + document1Id).request()
106                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
107                 .get(JsonObject.class);
108         Assert.assertTrue(json.getJsonNumber("update_date").longValue() > update1Date); // Adding a file to a document updates it
109 
110         // Get all logs for the document
111         json = target().path("/auditlog")
112                 .queryParam("document", document1Id)
113                 .request()
114                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, auditlog1Token)
115                 .get(JsonObject.class);
116         logs = json.getJsonArray("logs");
117         Assert.assertEquals(4, logs.size());
118         Assert.assertEquals(countByClass(logs, "Document"), 1);
119         Assert.assertEquals(countByClass(logs, "Acl"), 2);
120         Assert.assertEquals(countByClass(logs, "File"), 1);
121 
122         // Delete auditlog1
123         String adminToken = adminToken();
124         target().path("/user/auditlog1").request()
125                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
126                 .delete();
127     }

129     /**
130      * Count logs by class.
131      * 
132      * @param logs Logs
133      * @param clazz Class
134      * @return Count by class
135      */
136     private int countByClass(JsonArray logs, String clazz) {
137         int count = 0;
138         for (int i = 0; i < logs.size(); i++) {
139             if (logs.getJsonObject(i).getString("class").equals(clazz)) {
140                 count++;
141             }
142         }
143         return count;
144     }
