7   import jakarta.json.JsonArray;

9   import jakarta.ws.rs.client.Entity;
10  import jakarta.ws.rs.core.Form;

13  /**
14   * Test the metadata resource.
15   *
16   * @author bgamard
17   */
18  public class TestMetadataResource extends BaseJerseyTest {
19      /**
20       * Test the metadata resource.
21       */
22      @Test
23      public void testMetadataResource() {
24          // Login admin
25          String adminToken = adminToken();
26  
27          // Get all metadata with admin
28          JsonObject json = target().path("/metadata")
29                  .queryParam("sort_column", "2")
30                  .queryParam("asc", "false")
31                  .request()
32                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
33                  .get(JsonObject.class);
34          JsonArray metadata = json.getJsonArray("metadata");
35          Assert.assertEquals(0, metadata.size());
36  
37          // Create a metadata with admin
38          json = target().path("/metadata").request()
39                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
40                  .put(Entity.form(new Form()
41                          .param("name", "ISBN 13")
42                          .param("type", "STRING")), JsonObject.class);
43          String metadataIsbnId = json.getString("id");
44          Assert.assertNotNull(metadataIsbnId);
45          Assert.assertEquals("ISBN 13", json.getString("name"));
46          Assert.assertEquals("STRING", json.getString("type"));
47  
48          // Get all metadata with admin
49          json = target().path("/metadata")
50                  .queryParam("sort_column", "2")
51                  .queryParam("asc", "false")
52                  .request()
53                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
54                  .get(JsonObject.class);
55          metadata = json.getJsonArray("metadata");
56          Assert.assertEquals(1, metadata.size());
57  
58          // Update a metadata with admin
59          json = target().path("/metadata/" + metadataIsbnId).request()
60                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
61                  .post(Entity.form(new Form()
62                          .param("name", "ISBN 10")), JsonObject.class);
63          Assert.assertEquals(metadataIsbnId, json.getString("id"));
64          Assert.assertEquals("ISBN 10", json.getString("name"));
65          Assert.assertEquals("STRING", json.getString("type"));
66  
67          // Delete a metadata with admin
68          target().path("/metadata/" + metadataIsbnId).request()
69                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
70                  .delete(JsonObject.class);
71  
72          // Get all metadata with admin
73          json = target().path("/metadata")
74                  .queryParam("sort_column", "2")
75                  .queryParam("asc", "false")
76                  .request()
77                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
78                  .get(JsonObject.class);
79          metadata = json.getJsonArray("metadata");
80          Assert.assertEquals(0, metadata.size());
81      }
