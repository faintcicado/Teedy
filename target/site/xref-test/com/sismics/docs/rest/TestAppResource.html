45      @Test
46      public void testAppResource() {
47          // Login admin
48          String adminToken = adminToken();
49  
50          // Check the application info
51          JsonObject json = target().path("/app").request()
52                  .get(JsonObject.class);
53          Assert.assertNotNull(json.getString("current_version"));
54          Assert.assertNotNull(json.getString("min_version"));
55          Long freeMemory = json.getJsonNumber("free_memory").longValue();
56          Assert.assertTrue(freeMemory > 0);
57          Long totalMemory = json.getJsonNumber("total_memory").longValue();
58          Assert.assertTrue(totalMemory > 0 && totalMemory > freeMemory);
59          Assert.assertEquals(0, json.getJsonNumber("queued_tasks").intValue());
60          Assert.assertFalse(json.getBoolean("guest_login"));
61          Assert.assertFalse(json.getBoolean("ocr_enabled"));
62          Assert.assertEquals("eng", json.getString("default_language"));
63          Assert.assertTrue(json.containsKey("global_storage_current"));
64          Assert.assertTrue(json.getJsonNumber("active_user_count").longValue() > 0);
65  
66          // Rebuild Lucene index
67          Response response = target().path("/app/batch/reindex").request()
68                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
69                  .post(Entity.form(new Form()));
70          Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
71  
72          // Clean storage
73          response = target().path("/app/batch/clean_storage").request()
74                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
75                  .post(Entity.form(new Form()));
76          Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
77  
78          // Change the default language
79          response = target().path("/app/config").request()
80                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
81                  .post(Entity.form(new Form().param("default_language", "fra")));
82          Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
83  
84          // Check the application info
85          json = target().path("/app").request()
86                  .get(JsonObject.class);
87          Assert.assertEquals("fra", json.getString("default_language"));
88  
89          // Change the default language
90          response = target().path("/app/config").request()
91                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
92                  .post(Entity.form(new Form().param("default_language", "eng")));
93          Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
94  
95          // Check the application info
96          json = target().path("/app").request()
97                  .get(JsonObject.class);
98          Assert.assertEquals("eng", json.getString("default_language"));
99      }

138     @Test
139     public void testGuestLogin() {
140         // Login admin
141         String adminToken = adminToken();
142 
143         // Try to login as guest
144         Response response = target().path("/user/login").request()
145                 .post(Entity.form(new Form()
146                         .param("username", "guest")));
147         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), response.getStatus());
148 
149         // Enable guest login
150         target().path("/app/guest_login").request()
151                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
152                 .post(Entity.form(new Form()
153                         .param("enabled", "true")), JsonObject.class);
154 
155         // Login as guest
156         String guestToken = clientUtil.login("guest", "", false);
157 
158         // Guest cannot delete himself
159         response = target().path("/user").request()
160                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
161                 .delete();
162         Assert.assertEquals(Status.BAD_REQUEST.getStatusCode(), response.getStatus());
163 
164         // Guest cannot see opened sessions
165         JsonObject json = target().path("/user/session").request()
166                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
167                 .get(JsonObject.class);
168         Assert.assertEquals(0, json.getJsonArray("sessions").size());
169 
170         // Guest cannot delete opened sessions
171         response = target().path("/user/session").request()
172                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
173                 .delete();
174         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), response.getStatus());
175 
176         // Guest cannot enable TOTP
177         response = target().path("/user/enable_totp").request()
178                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
179                 .post(Entity.form(new Form()));
180         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), response.getStatus());
181 
182         // Guest cannot disable TOTP
183         response = target().path("/user/disable_totp").request()
184                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
185                 .post(Entity.form(new Form()));
186         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), response.getStatus());
187 
188         // Guest cannot update itself
189         response = target().path("/user").request()
190                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
191                 .post(Entity.form(new Form()));
192         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), response.getStatus());
193 
194         // Guest can see its documents
195         target().path("/document/list").request()
196                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, guestToken)
197                 .get(JsonObject.class);
198 
199         // Disable guest login (clean up state)
200         target().path("/app/guest_login").request()
201                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
202                 .post(Entity.form(new Form()
203                         .param("enabled", "false")), JsonObject.class);
204     }

400     /**
401      * Test the LDAP authentication.
402      */
403     @Test
404     public void testLdapAuthentication() throws Exception {
405         // Start LDAP server
406         final DirectoryServiceFactory factory = new DefaultDirectoryServiceFactory();
407         factory.init("Test");
408 
409         final DirectoryService directoryService = factory.getDirectoryService();
410         directoryService.getChangeLog().setEnabled(false);
411         directoryService.setShutdownHookEnabled(true);
412 
413         final Partition partition = new AvlPartition(directoryService.getSchemaManager());
414         partition.setId("Test");
415         partition.setSuffixDn(new Dn(directoryService.getSchemaManager(), "o=TEST"));
416         partition.initialize();
417         directoryService.addPartition(partition);
418 
419         final LdapServer ldapServer = new LdapServer();
420         ldapServer.setTransports(new TcpTransport("localhost", 11389));
421         ldapServer.setDirectoryService(directoryService);
422 
423         directoryService.startup();
424         ldapServer.start();
425 
426         // Load test data in LDAP
427         new LdifFileLoader(directoryService.getAdminSession(), new File(Resources.getResource("test.ldif").getFile()), null).execute();
428 
429         // Login admin
430         String adminToken = adminToken();
431 
432         // Get the LDAP configuration
433         JsonObject json = target().path("/app/config_ldap").request()
434                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
435                 .get(JsonObject.class);
436         if (!configLdapChanged) {
437                 Assert.assertFalse(json.getBoolean("enabled"));
438         }
439 
440         // Change LDAP configuration
441         target().path("/app/config_ldap").request()
442                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
443                 .post(Entity.form(new Form()
444                         .param("enabled", "true")
445                         .param("host", "localhost")
446                         .param("port", "11389")
447                         .param("usessl", "false")
448                         .param("admin_dn", "uid=admin,ou=system")
449                         .param("admin_password", "secret")
450                         .param("base_dn", "o=TEST")
451                         .param("filter", "(&(objectclass=inetOrgPerson)(uid=USERNAME))")
452                         .param("default_email", "devnull@teedy.io")
453                         .param("default_storage", "100000000")
454                 ), JsonObject.class);
455         configLdapChanged = true;
456 
457         // Get the LDAP configuration
458         json = target().path("/app/config_ldap").request()
459                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
460                 .get(JsonObject.class);
461         Assert.assertTrue(json.getBoolean("enabled"));
462         Assert.assertEquals("localhost", json.getString("host"));
463         Assert.assertEquals(11389, json.getJsonNumber("port").intValue());
464         Assert.assertEquals("uid=admin,ou=system", json.getString("admin_dn"));
465         Assert.assertEquals("secret", json.getString("admin_password"));
466         Assert.assertEquals("o=TEST", json.getString("base_dn"));
467         Assert.assertEquals("(&(objectclass=inetOrgPerson)(uid=USERNAME))", json.getString("filter"));
468         Assert.assertEquals("devnull@teedy.io", json.getString("default_email"));
469         Assert.assertEquals(100000000L, json.getJsonNumber("default_storage").longValue());
470 
471         // Login with a LDAP user
472         String ldapTopen = clientUtil.login("ldap1", "secret", false);
473 
474         // Check user informations
475         json = target().path("/user").request()
476                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, ldapTopen)
477                 .get(JsonObject.class);
478         Assert.assertEquals("ldap1@teedy.io", json.getString("email"));
479 
480         // List all documents
481         json = target().path("/document/list")
482                 .queryParam("sort_column", 3)
483                 .queryParam("asc", true)
484                 .request()
485                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, ldapTopen)
486                 .get(JsonObject.class);
487         JsonArray documents = json.getJsonArray("documents");
488         Assert.assertEquals(0, documents.size());
489 
490         // Stop LDAP server
491         ldapServer.stop();
492         directoryService.shutdown();
493     }
