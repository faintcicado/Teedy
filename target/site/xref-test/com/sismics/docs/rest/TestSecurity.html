26      @Test
27      public void testSecurity() {
28          // Create a user
29          clientUtil.createUser("testsecurity");
30  
31          // Changes a user's email KO : the user is not connected
32          Response response = target().path("/user").request()
33                  .post(Entity.form(new Form().param("email", "testsecurity2@docs.com")));
34          Assert.assertEquals(Status.FORBIDDEN, Status.fromStatusCode(response.getStatus()));
35          JsonObject json = response.readEntity(JsonObject.class);
36          Assert.assertEquals("ForbiddenError", json.getString("type"));
37          Assert.assertEquals("You don't have access to this resource", json.getString("message"));
38  
39          // User testsecurity logs in
40          String testSecurityToken = clientUtil.login("testsecurity");
41          
42          // User testsecurity creates a new user KO : no permission
43          response = target().path("/user").request()
44                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, testSecurityToken)
45                  .put(Entity.form(new Form()));
46          Assert.assertEquals(Status.FORBIDDEN, Status.fromStatusCode(response.getStatus()));
47          Assert.assertEquals("ForbiddenError", json.getString("type"));
48          Assert.assertEquals("You don't have access to this resource", json.getString("message"));
49          
50          // User testsecurity changes his email OK
51          json = target().path("/user").request()
52                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, testSecurityToken)
53                  .post(Entity.form(new Form()
54                          .param("email", "testsecurity2@docs.com")), JsonObject.class);
55          Assert.assertEquals("ok", json.getString("status"));
56  
57          // User testsecurity logs out
58          response = target().path("/user/logout").request()
59                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, testSecurityToken)
60                  .post(Entity.form(new Form()));
61          Assert.assertEquals(Status.OK, Status.fromStatusCode(response.getStatus()));
62          testSecurityToken = clientUtil.getAuthenticationCookie(response);
63          Assert.assertTrue(StringUtils.isEmpty(testSecurityToken));
64  
65          // User testsecurity logs out KO : he is not connected anymore
66          response = target().path("/user/logout").request()
67                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, testSecurityToken)
68                  .post(Entity.form(new Form()));
69          Assert.assertEquals(Status.FORBIDDEN, Status.fromStatusCode(response.getStatus()));
70  
71          // User testsecurity logs in with a long lived session
72          testSecurityToken = clientUtil.login("testsecurity", "12345678", true);
73  
74          // User testsecurity logs out
75          clientUtil.logout(testSecurityToken);
76  
77          // Delete the user
78          String adminToken = adminToken();
79          target().path("/user/testsecurity").request()
80                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
81                  .delete();
82      }

84      @Test
85      public void testHeaderBasedAuthentication() {
86          clientUtil.createUser("header_auth_test");
87  
88          Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), target()
89                  .path("/user/session")
90                  .request()
91                  .get()
92                  .getStatus());
93  
94          Assert.assertEquals(Status.OK.getStatusCode(), target()
95                  .path("/user/session")
96                  .request()
97                  .header(HeaderBasedSecurityFilter.AUTHENTICATED_USER_HEADER, "header_auth_test")
98                  .get()
99                  .getStatus());
100 
101         Assert.assertEquals(Status.FORBIDDEN.getStatusCode(), target()
102                 .path("/user/session")
103                 .request()
104                 .header(HeaderBasedSecurityFilter.AUTHENTICATED_USER_HEADER, "idontexist")
105                 .get()
106                 .getStatus());
107         
108         // Delete the user
109         String adminToken = adminToken();
110         target().path("/user/header_auth_test").request()
111                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
112                 .delete();
113     }
