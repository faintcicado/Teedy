435     /**
436      * Search documents and returns the number found.
437      * 
438      * @param query Search query
439      * @param token Authentication token
440      * @return Number of documents found
441      */
442     private int searchDocuments(String query, String token) {
443         JsonObject json = target().path("/document/list")
444                 .queryParam("search", query)
445                 .request()
446                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, token)
447                 .get(JsonObject.class);
448         return json.getJsonArray("documents").size();
449     }
