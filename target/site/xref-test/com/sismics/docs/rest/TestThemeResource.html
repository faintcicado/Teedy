4   import com.sismics.util.filter.TokenBasedSecurityFilter;
5   import org.glassfish.jersey.media.multipart.FormDataMultiPart;
6   import org.glassfish.jersey.media.multipart.MultiPartFeature;
7   import org.glassfish.jersey.media.multipart.file.StreamDataBodyPart;
8   import org.junit.Assert;
9   import org.junit.Test;
10  
11  import jakarta.json.JsonObject;
12  import jakarta.ws.rs.client.Entity;
13  import jakarta.ws.rs.core.Form;
14  import jakarta.ws.rs.core.MediaType;
15  import jakarta.ws.rs.core.Response;
16  import java.io.InputStream;
17  
18  /**
19   * Test the theme resource.
20   * 
21   * @author bgamard
22   */
23  public class TestThemeResource extends BaseJerseyTest {
24      /**
25       * Test the theme resource.
26       */
27      @Test
28      public void testThemeResource() throws Exception {
29          // Login admin
30          String adminToken = adminToken();
31  
32          // Get the stylesheet anonymously
33          String stylesheet = target().path("/theme/stylesheet").request()
34                  .get(String.class);
35          Assert.assertTrue(stylesheet.contains("background-color: #ffffff;"));
36  
37          // Get the theme configuration anonymously
38          JsonObject json = target().path("/theme").request()
39                  .get(JsonObject.class);
40          Assert.assertEquals("Teedy", json.getString("name"));
41          Assert.assertEquals("#ffffff", json.getString("color"));
42          Assert.assertEquals("", json.getString("css"));
43  
44          // Update the main color as admin
45          target().path("/theme").request()
46                  .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
47                  .post(Entity.form(new Form()
48                          .param("color", "#ff0000")
49                  .param("name", "My App")
50                  .param("css", ".body { content: 'Custom CSS'; }")), JsonObject.class);
51  
52          // Get the stylesheet anonymously
53          stylesheet = target().path("/theme/stylesheet").request()
54                  .get(String.class);
55          Assert.assertTrue(stylesheet.contains("background-color: #ff0000;"));
56          Assert.assertTrue(stylesheet.contains("Custom CSS"));
57  
58          // Get the theme configuration anonymously
59          json = target().path("/theme").request()
60                  .get(JsonObject.class);
61          Assert.assertEquals("My App", json.getString("name"));
62          Assert.assertEquals("#ff0000", json.getString("color"));
63          Assert.assertEquals(".body { content: 'Custom CSS'; }", json.getString("css"));
64  
65          // Get the logo
66          Response response = target().path("/theme/image/logo").request().get();
67          Assert.assertEquals(Response.Status.OK.getStatusCode(), response.getStatus());
68  
69          // Get the background
70          response = target().path("/theme/image/background").request().get();
71          Assert.assertEquals(Response.Status.OK.getStatusCode(), response.getStatus());
72  
73          // Change the logo
74          try (InputStream is = Resources.getResource("file/PIA00452.jpg").openStream()) {
75              StreamDataBodyPart streamDataBodyPart = new StreamDataBodyPart("image", is, "PIA00452.jpg");
76              try (FormDataMultiPart multiPart = new FormDataMultiPart()) {
77                  target()
78                          .register(MultiPartFeature.class)
79                          .path("/theme/image/logo").request()
80                          .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
81                          .put(Entity.entity(multiPart.bodyPart(streamDataBodyPart),
82                                  MediaType.MULTIPART_FORM_DATA_TYPE), JsonObject.class);
83              }
84          }
85  
86          // Change the background
87          try (InputStream is = Resources.getResource("file/Einstein-Roosevelt-letter.png").openStream()) {
88              StreamDataBodyPart streamDataBodyPart = new StreamDataBodyPart("image", is, "Einstein-Roosevelt-letter.png");
89              try (FormDataMultiPart multiPart = new FormDataMultiPart()) {
90                  target()
91                          .register(MultiPartFeature.class)
92                          .path("/theme/image/background").request()
93                          .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
94                          .put(Entity.entity(multiPart.bodyPart(streamDataBodyPart),
95                                  MediaType.MULTIPART_FORM_DATA_TYPE), JsonObject.class);
96              }
97          }
98  
99          // Get the logo
100         response = target().path("/theme/image/logo").request().get();
101         Assert.assertEquals(Response.Status.OK.getStatusCode(), response.getStatus());
102 
103         // Get the background
104         response = target().path("/theme/image/background").request().get();
105         Assert.assertEquals(Response.Status.OK.getStatusCode(), response.getStatus());
106 
107         // Reset the main color as admin
108         target().path("/theme").request()
109                 .cookie(TokenBasedSecurityFilter.COOKIE_NAME, adminToken)
110                 .post(Entity.form(new Form()
111                         .param("color", "#ffffff")
112                 .param("name", "Teedy")
113                 .param("css", "")), JsonObject.class);
114     }
115 }
